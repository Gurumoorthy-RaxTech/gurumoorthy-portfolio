Generic Repository Pattern

Medium
GenericsInterfaceRepository Pattern

Create a generic IRepository<T> interface with GetById, GetAll, Add, Update, Delete methods. Then implement it for a Student entity.

// Define the interface and implement for Student
// YOUR CODE HERE
👁 Show Answer 
public interface IRepository<T> where T : class
{
    Task<T> GetByIdAsync(int id);
    Task<IEnumerable<T>> GetAllAsync();
    Task AddAsync(T entity);
    Task UpdateAsync(T entity);
    Task DeleteAsync(int id);
}

public class StudentRepository : IRepository<Student>
{
    private readonly AppDbContext _context;

    public StudentRepository(AppDbContext context) => _context = context;

    public async Task<Student> GetByIdAsync(int id) =>
        await _context.Students.FindAsync(id);

    public async Task<IEnumerable<Student>> GetAllAsync() =>
        await _context.Students.ToListAsync();

    public async Task AddAsync(Student s)
    {
        await _context.Students.AddAsync(s);
        await _context.SaveChangesAsync();
    }

    public async Task UpdateAsync(Student s)
    {
        _context.Students.Update(s);
        await _context.SaveChangesAsync();
    }

    public async Task DeleteAsync(int id)
    {
        var s = await _context.Students.FindAsync(id);
        if (s != null) { _context.Students.Remove(s); await _context.SaveChangesAsync(); }
    }
}

Shape Area Calculator (Polymorphism)

Easy
InheritanceAbstractPolymorphism

Create an abstract Shape class with an abstract CalculateArea() method. Implement Circle, Rectangle, and Triangle. Print areas using a common method.

// Design using abstract class and override
// Circle: π × r²
// Rectangle: width × height
// Triangle: 0.5 × base × height
// YOUR CODE HERE
👁 Show Answer 
public abstract class Shape
{
    public abstract double CalculateArea();
    public virtual void PrintArea() =>
        Console.WriteLine($"{GetType().Name}: Area = {CalculateArea():F2}");
}

public class Circle : Shape
{
    private double _radius;
    public Circle(double r) => _radius = r;
    public override double CalculateArea() => Math.PI * _radius * _radius;
}

public class Rectangle : Shape
{
    private double _w, _h;
    public Rectangle(double w, double h) { _w = w; _h = h; }
    public override double CalculateArea() => _w * _h;
}

public class Triangle : Shape
{
    private double _b, _h;
    public Triangle(double b, double h) { _b = b; _h = h; }
    public override double CalculateArea() => 0.5 * _b * _h;
}

// Usage — polymorphism in action
List<Shape> shapes = new List<Shape>
{
    new Circle(5), new Rectangle(4, 6), new Triangle(3, 8)
};
shapes.ForEach(s => s.PrintArea());

// ❌ BAD — violates SRP (one class doing too much)
public class StudentManager
{

// O — Open/Closed: open for extension, closed for modification
// L — Liskov Substitution: derived class usable as base class
// I — Interface Segregation: small focused interfaces
// D — Dependency Inversion: depend on abstractions, not concretions

Find departments that have more than 2 employees. Show department name and employee count.

-- Tables:
-- Department(Id, Name)

-- YOUR QUERY HERE
👁 Show Answer 
SELECT d.Name AS Department, COUNT(e.Id) AS EmployeeCount
FROM Department d
INNER JOIN Employee e ON e.DeptId = d.Id
GROUP BY d.Id, d.Name

Top Earner Per Department

Medium

Find the highest-paid employee in each department. Show employee name, department, and salary.

👁 Show Answer 
SELECT Name, Department, Salary
FROM (
    SELECT
        e.Name,
        d.Name AS Department,
        e.Salary,
        ROW_NUMBER() OVER (
            PARTITION BY e.DeptId
            ORDER BY e.Salary DESC
        ) AS Rnk
    FROM Employee e
    JOIN Department d ON d.Id = e.DeptId
) ranked
WHERE Rnk = 1;

Delete Duplicate Records

Hard
CTEROW_NUMBERDELETE

An Employee table has duplicate records (same Name + DeptId). Keep only the record with the lowest ID. Delete the rest.

👁 Show Answer 
WITH CTE AS (
    SELECT Id,
           ROW_NUMBER() OVER (
               PARTITION BY Name, DeptId
               ORDER BY Id
           ) AS RowNum
    FROM Employee
)
DELETE FROM CTE WHERE RowNum > 1;

-- This keeps the first occurrence (lowest Id) and removes duplicates

[ApiController]
[Route("api/[controller]")]
public class StudentsController : ControllerBase
{
    private readonly IStudentService _service;
    public StudentsController(IStudentService service) => _service = service;

    [HttpGet]
    public async Task<IActionResult> GetAll()

    public async Task<IActionResult> GetById(int id)

        if (student == null) return NotFound(); // 404
        return Ok(student); // 200
    }

    public async Task<IActionResult> Create([FromBody] Student student)
    {
        if (!ModelState.IsValid) return BadRequest(ModelState); // 400

        return CreatedAtAction(nameof(GetById), new { id = created.Id }, created); // 201
    }

    [HttpPut("{id}")]
    public async Task<IActionResult> Update(int id, [FromBody] Student student)
    {
        if (id != student.Id) return BadRequest(); // 400
        await _service.UpdateAsync(student);
        return NoContent(); // 204
    }

    [HttpDelete("{id}")]
    public async Task<IActionResult> Delete(int id)
    {
        await _service.DeleteAsync(id);
        return NoContent(); // 204
    }
}

Middleware — Exception Handling

Medium
MiddlewareExceptionPipeline

Write a global exception handling middleware for ASP.NET Core Web API that returns JSON error responses.

👁 Show Answer 
public class ExceptionMiddleware
{
    private readonly RequestDelegate _next;

    public ExceptionMiddleware(RequestDelegate next) => _next = next;

    public async Task InvokeAsync(HttpContext context)
    {
        try
        {
            await _next(context);
        }
        catch (KeyNotFoundException ex)
        {
            context.Response.StatusCode = 404;
            await context.Response.WriteAsJsonAsync(new { error = ex.Message });
        }
        catch (Exception ex)
        {
            context.Response.StatusCode = 500;
            await context.Response.WriteAsJsonAsync(new { error = "Internal server error" });
        }
    }
}

// Register in Program.cs
app.UseMiddleware<ExceptionMiddleware>();

Reverse a String (Without Built-in)

Easy
StringArray

Reverse a string without using string.Reverse(). Input: "Hello" → Output: "olleH"

👁 Show Answer 
public string Reverse(string s)
{
    char[] chars = s.ToCharArray();
    int left = 0, right = chars.Length - 1;
    while (left < right)
    {
        (chars[left], chars[right]) = (chars[right], chars[left]);
        left++; right--;
    }
    return new string(chars);
}

// Time: O(n) | Space: O(n)

Check Palindrome

Easy
StringTwo Pointer

Check if a string is a palindrome (ignore case and spaces). "Race Car" → true, "Hello" → false

👁 Show Answer 
public bool IsPalindrome(string s)
{
    s = new string(s.ToLower()
                     .Where(char.IsLetterOrDigit)
                     .ToArray());
    int left = 0, right = s.Length - 1;
    while (left < right)
    {
        if (s[left] != s[right]) return false;
        left++; right--;
    }
    return true;
}

// "Race Car" → cleaned → "racecar" → true

Find Missing Number in 1..N

Easy
MathArray

Array contains 1 to N with one number missing. Find the missing number. Input: [1,2,4,5] N=5 → Output: 3

👁 Show Answer 
public int FindMissing(int[] nums, int n)
{
    // Sum formula: n*(n+1)/2
    int expected = n * (n + 1) / 2;
    int actual = nums.Sum();
    return expected - actual;
}

// Time: O(n) | Space: O(1) — optimal!

Fibonacci — Iterative & Recursive

Medium
RecursionDynamic Programming

Return the Nth Fibonacci number. Then explain why iterative is better than naive recursion.

👁 Show Answer 
// Recursive — O(2^n) time, NOT recommended for large N
public int FibRecursive(int n)
{
    if (n <= 1) return n;
    return FibRecursive(n - 1) + FibRecursive(n - 2);
}

// Iterative — O(n) time, O(1) space — PREFERRED
public int FibIterative(int n)
{
    if (n <= 1) return n;
    int a = 0, b = 1;
    for (int i = 2; i <= n; i++)
    {
        int temp = a + b;
        a = b;
        b = temp;
    }
    return b;
}

// Memo (best of both worlds) — O(n) time + space
private Dictionary<int, int> _memo = new();
public int FibMemo(int n)
{
    if (n <= 1) return n;
    if (_memo.ContainsKey(n)) return _memo[n];
    _memo[n] = FibMemo(n-1) + FibMemo(n-2);
    return _memo[n];
}
