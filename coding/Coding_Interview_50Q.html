2Check if a string is a PalindromeC#▼
bool IsPalindrome(string s)
{
    s = s.ToLower();
    int left = 0, right = s.Length - 1;
    while (left < right)
    {
        if (s[left] != s[right]) return false;
        left++; right--;
    }
    return true;
}

bool IsAnagram(string a, string b)
{
    if (a.Length != b.Length) return false;
    var sorted_a = string.Concat(a.ToLower().OrderBy(c => c));
    var sorted_b = string.Concat(b.ToLower().OrderBy(c => c));
    return sorted_a == sorted_b;
}

11Find the 2nd highest number in an arrayC#▼
int SecondHighest(int[] arr)
{
    int first = int.MinValue, second = int.MinValue;
    foreach (int n in arr)
    {
        if (n > first) { second = first; first = n; }
        else if (n > second && n != first) second = n;
    }
    return second;
}

{

    if (n % 2 == 0) return false;

16Reverse an integer without converting to stringC#▼
int ReverseInt(int n)
{
    int reversed = 0;
    while (n != 0)
    {
        int digit = n % 10;      // get last digit
        reversed = reversed * 10 + digit;
        n /= 10;               // remove last digit
    }
    return reversed;
}

Console.WriteLine(ReverseInt(12345)); // 54321

18Sort an array without using built-in Sort (Bubble Sort)C#▼
void BubbleSort(int[] arr)
{
    int n = arr.Length;
    for (int i = 0; i < n - 1; i++)
    {
        for (int j = 0; j < n - i - 1; j++)
        {
            if (arr[j] > arr[j + 1])
            {
                // swap
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int[] arr = { 5,2,8,1,9 };
BubbleSort(arr);

19Find missing number in array 1 to NC#▼
int FindMissing(int[] arr, int n)
{
    // Sum formula: n*(n+1)/2 - actual sum = missing
    int expected = n * (n + 1) / 2;
    int actual = arr.Sum();
    return expected - actual;
}

Console.WriteLine(FindMissing(arr, 6));

20Check if a number is Armstrong number (e.g. 153 = 1³+5³+3³)C#▼
bool IsArmstrong(int n)
{
    string s = n.ToString();
    int power = s.Length;
    int sum = s.Sum(c => (int)Math.Pow(c - '0', power));
    return sum == n;
}

    SUM(CASE WHEN a.Status = 'Present'

        SUM(CASE WHEN a.Status = 'Present'

41Implement a Stack using an array (Push, Pop, Peek)C#▼
public class MyStack
{
    private int[] _data;
    private int _top = -1;

    public MyStack(int size) => _data = new int[size];

    public void Push(int val)
    {
        if (_top == _data.Length - 1) throw new Exception("Stack full");
        _data[++_top] = val;
    }

    public int Pop()
    {
        if (_top == -1) throw new Exception("Stack empty");
        return _data[_top--];
    }

    public int Peek() => _top == -1
        ? throw new Exception("Empty") : _data[_top];

    public bool IsEmpty => _top == -1;
}

var s = new MyStack(5);
s.Push(10); s.Push(20); s.Push(30);
Console.WriteLine(s.Pop());  // 30
Console.WriteLine(s.Peek()); // 20
42Singleton Design Pattern – only one DB connection instanceC#▼
public sealed class DatabaseConnection
{
    private static DatabaseConnection _instance;
    private static readonly object _lock = new object();

    private DatabaseConnection()
    {
        Console.WriteLine("DB Connection created");
    }

    public static DatabaseConnection Instance
    {
        get
        {
            if (_instance == null)
            {
                lock (_lock)  // thread-safe
                {
                    if (_instance == null)
                        _instance = new DatabaseConnection();
                }
            }
            return _instance;
        }
    }

    public void Query(string sql)
        => Console.WriteLine($"Executing: {sql}");
}

// Usage
DatabaseConnection.Instance.Query("SELECT * FROM Students");
DatabaseConnection.Instance.Query("SELECT * FROM Bills");
// "DB Connection created" prints only ONCE

    private bool _disposed = false;

// O(n) solution using HashSet
(int, int) TwoSum(int[] arr, int target)
{
    var seen = new HashSet<int>();
    foreach (int n in arr)
    {
        int complement = target - n;
        if (seen.Contains(complement))
            return (complement, n);
        seen.Add(n);
    }
    return (-1, -1);
}
