var students = new List<Student> { s1, s2, s3, s4, s5 };

// for loop — when you need index
for (int i = 0; i < students.Count; i++)
{
    Console.WriteLine($"Student {i + 1}: {students[i].Name}");

// foreach — most common, cleaner, no index needed
foreach (var student in students)
{

// while — condition-based
int page = 1;
while (true)
{
    var batch = GetStudents(schoolId: 1, page: page, size: 100);
    if (!batch.Any()) break;         // exit when no more data
    ProcessStudents(batch);
    page++;
}

// do-while — executes AT LEAST ONCE
int attempts = 0;
do
{
    attempts++;
    var result = TrySendSms(phone, message);
    if (result.IsSuccess) break;
} while (attempts < 3);  // retry up to 3 times

// break — exit loop immediately
// continue — skip current iteration, go to next
foreach (var student in students)
{
    if (!student.IsActive) continue;  // skip inactive
    if (student.FeeAmount == 0) break;  // stop at first free student
    ProcessFee(student);
}

📦 Collections — Store Groups of Data

List<T> — Most Used Collection

var students = new List<Student>();  // empty list

// Add
students.Add(new Student { Name = "Basha" });
students.AddRange(GetNewStudents()); // add many

// Access
var first = students[0];              // by index (0-based)
int count = students.Count;           // total count

// Remove
students.Remove(student1);            // by object
students.RemoveAt(0);               // by index
students.RemoveAll(s => !s.IsActive); // by condition

// Check
students.Contains(student1);          // true/false
students.Any(s => s.Name == "Basha"); // any match?
students.All(s => s.IsActive);       // all match?

// Sort
students.Sort((a, b) => a.Name.CompareTo(b.Name));
var sorted = students.OrderBy(s => s.Name).ToList();  // LINQ way

// Collection initializer syntax
var classIds = new List<int> { 1, 2, 3, 4, 5 };
var names = new List<string>(["Basha", "Priya"]); // C# 12

Array — Fixed Size

// Array — fixed size, faster than List for fixed data
int[] classIds = new int[5];    // 5 elements, all 0
int[] ids = { 1, 2, 3, 4, 5 };  // initialize
string[] terms = ["Term1", "Term2", "Term3"];  // C# 12 collection expression

ids.Length;         // 5
ids[0] = 10;         // modify element
Array.Sort(ids);    // sort in-place
Array.Reverse(ids); // reverse
var list = ids.ToList(); // convert to List

🗂️ Dictionary — Key-Value Storage

// Dictionary<TKey, TValue> — O(1) lookup by key
var studentMap = new Dictionary<string, Student>();

// Add
studentMap.Add("ADM001", basha);        // throws if key exists
studentMap["ADM002"] = priya;           // add or update
studentMap.TryAdd("ADM001", basha);     // no throw if exists

// Lookup
var s = studentMap["ADM001"];           // throws KeyNotFoundException if missing

// Safe lookup
if (studentMap.TryGetValue("ADM001", out var found))
    Console.WriteLine(found.Name);

// Check existence
studentMap.ContainsKey("ADM001");  // true/false

// Remove
studentMap.Remove("ADM001");

// Iterate
foreach (var kvp in studentMap)
    Console.WriteLine($"Key: {kvp.Key}, Student: {kvp.Value.Name}");

// Real ZenCampus use: group students by class
var byClass = new Dictionary<int, List<Student>>();
foreach (var s in students)
{
    if (!byClass.ContainsKey(s.ClassId))
        byClass[s.ClassId] = new List<Student>();
    byClass[s.ClassId].Add(s);
}
// Simpler with LINQ: students.GroupBy(s => s.ClassId).ToDictionary(g => g.Key, g => g.ToList())

// HashSet — unique values only, O(1) lookup
var presentStudentIds = new HashSet<int>();
presentStudentIds.Add(1);
presentStudentIds.Add(1);  // duplicate ignored
presentStudentIds.Contains(1);  // O(1) — very fast!
// Use HashSet when: storing unique IDs, fast membership check

🔍 LINQ Basics — Query Collections

var students = GetAllStudents(); // List<Student>

// Where — filter
var active = students.Where(s => s.IsActive).ToList();
var class5 = students.Where(s => s.ClassId == 5).ToList();

// Select — transform/project
var names = students.Select(s => s.Name).ToList();
var dtos  = students.Select(s => new StudentDto(s.Id, s.Name)).ToList();

// OrderBy / OrderByDescending
var sorted  = students.OrderBy(s => s.Name).ToList();
var sorted2 = students.OrderByDescending(s => s.FeeAmount).ToList();

// First / FirstOrDefault
var first  = students.First(s => s.Name == "Basha");           // throws if not found
var firstOrNull = students.FirstOrDefault(s => s.Name == "Basha"); // null if not found

// Aggregates
decimal total = students.Sum(s => s.FeeAmount);
decimal avg   = students.Average(s => s.FeeAmount);
int     max   = students.Max(s => s.ClassId);
int     count = students.Count(s => s.IsActive);

// Any / All / Contains
bool hasActive = students.Any(s => s.IsActive);
bool allActive = students.All(s => s.IsActive);

// GroupBy — group by class
var byClass = students
    .GroupBy(s => s.ClassId)
    .Select(g => new { ClassId = g.Key, Count = g.Count(), Students = g.ToList() })
    .ToList();

// Chain multiple LINQ methods
var top5FeeStudents = students
    .Where(s => s.IsActive)
    .OrderByDescending(s => s.FeeAmount)
    .Take(5)
    .Select(s => new { s.Name, s.FeeAmount })
    .ToList();

🎯 10 Interview Scenarios

1
When do you use for loop vs foreach?Loops

foreach: preferred for iterating collections — cleaner, reads like English, works with any IEnumerable. for: when you need the INDEX (i) — modifying list while iterating, iterating backwards, accessing adjacent elements, or when performance matters (array iteration with for is slightly faster). Rule: default to foreach. Switch to for when you need the index or position.

2
What is the difference between List and Array?Collections

Array: fixed size, set at creation. Cannot add/remove elements. Faster for random access. int[] ids = new int[5]. List<T>: dynamic size, can add/remove. Built on top of array internally — resizes by doubling. Count property (not Length). Most code uses List because student count is not known upfront. Array for: configuration values, lookup tables with known size.

3
What is the difference between Dictionary and List for student lookup?Collections

List lookup: students.FirstOrDefault(s => s.AdmissionNo == "ADM001") — O(n) linear scan, checks every element. Gets slower as list grows. Dictionary lookup: studentMap["ADM001"] — O(1) hash lookup, instant regardless of size. For 50,000 students: List lookup = 50,000 comparisons worst case. Dictionary = 1 hash calculation. Use Dictionary when you frequently look up by key. Use List when you iterate all, order matters, or key-based lookup is rare.

4
What happens if you modify a List while iterating with foreach?Gotcha

InvalidOperationException: "Collection was modified; enumeration operation may not execute." foreach uses an enumerator that detects modification. Solution: 1) Iterate a copy: foreach (var s in students.ToList()) — iterate copy, modify original. 2) Collect items to remove, then remove after loop. 3) Use for loop with index (be careful with index shifting when removing). Never add/remove from a collection you're currently foreach-ing.

5
What is the difference between First() and FirstOrDefault()?LINQ

First(): throws InvalidOperationException if no element matches. Use when: you KNOW an element exists and its absence is a bug. FirstOrDefault(): returns null (reference type) or default value (0 for int) if no match. Use when: element might not exist — check for null before using result. ZenCampus: GetStudent by AdmissionNumber — use FirstOrDefault, then check if null (student might have left school). Getting CURRENT_TERM — use First (always should exist — bug if missing).

6
Explain deferred vs immediate LINQ execution.LINQ

Deferred execution: query is NOT run when defined. It runs when iterated. var q = students.Where(s => s.IsActive) — no execution yet. q.ToList() — NOW it runs. This means if you add students after defining q but before .ToList(), they're included. Immediate execution: ToList(), ToArray(), Count(), Sum(), First() — force execution immediately. Bug: var q = students.Where(...); // modify students; q.ToList() — modified students included! Use .ToList() immediately if you want a snapshot.

7
When should you use HashSet instead of List?Collections

HashSet: stores UNIQUE values, O(1) Contains/Add. Use when: 1) Need to check membership frequently (presentStudentIds.Contains(id)). 2) Need unique values (no duplicates). 3) Set operations needed (Union, Intersect, Except). ZenCampus: tracking which students have paid fee — HashSet<int> paidStudentIds. Contains() check for each student: O(1). If List: O(n) for each check. For 50,000 students: HashSet wins massively.

8
How does GroupBy work in LINQ? Give a ZenCampus example.LINQ
// Group students by class, get count per class
var classSummary = students
    .GroupBy(s => s.ClassId)
    .Select(g => new {
        ClassId = g.Key,          // the grouping key
        StudentCount = g.Count(),
        TotalFee = g.Sum(s => s.FeeAmount),
        ActiveCount = g.Count(s => s.IsActive)
    })
    .OrderBy(x => x.ClassId)
    .ToList();
// g = IGrouping — acts like a List + has Key property
// g.Key = ClassId value for this group
9
What is the difference between IEnumerable<T> and List<T>?Collections

IEnumerable<T>: interface. Represents a sequence that can be iterated. Read-only, forward-only. Supports foreach and LINQ. Can be lazy (not materialized). List<T> implements IEnumerable<T> plus adds: indexing, Add/Remove, Sort, Count. Method signatures: return IEnumerable<Student> when caller only needs to iterate. Return List<Student> when caller needs Add/Remove/index. In EF Core: IQueryable<T> is the DB version — stays as SQL until ToList() called.

10
How do you implement pagination using LINQ?LINQ
// Page 1 = first 20 items, Page 2 = items 21-40, etc.
int page = 2;
int pageSize = 20;

var pagedStudents = students
    .Where(s => s.SchoolId == schoolId)
    .OrderBy(s => s.Name)        // always sort before paging!
    .Skip((page - 1) * pageSize) // skip previous pages
    .Take(pageSize)              // take current page
    .ToList();

// In EF Core: same LINQ translates to SQL
// SELECT * FROM Students
// ORDER BY Name
// OFFSET 20 ROWS FETCH NEXT 20 ROWS ONLY
