// var — function scoped, hoisted (AVOID in modern JS)
var studentName = "Basha";    // accessible before declaration (undefined)
var studentName = "Priya";    // re-declaration allowed! (bug-prone)

// let — block scoped, not hoisted (USE THIS)
let admissionNo = "ADM001";  // block scoped
if (true) {
    let temp = "inside block";
}
// console.log(temp); ← ReferenceError: not accessible outside block

// const — block scoped, cannot be reassigned (USE FOR VALUES THAT DON'T CHANGE)
const MAX_STUDENTS = 50;     // cannot reassign
const student = { name: "Basha", classId: 5 };
student.name = "Raja";       // ALLOWED — const prevents reassignment, not mutation
// student = { name: "Other" }; ← Error: cannot reassign const

// Scope rules
let schoolId = 1;
function getStudents() {
    console.log(schoolId);  // ✓ outer scope accessible
    let page = 1;           // only inside this function
}

⚡ Functions — Multiple Ways to Define

// 1. Function declaration — hoisted (can call before definition)
function calculateFee(studentId, term) {
    return studentId * term * 1000;
}

// 2. Function expression — not hoisted
const getFeeLabel = function(amount) {
    return `Fee: ₹${amount}`;
};

// 3. Arrow function — shorter syntax, no own 'this'
const double = (x) => x * 2;
const add = (a, b) => a + b;
const greet = name => `Hello, ${name}!`;  // single param, no parens needed
const getStudent = (id) => {             // multi-line needs braces + return
    const student = db.find(id);
    return student;
};

// Destructuring parameters
const displayStudent = ({ name, admNo, classId }) => {
    console.log(`${name} | ${admNo} | Class ${classId}`);
};
displayStudent({ name: "Basha", admNo: "ADM001", classId: 5 });

// Default parameters
const getStudents = (schoolId, page = 1, size = 20) => {
    // page=1, size=20 if not passed
};

// Rest parameters
const sumFees = (...amounts) => amounts.reduce((sum, a) => sum + a, 0);

📋 Arrays and Objects

// Arrays
const students = ["Basha", "Priya", "Raja"];

// Array methods (all return NEW array — original unchanged)
students.map(s => s.toUpperCase())         // transform: ["BASHA", "PRIYA", "RAJA"]
students.filter(s => s.length > 4)       // filter: ["Basha", "Priya"]
students.find(s => s === "Priya")        // first match: "Priya" (or undefined)
students.some(s => s === "Basha")        // any match: true
students.every(s => s.length > 3)        // all match: true
students.includes("Raja")               // contains: true
students.indexOf("Priya")               // position: 1
students.reduce((acc, s) => acc + " " + s) // accumulate: "Basha Priya Raja"
students.forEach(s => console.log(s))      // iterate (no return value)
students.sort()                            // sort alphabetically (mutates original!)

// Spread operator
const allStudents = [...students, "Arun", "Kavya"];  // merge arrays

// Objects
const student = {
    id: 1,
    name: "Basha",
    admNo: "ADM001",
    isActive: true,
    address: { city: "Chennai", state: "TN" }
};

// Access
student.name;          // "Basha" — dot notation
student["admNo"];    // "ADM001" — bracket notation (for dynamic keys)
student.address.city;  // "Chennai" — nested

// Destructuring
const { name, admNo, isActive } = student;  // extract properties
const { name: studentName, id } = student;  // rename while destructuring
const { address: { city } } = student;      // nested destructuring

// Spread for copying/merging objects
const updated = { ...student, name: "Raja", isActive: false }; // shallow copy + update

// Optional chaining
const city = student?.address?.city ?? "Unknown";  // safe nested access

⏳ Promises and async/await

// Promise — represents a future value
const fetchStudent = (id) => {
    return new Promise((resolve, reject) => {
        if (id > 0) {
            resolve({ id, name: "Basha" });  // success
        } else {
            reject(new Error("Invalid student ID"));  // failure
        }
    });
};

// .then/.catch (old way)
fetchStudent(1)
    .then(student => console.log(student.name))
    .catch(err => console.error(err.message));

// async/await (modern, preferred — reads like sync code)
async function loadStudentData(studentId) {
    try {
        const student = await fetchStudent(studentId);      // waits for promise
        const fees    = await fetchFees(student.id);         // sequential
        const attendance = await fetchAttendance(student.id);
        return { student, fees, attendance };
    } catch (error) {
        console.error("Failed to load:", error.message);
        throw error;  // re-throw for caller to handle
    }
}

// Parallel execution — don't await one by one if independent
const [student, fees] = await Promise.all([
    fetchStudent(studentId),
    fetchFees(studentId)
]);  // both run in parallel — faster!

// Promise.allSettled — get results even if some fail
const results = await Promise.allSettled([
    sendSms(parent1),
    sendSms(parent2),
    sendSms(parent3)
]);  // results[i].status = 'fulfilled' or 'rejected'

// fetch API — built-in HTTP client
async function getStudentsFromApi(schoolId) {
    const response = await fetch(`/api/students?schoolId=${schoolId}`, {
        headers: { 'Authorization': `Bearer ${getToken()}` }
    });
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    const data = await response.json();  // parse JSON
    return data;
}

🎯 10 Interview Scenarios

1
What is the difference between var, let, and const?Basics

var: function-scoped, hoisted as undefined, can be re-declared. Bug source. let: block-scoped, not hoisted (TDZ), cannot re-declare. const: block-scoped, cannot be reassigned (but object properties can change). Rule: default to const. Use let when you need to reassign. Never use var in modern JS. Hoisting: var declarations move to top of function scope. let/const are in "Temporal Dead Zone" — accessing before declaration throws ReferenceError.

2
What is the difference between == and === in JavaScript?Basics

==: loose equality. Converts types before comparing. "5" == 5 → true. null == undefined → true. === : strict equality. No type conversion. "5" === 5 → false. null === undefined → false. Always use ===. == leads to unexpected bugs. Famous example: [] == false → true (type coercion). ZenCampus: always === for ID comparisons, status checks.

3
What is 'this' in JavaScript arrow functions vs regular functions?Functions

Regular function: 'this' is determined at call time. Inside an event listener: this = the element. Inside a method called as function(): this = undefined (strict) or global. Arrow function: 'this' lexically bound — inherited from surrounding scope. Never has its own 'this'. In class methods: use arrow functions for callbacks to keep 'this' as the class instance. document.addEventListener('click', () => this.handleClick()) — 'this' is the class, not the element.

4
What is a Promise and how does async/await relate to it?Async

Promise: object representing eventual completion or failure of async operation. Three states: pending → fulfilled or rejected. async/await is syntactic sugar over Promises. async function always returns a Promise. await pauses execution until promise resolves. Equivalent: await fetchStudent(1) === fetchStudent(1).then(result => result). async/await makes async code look and feel like synchronous code — easier to read, debug, and handle errors with try/catch.

5
What is the difference between map() and forEach() in JavaScript arrays?Arrays

map(): transforms each element, RETURNS NEW ARRAY. students.map(s => s.name) → ["Basha", "Priya"]. forEach(): iterates, RETURNS UNDEFINED. Cannot chain after forEach. Use map when you need transformed array. Use forEach when you just want to execute something for each item (logging, DOM updates). Never forEach to build an array — that's what map is for. ZenCampus: .map(s => renderStudentRow(s)) to create row elements.

6
What is event delegation and why is it useful?DOM

Instead of adding click listener to each student row (100 listeners for 100 rows), add ONE listener to the parent table. When any row is clicked, event bubbles up to table. Check event.target to identify which row. Benefits: works for dynamically added rows, better performance (1 listener vs N). ZenCampus student table: tbody.addEventListener('click', (e) => { const row = e.target.closest('tr'); const id = row.dataset.id; })

7
What is the spread operator (...) and when to use it?ES6

Spread: "unfolds" an array/object. Arrays: [...arr1, ...arr2] = merge. [...arr] = copy. Math.max(...arr) = pass as args. Objects: {...obj1, ...obj2} = merge/override properties. {...student, isActive: false} = copy with one property changed. Immutable update — create new object instead of mutating. React state updates use spread: setStudent({...student, name: newName}).

8
What is closure in JavaScript?Concept

Closure: inner function remembers and accesses outer function's variables even after outer function returns. const makeCounter = () => { let count = 0; return () => ++count; }; const counter = makeCounter(); counter() → 1, counter() → 2. ZenCampus: function that "remembers" the schoolId: const getStudents = (schoolId) => (page) => fetch(`/api/students?schoolId=${schoolId}&page=${page}`). The returned function closes over schoolId.

9
What is the difference between null and undefined?Basics

undefined: variable declared but not assigned. Function returns undefined if no return statement. Accessing missing property: obj.nonExistent → undefined. null: explicitly assigned "no value." typeof null === 'object' (historic JS bug). null == undefined → true (loose), null === undefined → false (strict). ZenCampus: API returns null for "not found" — use null check. Default function parameters and uninitialized variables → undefined.

10
Explain Promise.all vs Promise.allSettled.Async

Promise.all: runs all promises in parallel. Returns array of results when ALL resolve. If ANY rejects — immediately rejects (fail-fast). Use when: all must succeed or abort. Promise.allSettled: runs all promises. Returns array of {status, value/reason} for EACH. Never rejects. Use when: best-effort — some may fail. ZenCampus: Sending SMS to 100 parents: Promise.allSettled — one failure shouldn't stop others. Loading student + fee + attendance for display: Promise.all — need all to render page.
