// Redis Cache Service — ZenCampus wrapper
public class RedisCacheService : ICacheService
{
    private readonly IDistributedCache _cache;
    private readonly JsonSerializerOptions _jsonOpts = new() { PropertyNameCaseInsensitive = true };

    public async Task<T?> GetAsync<T>(string key)
    {
        var bytes = await _cache.GetAsync(key);
        if (bytes is null) return default;
        return JsonSerializer.Deserialize<T>(bytes, _jsonOpts);
    }

    public async Task SetAsync<T>(string key, T value, TimeSpan ttl)
    {
        var bytes = JsonSerializer.SerializeToUtf8Bytes(value, _jsonOpts);
        await _cache.SetAsync(key, bytes, new DistributedCacheEntryOptions
        {
            AbsoluteExpirationRelativeToNow = ttl
        });
    }

    public async Task RemoveAsync(string key)
        => await _cache.RemoveAsync(key);
}

// Cache-Aside pattern — ZenCampus school master data
public class SchoolService : ISchoolService
{
    private readonly ICacheService       _cache;
    private readonly ISchoolRepository   _repo;

    public async Task<SchoolDto?> GetByIdAsync(int schoolId)
    {
        var cacheKey = $"school:{schoolId}"; // "school:42"

        // Step 1: Check cache
        var cached = await _cache.GetAsync<SchoolDto>(cacheKey);
        if (cached is not null) return cached; // cache HIT!

        // Step 2: Cache MISS → query DB
        var school = await _repo.GetByIdAsync(schoolId);
        if (school is null) return null;

        // Step 3: Store in cache — school data changes rarely, 1 hour TTL
        await _cache.SetAsync(cacheKey, school, TimeSpan.FromHours(1));

        return school;
    }

    // Generic helper — GetOrCreateAsync pattern
    public async Task<T> GetOrCreateAsync<T>(
        string key,
        Func<Task<T>> factory,
        TimeSpan ttl)
    {
        var cached = await _cache.GetAsync<T>(key);
        if (cached is not null) return cached;

        var value = await factory(); // DB call
        if (value is not null)
            await _cache.SetAsync(key, value, ttl);
        return value!;
    }
}

    var key     = $"fee:structure:{schoolId}:v{version}";

{

    await _redis.StringIncrementAsync($"fee:version:{schoolId}");

⚠️ KEYS command in production — NEVER!

KEYS * blocks entire Redis server while scanning. With 1 million keys = 500ms block = all requests timeout! Use SCAN with cursor instead — iterates in small batches, non-blocking.

6 Cache Stampede — Thundering Herd Problem!
🎬 Sarpatta — All fighters rush gate at same time — chaos! Cache miss + 1000 concurrent requests = all hit DB simultaneously!

9 AM school opens — 500 teachers login simultaneously. Cache expired (overnight). All 500 see cache miss → all 500 query DB → DB crashes! That's thundering herd. Stampede protection needed!

// Problem: 500 simultaneous cache misses → 500 DB queries!
// Fix: SemaphoreSlim lock — only one request fetches from DB

public class StampedeProtectedCache
{
    private readonly ICacheService       _cache;
    private readonly ConcurrentDictionary<string, SemaphoreSlim> _locks = new();

    public async Task<T> GetOrCreateAsync<T>(
        string key,
        Func<Task<T>> factory,
        TimeSpan ttl)
    {
        // Fast path — cache hit, no lock needed
        var cached = await _cache.GetAsync<T>(key);
        if (cached is not null) return cached;

        // Slow path — acquire per-key lock
        var semaphore = _locks.GetOrAdd(key, _ => new SemaphoreSlim(1, 1));
        await semaphore.WaitAsync();
        try
        {
            // Double-check after acquiring lock!
            // (Another thread may have populated cache while we waited)
            cached = await _cache.GetAsync<T>(key);
            if (cached is not null) return cached; // winner already cached it!

            // Only ONE thread reaches here — fetch from DB
            var value = await factory();
            await _cache.SetAsync(key, value, ttl);
            return value;
        }
        finally
        {
            semaphore.Release();
            _locks.TryRemove(key, out _);
        }
    }
}

// Alternative: Redis distributed lock (for multi-server stampede)
public async Task<T> GetOrCreateDistributedAsync<T>(string key, Func<Task<T>> factory, TimeSpan ttl)
{
    var cached = await _cache.GetAsync<T>(key);
    if (cached is not null) return cached;

    var lockKey     = $"lock:{key}";
    var lockAcquired = await _redis.StringSetAsync(
        lockKey, "1",
        TimeSpan.FromSeconds(10),
        When.NotExists); // NX flag — atomic set if not exists

    if (!lockAcquired)
    {
        // Wait for winner, then read cache
        await Task.Delay(100);
        return (await _cache.GetAsync<T>(key))!;
    }

    try
    {
        var value = await factory();
        await _cache.SetAsync(key, value, ttl);
        return value;
    }
    finally { await _redis.KeyDeleteAsync(lockKey); }
}

public class ResilientCacheService : ICacheService

        catch (RedisException ex)

            // Don't throw — app continues with DB-only mode!
        }
    }
}

Rate limiting for SMS OTP — Redis counter approach Medium Real Interview
📋 Scenario: "ZenCampus parent login — SMS OTP. User can request OTP spam பண்றான் (100 OTPs/minute to same phone = ₹100 SMS cost + abuse). Rate limit implement பண்ணு using Redis."

// Redis atomic increment — thread-safe, distributed!
public async Task<bool> IsAllowedAsync(string phone)
{
    var key   = $"rate:otp:{phone}:{DateTime.UtcNow:yyyyMMddHHmm}"; // per minute
    var count = await _redis.StringIncrementAsync(key); // atomic +1

    if (count == 1)
        await _redis.KeyExpireAsync(key, TimeSpan.FromMinutes(1)); // set TTL on first hit

    if (count > 3) // max 3 OTPs per minute
    {
        _log.LogWarning("OTP rate limit hit for {Phone}", phone);
        return false;
    }
    return true;
}

// Sliding window — more precise
public async Task<bool> SlidingWindowAllowedAsync(string phone)
{
    var key = $"rate:otp:sliding:{phone}";
    var now = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();

    // Add current timestamp to sorted set
    await _redis.SortedSetAddAsync(key, now.ToString(), now);
    // Remove entries older than 1 minute
    await _redis.SortedSetRemoveRangeByScoreAsync(key, 0, now - 60000);
    // Count requests in last 1 minute
    var count = await _redis.SortedSetLengthAsync(key);
    await _redis.KeyExpireAsync(key, TimeSpan.FromMinutes(2));

    return count <= 3;
}
6
IMemoryCache vs IDistributedCache — when to use which? Easy Real Interview
📋 Scenario: "Interviewer: 'IMemoryCache vs IDistributedCache — difference? ZenCampus-la both use பண்ணுவியா? When?'"
✅ CLEAR ANSWER

IMemoryCache: Stored in app server's own RAM. No network hop = ultra-fast (~0.01ms). But: each server has its own copy. Load balancer routes user to Server B — Server B has empty cache, misses! Data lost on restart. Only for single-server or data OK to be per-server (like compiled Razor pages, singleton config).

IDistributedCache (Redis): External Redis server. All app servers share same cache. Network hop = 0.3ms vs 0.01ms — but consistent! Session, fee structure, school data — one source of truth across all servers.

ZenCampus: IMemoryCache for HttpContext.Items (request-scoped), static lookup tables loaded at startup. IDistributedCache (Redis) for everything else: school data, fee structure, user sessions, dashboard, rate limiting.

10 ⚡ Quick Quiz

Q1: 3 servers behind load balancer. IMemoryCache use பண்ணா — problem?

A) No problem — all servers share IMemoryCache automatically
B) Each server has its own cache — cache miss on different servers, inconsistency!
C) IMemoryCache works fine with load balancers
D) Only problem if sticky sessions not enabled
✅ IMemoryCache = in-process, per-server. 3 servers = 3 independent caches. Server A caches school:42. User routed to Server B = cache miss! Use IDistributedCache (Redis) for shared cache across all servers!
❌ IMemoryCache lives in the app process — each server has completely separate memory! One server's cache is invisible to others. Multi-server = must use Redis (IDistributedCache)!

Q2: Cache stampede fix — which is MOST effective?

A) Increase Redis server RAM
B) Longer TTL — cache expires less often
C) SemaphoreSlim per-key lock + Jitter TTL + Background pre-warm
D) Use NOLOCK hint in SQL
✅ Stampede = simultaneous cache miss → DB flood. 3-layer fix: (1) Jitter TTL — stagger expiry, not all at once. (2) Lock — only 1 request hits DB, others wait. (3) Pre-warm — cache ready before rush hour. No stampede!
❌ Longer TTL reduces frequency but doesn't prevent stampede when it expires! Need: Jitter TTL (stagger expiry) + distributed lock (1 DB call, others wait) + background pre-warm (proactive refresh)!

Q3: KEYS * command production Redis-la use பண்ணலாமா?

A) Yes — it's the standard way to find keys by pattern
B) Never! KEYS blocks entire Redis server — use SCAN with cursor instead
C) OK for Redis with fewer than 10,000 keys
D) Only avoid in Azure, OK on self-hosted Redis
✅ KEYS is O(N) — blocks entire Redis while scanning. 1 million keys = 500ms block = all other Redis commands queued! Use SCAN cursor: iterates 100 keys at a time, non-blocking. Production Redis always SCAN, never KEYS!
❌ NEVER KEYS in production! Redis is single-threaded — KEYS blocks ALL other commands while it scans. With millions of keys = seconds of complete Redis unavailability. Always use SCAN (cursor-based, batched, non-blocking)!

Q4: Redis down ஆனா — application down ஆகணுமா?

A) Yes — cache is critical infrastructure
B) No — cache miss → fallback to DB. Slow but functional!
C) Only if using session storage
D) Application should retry Redis forever
✅ Cache = performance optimization, NOT required for correctness. Redis down → catch RedisException → return default (cache miss) → DB handles it. App is slower but operational. Session in Redis needs special handling — cookie fallback or JWT!
❌ Cache is optional! Wrap all Redis calls in try/catch. Failure = treat as cache miss → fall through to DB. Polly circuit breaker stops retrying a dead Redis. App degrades gracefully — slower, not broken!
