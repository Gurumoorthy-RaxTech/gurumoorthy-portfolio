// Token generation service — ZenCampus
public class TokenService : ITokenService
{
    private readonly JwtSettings _cfg;

    public string GenerateAccessToken(AppUser user)
    {
        var key = new SymmetricSecurityKey(
            Encoding.UTF8.GetBytes(_cfg.Secret));

        var claims = new[]
        {
            new Claim(ClaimTypes.NameIdentifier, user.Id.ToString()),
            new Claim(ClaimTypes.Name,           user.Username),
            new Claim(ClaimTypes.Role,           user.Role),      // Teacher/Admin
            new Claim("schoolId",               user.SchoolId.ToString()),
            new Claim("tenantId",               user.TenantId.ToString()),
            new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString()), // unique!
        };

        var token = new JwtSecurityToken(
            issuer:             _cfg.Issuer,
            audience:           _cfg.Audience,
            claims:             claims,
            notBefore:          DateTime.UtcNow,
            expires:            DateTime.UtcNow.AddMinutes(_cfg.AccessTokenExp),
            signingCredentials: new SigningCredentials(key, SecurityAlgorithms.HmacSha256)
        );

        return new JwtSecurityTokenHandler().WriteToken(token);
    }

    public string GenerateRefreshToken()
    {
        var bytes = RandomNumberGenerator.GetBytes(64);
        return Convert.ToBase64String(bytes); // 64-byte cryptographic random!
    }
}

// Refresh Token DB entity
public class RefreshToken
{
    public int      Id          { get; set; }
    public string  Token       { get; set; } = ""; // 64-byte random
    public int      UserId      { get; set; }
    public DateTime ExpiresAt   { get; set; }
    public DateTime? UsedAt     { get; set; } // one-time use!
    public DateTime? RevokedAt  { get; set; }
    public string?  ReplacedBy  { get; set; } // chain tracking!
    public string   DeviceInfo  { get; set; } = "";

    public bool IsActive => RevokedAt == null &&
                              UsedAt    == null &&
                              ExpiresAt  > DateTime.UtcNow;
}

// Auth Service — login flow
public async Task<AuthResponse> LoginAsync(LoginDto dto)
{
    var user = await _userRepo.GetByUsernameAsync(dto.Username)
        ?? throw new UnauthorizedAccessException("Invalid credentials");

    if (!BCrypt.Verify(dto.Password, user.PasswordHash))
        throw new UnauthorizedAccessException("Invalid credentials");

    var accessToken  = _tokenSvc.GenerateAccessToken(user);
    var refreshToken = new RefreshToken
    {
        Token      = _tokenSvc.GenerateRefreshToken(),
        UserId     = user.Id,
        ExpiresAt  = DateTime.UtcNow.AddDays(7),
        DeviceInfo = dto.DeviceInfo ?? "Unknown"
    };

    await _tokenRepo.SaveAsync(refreshToken);
    return new AuthResponse
    {
        AccessToken  = accessToken,
        RefreshToken = refreshToken.Token,
        ExpiresAt    = DateTime.UtcNow.AddMinutes(15)
    };
}

// Refresh endpoint — rotate token (one-time use!)
public async Task<AuthResponse> RefreshAsync(string refreshToken)
{
    var stored = await _tokenRepo.GetByTokenAsync(refreshToken)
        ?? throw new UnauthorizedAccessException("Invalid token");

    if (!stored.IsActive)
    {
        // SECURITY: Used token reuse = theft detected! Revoke ALL tokens for user!
        if (stored.UsedAt != null)
            await _tokenRepo.RevokeAllForUserAsync(stored.UserId);
        throw new UnauthorizedAccessException("Token already used or expired");
    }

    // Rotate — mark old used, create new
    stored.UsedAt     = DateTime.UtcNow;
    var newRefresh = new RefreshToken
    {
        Token      = _tokenSvc.GenerateRefreshToken(),
        UserId     = stored.UserId,
        ExpiresAt  = DateTime.UtcNow.AddDays(7),
        DeviceInfo = stored.DeviceInfo
    };
    stored.ReplacedBy = newRefresh.Token;
    await _tokenRepo.SaveAsync(newRefresh);
    await _ctx.SaveChangesAsync();

    var user = await _userRepo.GetByIdAsync(stored.UserId);
    return new AuthResponse
    {
        AccessToken  = _tokenSvc.GenerateAccessToken(user!),
        RefreshToken = newRefresh.Token
    };
}

💡 Access Token Storage — Where?

Memory (JS variable): Most secure — XSS can't steal, but lost on page refresh.
HttpOnly Cookie: XSS safe, CSRF risk (mitigate with SameSite=Strict + CSRF token).
localStorage: Persistent but XSS steals it — risky!
ZenCampus: Access token in memory, refresh token in HttpOnly cookie.

// Frontend — Axios interceptor for silent refresh
// (JavaScript — understanding the client side)

let isRefreshing = false;
let failedQueue  = [];

axios.interceptors.response.use(
    response => response,
    async error => {
        const original = error.config;

        if (error.response?.status === 401 && !original._retry) {
            original._retry = true;

            if (isRefreshing) {
                // Queue concurrent requests during refresh
                return new Promise((res, rej) =>
                    failedQueue.push({ res, rej }));
            }

            isRefreshing = true;
            try {
                // Refresh token in HttpOnly cookie — sent automatically!
                const { data } = await axios.post('/api/auth/refresh');
                accessToken = data.accessToken; // store in memory!
                processQueue(null, accessToken);
                return axios(original); // retry original request!
            } catch {
                processQueue(new Error('Refresh failed'));
                window.location.href = '/login'; // only now logout!
            } finally {
                isRefreshing = false;
            }
        }
        return Promise.reject(error);
    }
);
⭐ Backend refresh endpoint

Refresh token comes from HttpOnly cookie automatically (no JS reads it). Backend validates, issues new tokens, sets new HttpOnly cookie. Zero user friction — fee entry continues seamlessly!

// Strategy 1: JTI Blacklist in Redis (fastest — production grade!)
public async Task RevokeUserAsync(int userId)
{
    // Get all active JTIs for user from DB
    var activeJtis = await _ctx.UserTokenJtis
        .Where(j => j.UserId == userId && j.ExpiresAt > DateTime.UtcNow)
        .ToListAsync();

    // Blacklist each JTI in Redis with TTL = token expiry
    foreach (var jti in activeJtis)
    {
        var ttl = jti.ExpiresAt - DateTime.UtcNow;
        await _redis.SetAsync($"blacklist:{jti.Value}", "1", ttl);
    }
    // Revoke refresh tokens too
    await _ctx.RefreshTokens
        .Where(t => t.UserId == userId)
        .ExecuteUpdateAsync(s => s.SetProperty(t => t.RevokedAt, DateTime.UtcNow));
}

Role tamper — user change JWT payload-la role=Admin panna mudiyuma? Easy 🛡️ Security
📋 Scenario: "Interviewer kekuran: 'Teacher JWT payload decode பண்ணி role=Admin-a change பண்ணி modified token use பண்ணினா — API access kidekkumaa?' Answer with explanation."
✅ ANSWER — Signature Validation prevents this!

No — impossible without the secret key!

JWT signature = HMACSHA256(Base64(Header) + "." + Base64(Payload), SECRET_KEY)

Attacker decodes payload (Base64, not encrypted), changes role to "Admin", re-encodes. But signature was computed with ORIGINAL payload. New payload = new signature needed = needs SECRET_KEY!

Without secret key → can't generate valid signature → API returns 401 "Invalid signature".

// Tampered token flow:

// API: ValidateIssuerSigningKey=true → HMACSHA256 recomputed
//      recomputed ≠ ORIGINAL_SIGNATURE → 401 Unauthorized!

// That's WHY ValidateIssuerSigningKey must always be TRUE!
opts.TokenValidationParameters = new

Auth flow: Login → BCrypt password verify → JWT access token (15 min, HS256) + refresh token (7 days, 64-byte random). Access token in memory (React), refresh token in HttpOnly+Secure+SameSite=Strict cookie — XSS cannot steal it. Silent refresh via Axios interceptor — 401 → auto refresh → retry. User never sees logout.

JWT Claims: userId, role (SuperAdmin/SchoolAdmin/Teacher/Parent/Accountant), schoolId, tenantId, JTI. Claims-based policies for fee module — role + canAccessFees claim combination.

Multi-tenant isolation: EF Core global query filter on all entities — schoolId auto-appended to every query. Service layer double-checks. SuperAdmin bypasses filter.

Security incident: Found teacher using Postman to access other school's student (valid JWT, different schoolId in URL). Fixed with service-layer school validation — student.SchoolId != currentUser.SchoolId → 403.

Refresh rotation: One-time use refresh tokens. Reuse detected → revoke ALL user tokens (theft indicator). DbUpdateConcurrencyException handles concurrent refresh race condition.

Secrets: JWT secret in Azure Key Vault — never in appsettings.json. Pre-commit hooks with git-secrets to prevent accidental commits.

8 ⚡ Quick Quiz

Q1: JWT payload-la password store பண்ணலாமா?

A) Yes — it's Base64 encoded so secure
B) No — payload is only Base64, anyone can decode it. Never store sensitive data.
C) Yes — if using HS512
D) Only if payload is encrypted separately
✅ JWT payload = Base64 encoded (NOT encrypted). Visit jwt.io — paste any token, see payload immediately without secret! Only put: userId, role, schoolId. NEVER password, Aadhaar, sensitive data!
❌ Base64 ≠ encryption! Anyone can decode. jwt.io decodes without secret. Payload is for identity claims only — never secrets!

Q2: Access token 15 minutes மட்டும் valid — why not 7 days?

A) Server performance
B) Short TTL = if stolen, attacker has only 15 min. Long TTL = stolen token usable for days.
C) JWT specification limit
D) Database optimization
✅ JWT access token is STATELESS — can't be revoked (no DB check). Stolen token = usable until expiry. 15 min = theft window. 7 days = disaster. Refresh token (DB stored) can be revoked anytime!
❌ Security reason! JWT = stateless = can't revoke. Short TTL limits damage window. Refresh token (DB stored, revokable) handles "remember me" long sessions!

Q3: Refresh token reuse detected (used token used again) — what should happen?

A) Just return 401
B) Revoke ALL tokens for that user — reuse indicates token theft, force re-login everywhere
C) Issue new token anyway
D) Lock account permanently
✅ Refresh token reuse = security red flag. Either attacker used it, or legitimate user's device is compromised. Safest: revoke ALL refresh tokens for that user — forces re-login on all devices. Attacker kicked out!
❌ Reuse = possible token theft! Revoke ALL sessions for that user. This is "refresh token rotation + reuse detection" — industry best practice. All devices re-login = attacker loses access!

Q4: ValidateIssuerSigningKey = false set பண்ணினா?

A) Performance improves
B) Still secure — other validations protect
C) Anyone can forge JWT with any role — critical security hole!
D) Only development safe, production fine
✅ ValidateIssuerSigningKey=false = NO signature check = anyone generates JWT with role=SuperAdmin and accesses everything. This is the alg:none vulnerability! ALWAYS true in production!
❌ Critical security hole! Without signature validation, any fake JWT accepted. Attacker sets role=SuperAdmin, accesses all data. ALWAYS ValidateIssuerSigningKey=true!
