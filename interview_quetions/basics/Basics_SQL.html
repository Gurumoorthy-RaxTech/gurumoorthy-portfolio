-- Basic SELECT
SELECT * FROM Students;                   -- all columns (avoid in prod)
SELECT Id, Name, AdmNo, ClassId FROM Students;  -- specific columns

-- WHERE clause
SELECT * FROM Students WHERE SchoolId = 1;
SELECT * FROM Students WHERE ClassId = 5 AND IsActive = 1;
SELECT * FROM Students WHERE ClassId IN (5, 6, 7);
SELECT * FROM Students WHERE Name LIKE 'B%';     -- starts with B
SELECT * FROM Students WHERE Name LIKE '%Kumar%'; -- contains Kumar
SELECT * FROM Students WHERE Fee BETWEEN 5000 AND 10000;
SELECT * FROM Students WHERE Email IS NULL;
SELECT * FROM Students WHERE Email IS NOT NULL;

-- ORDER BY
SELECT * FROM Students ORDER BY Name ASC;     -- alphabetical
SELECT * FROM Students ORDER BY Fee DESC;     -- highest fee first
SELECT * FROM Students ORDER BY ClassId, Name;  -- multiple columns

-- TOP / LIMIT (SQL Server uses TOP)
SELECT TOP 10 * FROM Students ORDER BY Fee DESC;  -- top 10 highest fee

-- Pagination with OFFSET FETCH
SELECT * FROM Students
ORDER BY Name
OFFSET 20 ROWS           -- skip first 20 (page 1)
FETCH NEXT 20 ROWS ONLY;  -- take next 20 (page 2)

-- Column aliases
SELECT
    Id            AS StudentId,
    Name          AS StudentName,
    Fee * 12      AS AnnualFee
FROM Students;

-- DISTINCT
SELECT DISTINCT ClassId FROM Students WHERE SchoolId = 1;

🔗 JOINs — Combine Multiple Tables

-- INNER JOIN — only matching rows in BOTH tables
SELECT
    s.Name,
    s.AdmNo,
    c.ClassName,
    f.PendingAmount
FROM Students s
INNER JOIN Classes  c ON s.ClassId  = c.Id
INNER JOIN Fees     f ON s.Id       = f.StudentId
WHERE s.SchoolId = 1 AND f.PendingAmount > 0
ORDER BY f.PendingAmount DESC;

-- LEFT JOIN — all rows from LEFT table, NULL if no match on right
SELECT
    s.Name,
    s.AdmNo,
    f.PendingAmount  -- NULL if no fee record
FROM Students s
LEFT JOIN Fees f ON s.Id = f.StudentId
WHERE s.SchoolId = 1;
-- Shows ALL students, even those with no fee record

-- Find students with NO fee record
SELECT s.Name
FROM Students s
LEFT JOIN Fees f ON s.Id = f.StudentId
WHERE f.Id IS NULL;  -- right side NULL = no match

-- RIGHT JOIN — all from RIGHT, NULL if no match on left (uncommon, flip to LEFT JOIN)
-- FULL OUTER JOIN — all rows from both, NULL where no match

-- Self JOIN — join table to itself
SELECT s.Name AS Student, p.Name AS ParentStudent
FROM Students s
INNER JOIN Students p ON s.ParentStudentId = p.Id;  -- sibling relationship
JOIN TypeReturnsZenCampus Use
INNER JOINOnly matching rows in both tablesStudents with fees, students with attendance
LEFT JOINAll left rows + matching right (NULL if no match)All students even those without fee records
RIGHT JOINAll right rows + matching left (NULL if no match)Rare — flip to LEFT JOIN instead
FULL OUTER JOINAll rows from both tablesFind unmatched records on either side

📊 GROUP BY and Aggregations

-- Aggregate functions
SELECT
    COUNT(*),            -- total rows
    COUNT(Email),        -- non-NULL emails only
    SUM(Fee),            -- total fee
    AVG(Fee),            -- average fee
    MAX(Fee),            -- highest fee
    MIN(Fee)             -- lowest fee
FROM Students WHERE SchoolId = 1;

-- GROUP BY — aggregate per group
SELECT
    ClassId,
    COUNT(*) AS StudentCount,
    SUM(Fee) AS TotalFee,
    AVG(Fee) AS AvgFee
FROM Students
WHERE SchoolId = 1 AND IsActive = 1
GROUP BY ClassId
ORDER BY ClassId;

-- HAVING — filter AFTER grouping (WHERE filters before grouping)
SELECT
    ClassId,
    COUNT(*) AS StudentCount
FROM Students
WHERE SchoolId = 1          -- filter rows BEFORE grouping
GROUP BY ClassId
HAVING COUNT(*) > 30        -- filter groups AFTER grouping
ORDER BY StudentCount DESC;
-- Shows only classes with more than 30 students

-- GROUP BY with JOIN
SELECT
    c.ClassName,
    COUNT(s.Id) AS StudentCount,
    SUM(f.PendingAmount) AS TotalPending
FROM Classes c
LEFT JOIN Students  s ON c.Id = s.ClassId
LEFT JOIN Fees      f ON s.Id = f.StudentId
WHERE c.SchoolId = 1
GROUP BY c.Id, c.ClassName
ORDER BY TotalPending DESC;

🔍 Subqueries and CTEs

-- Subquery in WHERE
SELECT * FROM Students
WHERE Id IN (
    SELECT StudentId FROM Fees
    WHERE PendingAmount > 0
);

-- Correlated subquery — references outer query
SELECT s.Name,
    (SELECT SUM(PendingAmount)
     FROM Fees f
     WHERE f.StudentId = s.Id) AS TotalPending
FROM Students s
WHERE s.SchoolId = 1;

-- CTE (Common Table Expression) — cleaner than subquery
WITH UnpaidStudents AS (
    SELECT StudentId, SUM(PendingAmount) AS TotalPending
    FROM Fees
    WHERE PendingAmount > 0
    GROUP BY StudentId
),
ActiveStudents AS (
    SELECT * FROM Students
    WHERE IsActive = 1 AND SchoolId = 1
)
SELECT
    a.Name,
    a.AdmNo,
    u.TotalPending
FROM ActiveStudents a
INNER JOIN UnpaidStudents u ON a.Id = u.StudentId
ORDER BY u.TotalPending DESC;

-- EXISTS — efficient existence check
SELECT * FROM Students s
WHERE EXISTS (
    SELECT 1 FROM Fees f
    WHERE f.StudentId = s.Id AND f.PendingAmount > 0
);  -- faster than IN for large datasets

⚡ Indexes — Speed Up Queries

-- Indexes make SELECT faster, INSERT/UPDATE/DELETE slightly slower

-- Clustered index — physical order of table rows (Primary Key usually)
-- Only ONE per table

-- Non-clustered index — separate structure for fast lookup
CREATE INDEX IX_Students_SchoolId
    ON Students (SchoolId);

-- Composite index — multiple columns
CREATE INDEX IX_Students_School_Class
    ON Students (SchoolId, ClassId);
-- Covers: WHERE SchoolId = 1 AND ClassId = 5

-- Covering index — include extra columns to avoid table lookup
CREATE INDEX IX_Students_School_Name
    ON Students (SchoolId)
    INCLUDE (Name, AdmNo, IsActive);
-- Query: SELECT Name, AdmNo WHERE SchoolId = 1 → all from index, no table!

-- Unique index
CREATE UNIQUE INDEX UIX_Students_AdmNo
    ON Students (AdmNo, SchoolId);
-- Ensures no duplicate admission numbers within same school

-- Stored Procedure
CREATE PROCEDURE GetStudentsByClass
    @SchoolId INT,
    @ClassId  INT,
    @Page     INT = 1,
    @PageSize INT = 20
AS
BEGIN
    SELECT
        s.Id, s.Name, s.AdmNo,
        c.ClassName,
        f.PendingAmount
    FROM Students s
    INNER JOIN Classes c ON s.ClassId = c.Id
    LEFT JOIN  Fees    f ON s.Id      = f.StudentId
    WHERE s.SchoolId = @SchoolId
      AND s.ClassId  = @ClassId
      AND s.IsActive = 1
    ORDER BY s.Name
    OFFSET (@Page - 1) * @PageSize ROWS
    FETCH NEXT @PageSize ROWS ONLY;
END

🎯 10 Interview Scenarios

1
What is the difference between INNER JOIN and LEFT JOIN?JOINs

INNER JOIN: returns ONLY rows where a match exists in BOTH tables. If Student has no Fee record → student NOT included. LEFT JOIN: returns ALL rows from left table. If no match on right → right side columns are NULL. Student with no fee record → still appears, fee columns show NULL. ZenCampus: Attendance report — use INNER JOIN (only students who have attendance records). Student list with optional fee info — use LEFT JOIN (show all students, NULL fee if not yet assigned).

2
What is the difference between WHERE and HAVING?GROUP BY

WHERE: filters ROWS before grouping. Applied to individual row data. Cannot use aggregate functions. HAVING: filters GROUPS after GROUP BY. Applied to grouped/aggregated data. Can use COUNT(), SUM(), etc. Example: WHERE IsActive = 1 → filter individual rows first. HAVING COUNT(*) > 30 → then filter groups with < 30 students. Order: FROM → JOIN → WHERE → GROUP BY → HAVING → SELECT → ORDER BY → TOP/OFFSET.

3
What is a CTE and why use it instead of subquery?Advanced

CTE (Common Table Expression): named temporary result set. WITH CTE_Name AS (SELECT...) SELECT * FROM CTE_Name. Benefits over subquery: 1) Readable — complex logic broken into named steps. 2) Reusable — reference same CTE multiple times. 3) Recursive CTEs possible (tree structures). Subquery: inline, anonymous, hard to read when nested. Rule: simple one-level subquery is fine. Complex multi-step → CTE. ZenCampus: UnpaidStudents CTE → ActiveStudents CTE → JOIN both = clear and readable.

4
What is an index and what types exist?Performance

Index: separate data structure that speeds up SELECT by allowing direct row lookup without full table scan. Like a book index — go directly to page instead of scanning all pages. Types: 1) Clustered: rows stored physically sorted by key (usually PK). One per table. 2) Non-clustered: separate B-tree structure with pointers to rows. Multiple allowed. 3) Composite: covers multiple columns. Column order matters (leftmost prefix rule). 4) Unique: enforces uniqueness. 5) Covering: includes extra columns to avoid main table lookup. Trade-off: indexes slow INSERT/UPDATE/DELETE (index must be updated too).

5
Explain the SQL query execution order.Concept

SQL is DECLARED not executed top-to-bottom. Logical processing order: 1) FROM (which tables) 2) JOIN (combine tables) 3) WHERE (filter rows) 4) GROUP BY (group) 5) HAVING (filter groups) 6) SELECT (pick columns) 7) DISTINCT (remove duplicates) 8) ORDER BY (sort) 9) TOP/OFFSET (limit rows). This is why you CANNOT reference a SELECT alias in WHERE — WHERE runs before SELECT. You CAN reference alias in HAVING or ORDER BY in some DB engines.

6
What is N+1 query problem and how to fix in SQL?Performance

N+1: 1 query to get N students, then N separate queries to get each student's fee = N+1 total queries. Example: SELECT 100 students, then for each: SELECT fee WHERE StudentId = 1... 2... 3... → 101 queries! Fix: JOIN in one query. SELECT s.*, f.PendingAmount FROM Students s LEFT JOIN Fees f ON s.Id = f.StudentId WHERE s.SchoolId = 1 → 1 query. ZenCampus: Real bug we fixed — went from 8 seconds (N+1) to 0.4 seconds (JOIN + include in EF Core).

7
What is the difference between EXISTS and IN?Advanced

IN (subquery): fetches all values from subquery, then checks membership. If subquery returns 50,000 IDs → all loaded into memory for comparison. EXISTS: stops at FIRST match. Doesn't fetch all results. Often faster when subquery returns large set. Rule: prefer EXISTS for existence checks with correlated subquery. IN is fine for small static lists: WHERE Status IN ('Active', 'Pending'). For subqueries with large results: WHERE EXISTS (SELECT 1 FROM Fees WHERE StudentId = s.Id).

8
Write a query to find the 2nd highest fee paid student.Problem
-- Method 1: Subquery
SELECT TOP 1 Name, Fee
FROM Students
WHERE Fee < (SELECT MAX(Fee) FROM Students)
ORDER BY Fee DESC;

-- Method 2: ROW_NUMBER (handles ties correctly)
WITH Ranked AS (
    SELECT Name, Fee,
        ROW_NUMBER() OVER (ORDER BY Fee DESC) AS RowNum
    FROM Students
    WHERE SchoolId = 1
)
SELECT Name, Fee FROM Ranked WHERE RowNum = 2;

-- DENSE_RANK() for "2nd unique fee value" (handles ties)
-- ROW_NUMBER() for "2nd row by fee"
9
What are window functions? Give an example.Advanced
-- Window functions: aggregate over a partition WITHOUT collapsing rows
-- OVER (PARTITION BY ... ORDER BY ...)
SELECT
    Name,
    ClassId,
    Fee,
    RANK()       OVER (PARTITION BY ClassId ORDER BY Fee DESC) AS FeeRank,
    ROW_NUMBER() OVER (PARTITION BY ClassId ORDER BY Fee DESC) AS RowNum,
    SUM(Fee)     OVER (PARTITION BY ClassId) AS ClassTotalFee,
    AVG(Fee)     OVER (PARTITION BY ClassId) AS ClassAvgFee
FROM Students
WHERE SchoolId = 1;
-- Each row kept, but enriched with class-level aggregates and ranking
-- FeeRank=1 means highest fee in that class
10
What is a transaction and when do you use it?Transactions
-- Transaction: atomic unit — ALL succeed or ALL rollback
BEGIN TRANSACTION;

BEGIN TRY
    -- Admit student
    INSERT INTO Students (Name, AdmNo, ClassId, SchoolId)
    VALUES ('Basha', 'ADM001', 5, 1);

    DECLARE @StudentId INT = SCOPE_IDENTITY();

    -- Create fee record
    INSERT INTO Fees (StudentId, TotalAmount, PaidAmount, Term)
    VALUES (@StudentId, 5000, 0, 'Term1');

    COMMIT TRANSACTION;  -- both succeed → commit
END TRY
BEGIN CATCH
    ROLLBACK TRANSACTION;  -- any error → undo both
    THROW;
END CATCH;
-- If fee insert fails → student insert also undone. Data consistency!
