⚡ Controllers — Handle HTTP Requests

[Route("students")]
public class StudentController : Controller
{
    private readonly IStudentService _service;

    public StudentController(IStudentService service)
        => _service = service;  // dependency injection

    // GET /students — list all students
    [HttpGet]
    public async Task<IActionResult> Index(int schoolId)
    {
        var students = await _service.GetAllAsync(schoolId);
        return View(students);  // pass model to View
    }

    // GET /students/5 — show single student
    [HttpGet("{id:int}")]
    public async Task<IActionResult> Details(int id)
    {
        var student = await _service.GetByIdAsync(id);
        if (student is null) return NotFound();  // 404
        return View(student);
    }

    // GET /students/create — show empty form
    [HttpGet("create")]
    public IActionResult Create() => View();

    // POST /students/create — process form submission
    [HttpPost("create")]
    [ValidateAntiForgeryToken]  // CSRF protection
    public async Task<IActionResult> Create(CreateStudentViewModel model)
    {
        if (!ModelState.IsValid)    // check validation annotations
            return View(model);       // re-show form with errors

        await _service.CreateAsync(model);

        TempData["Success"] = "Student admitted successfully!";
        return RedirectToAction(nameof(Index));  // POST-REDIRECT-GET
    }

    // DELETE /students/5 — AJAX delete
    [HttpPost("{id}/delete")]
    public async Task<IActionResult> Delete(int id)
    {
        await _service.DeleteAsync(id);
        return Json(new { success = true });  // JSON for AJAX
    }
}

IActionResult Return Types

return View();                        // render View/ControllerName/ActionName.cshtml
return View("CustomView", model);    // specify view name + model
return RedirectToAction("Index");    // 302 redirect to action
return RedirectToAction("Details", new { id = student.Id }); // with route data

return Unauthorized();               // 401

return File(bytes, "application/pdf", "report.pdf"); // file download

📋 Models and Data Annotations

public class CreateStudentViewModel
{
    [Required(ErrorMessage = "Student name is required")]
    [StringLength(100, MinimumLength = 2)]
    public string Name { get; set; } = string.Empty;

    [Required]
    [Range(1, 12, ErrorMessage = "Class must be 1 to 12")]
    public int ClassId { get; set; }

    [EmailAddress(ErrorMessage = "Invalid email format")]
    public string? Email { get; set; }

    [Phone]
    [Required(ErrorMessage = "Parent mobile required")]
    public string ParentMobile { get; set; } = string.Empty;

    [DataType(DataType.Date)]
    [DisplayFormat(DataFormatString = "{0:dd-MM-yyyy}")]
    public DateTime DateOfBirth { get; set; }
}

// ViewBag — dynamic, one-way Controller → View (NOT across redirects)
ViewBag.PageTitle = "Student List";
ViewBag.TotalCount = students.Count;
// In Razor: @ViewBag.PageTitle

// ViewData — dictionary version (string key)
ViewData["Title"] = "Admission Form";

// TempData — persists across ONE redirect (POST-Redirect-GET pattern)
TempData["SuccessMessage"] = "Student admitted!";
return RedirectToAction("Index");
// In Index view: @TempData["SuccessMessage"] — shows message after redirect

// SelectList — for dropdowns in forms

    await _classService.GetAllAsync(schoolId),
    "Id",       // value field
    "ClassName" // display field

🛣️ Routing — URL to Action Mapping

// Attribute routing (PREFERRED in .NET Core)
[Route("school/{schoolId}/students")]
public class StudentController : Controller
{
    [HttpGet]
    public IActionResult Index(int schoolId) { ... }  // GET /school/1/students

    [HttpGet("{id:int}")]
    public IActionResult Details(int id) { ... }       // GET /school/1/students/5

    [HttpGet("search")]
    public IActionResult Search(string q) { ... }      // GET /school/1/students/search?q=basha
}

// Route constraints
[HttpGet("{id:int}")]          // only integers
[HttpGet("{name:alpha}")]      // only letters
[HttpGet("{admNo:regex(^ADM)}")] // regex pattern

// Convention routing (program.cs)
app.MapControllerRoute(
    name: "default",
    pattern: "{controller=Home}/{action=Index}/{id?}");
// /Student/Details/5 → StudentController.Details(id: 5)

// Model Binding — auto-binds from route/query/form
public IActionResult Index(
    int schoolId,     // from route: /students?schoolId=1
    string? search,   // from query: ?search=basha
    int page = 1     // default value if not provided
) { ... }

// [FromBody] — for JSON/API controllers
[HttpPost]
public IActionResult Create([FromBody] CreateStudentDto dto) { ... }

🎯 10 Interview Scenarios

1
Explain the MVC pattern and what each part does.Concept

Model: holds data and business rules. Defines what data looks like and validation. Student.cs, StudentCreateDto.cs. View: renders HTML to user. Razor .cshtml files. Knows nothing about where data came from. Controller: receives HTTP request, reads data from Model, passes to View. Orchestrates Model + View. No business logic should live here. Separation of concerns means: change DB schema → update Model only. Change HTML layout → update View only. Change URL structure → update Controller/Routing only.

2
What is the difference between ViewBag, ViewData, and TempData?Data Passing

ViewBag: dynamic property bag. ViewBag.Title = "Students". Accessed in View as @ViewBag.Title. Lost after redirect. ViewData: dictionary version. ViewData["Title"] = "Students". @ViewData["Title"]. Same limitations. TempData: stored in session, survives ONE redirect. Used for POST-Redirect-GET pattern: after form submit → redirect → show success message. TempData["Success"] = "Saved!"; RedirectToAction → next page shows message. After one render, TempData is cleared. Use TempData.Keep("key") if needed beyond first read.

3
What is the POST-Redirect-GET pattern and why use it?Pattern

Problem: After POST form submit → return View directly. User presses F5 (refresh) → re-submits form → duplicate student created! Solution: POST form → process → RedirectToAction (GET) → show result page. Now F5 refreshes the GET page, not re-submits POST. ZenCampus: [HttpPost] Create() → save student → TempData["Success"] → RedirectToAction("Index"). Index GET action shows students + reads TempData success message.

4
How does Model validation work in ASP.NET MVC?Validation

1) Annotate ViewModel with [Required], [StringLength], [Range], [EmailAddress], etc. 2) In controller: if (!ModelState.IsValid) return View(model) — re-shows form with errors. 3) In Razor view: asp-validation-for="PropertyName" renders error messages. 4) Client-side: jQuery Unobtrusive Validation script validates in browser before submit. Data annotations are checked on both client AND server. Never skip server-side check — client can be bypassed with curl/Postman.

5
What is a ViewModel vs a Model/Entity?Architecture

Entity/Model: maps to DB table. Has all fields. Student.cs with Id, Name, SchoolId, etc. ViewModel: tailored for a specific View. Contains EXACTLY what the view needs — no more. Advantages: 1) Don't expose sensitive entity fields (passwords, internal IDs). 2) Add computed/display-only fields not in DB (Age calculated from DOB, FormattedFee). 3) Combine multiple entities: StudentDetailsViewModel = Student + List<Fee> + List<Attendance>. Never pass raw entities to views — use ViewModels.

6
What is model binding and how does it work?Binding

Model binding: ASP.NET Core automatically maps HTTP request data to action method parameters. Sources (checked in order): 1) Route data (/students/5 → id=5). 2) Query string (?page=2 → page=2). 3) Form body (POST form fields → ViewModel properties). 4) JSON body ([FromBody]). 5) HTTP headers ([FromHeader]). For complex types: each property matched by name. [BindNever] to exclude properties. [BindRequired] to enforce required. ZenCampus: Create(CreateStudentViewModel model) — MVC populates all model properties from form POST.

7
What is [ValidateAntiForgeryToken] and why is it needed?Security

Protects against CSRF (Cross-Site Request Forgery). Without it: attacker's website can make a POST to your site using the logged-in user's cookies. [ValidateAntiForgeryToken] on POST actions validates a hidden form field token generated by @Html.AntiForgeryToken() (or asp-action tag helper does it automatically). If token missing or mismatched → 400 error. Always add to POST, PUT, DELETE actions that change data. AntiForgery token included automatically when using form asp-action tag helper.

8
Explain the difference between Controller and ApiController.Architecture

Controller (inherits Controller): for MVC — returns IActionResult, Views, ViewBag. For browser-rendered HTML pages. ApiController ([ApiController] attribute + inherits ControllerBase): for REST APIs — returns JSON/XML. Automatic model validation (400 if invalid, no manual ModelState.IsValid check). [FromBody] assumed for complex types. Returns ActionResult<T>. ZenCampus: StudentController : Controller for web pages (admission forms). StudentsApiController : ControllerBase for mobile app / React frontend API calls.

9
What are Tag Helpers and how are they different from HTML Helpers?Views

HTML Helpers (old): @Html.TextBoxFor(m => m.Name), @Html.ActionLink("Details", "Student", new { id = 1 }) — C# method calls embedded in HTML. Tag Helpers (modern): <input asp-for="Name" />, <a asp-action="Details" asp-route-id="1"> — look like HTML attributes. More readable, IntelliSense support, designer-friendly. Tag helpers generate correct HTML including name, id, validation attributes automatically. Prefer tag helpers in new projects — they're the ASP.NET Core way.

10
How do you handle AJAX calls from MVC views?AJAX
// Controller returns JSON for AJAX
[HttpPost("{id}/delete")]
public async Task Delete(int id)
{
    await _service.DeleteAsync(id);
    return Json(new { success = true, message = "Deleted" });
}

// OR return IActionResult with status code
return Ok(new { success = true });
return BadRequest(new { error = "Not found" });

// JavaScript in View
fetch('/students/5/delete', {
    method: 'POST',
    headers: {
        'RequestVerificationToken': document.querySelector('[name="__RequestVerificationToken"]').value
    }
})
.then(r => r.json())
.then(data => {
    if (data.success) row.remove();
});
