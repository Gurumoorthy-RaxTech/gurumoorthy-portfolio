🎣 React Hooks — ZenCampus Student List

// StudentList.tsx — Core hooks pattern
import { useState, useEffect, useCallback } from 'react';
import { studentApi } from '../api/studentApi';
import { useAuth } from '../context/AuthContext';

interface Student {
  id: number;
  name: string;
  admissionNumber: string;
  className: string;
  isActive: boolean;
}

export function StudentList() {
  // State hooks — like private fields in a class
  const [students, setStudents] = useState<Student[]>([]);
  const [loading, setLoading]   = useState(true);
  const [error, setError]       = useState<string | null>(null);
  const [page, setPage]         = useState(1);
  const [search, setSearch]     = useState('');
  const { user } = useAuth(); // from context

  // Effect hook — runs after render (like OnParametersSetAsync in Blazor)
  const loadStudents = useCallback(async () => {
    try {
      setLoading(true);
      setError(null);
      const data = await studentApi.getAll({
        schoolId: user!.schoolId,
        page,
        search
      });
      setStudents(data.items);
    } catch (err) {
      setError('Failed to load students. Please try again.');
      console.error(err);
    } finally {
      setLoading(false);
    }
  }, [user, page, search]); // re-run when these change

  useEffect(() => {
    const debounce = setTimeout(loadStudents, 300); // debounce search
    return () => clearTimeout(debounce); // cleanup
  }, [loadStudents]);

  if (loading) return <LoadingSpinner />;
  if (error)   return <ErrorAlert message={error} onRetry={loadStudents} />;

  return (
    <div>
      <SearchInput value={search} onChange={e => setSearch(e.target.value)} />
      <table>
        <thead><tr><th>Name</th><th>Adm No</th><th>Class</th></tr></thead>
        <tbody>
          {students.map(s => (
            <tr key={s.id}>
              <td>{s.name}</td>
              <td>{s.admissionNumber}</td>
              <td>{s.className}</td>
            </tr>
          ))}
        </tbody>
      </table>
      <Pagination page={page} onPageChange={setPage} />
    </div>
  );
}

🌐 Axios — Typed API Client for ZenCampus

// src/api/axiosInstance.ts
import axios from 'axios';

const api = axios.create({
  baseURL: import.meta.env.VITE_API_URL ?? 'https://api.zencampus.com',
  timeout: 10000,
  headers: { 'Content-Type': 'application/json' }
});

// Request interceptor — attach JWT token
api.interceptors.request.use(config => {
  const token = localStorage.getItem('zencampus_token');
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Response interceptor — handle 401 globally
api.interceptors.response.use(
  response => response,
  error => {
    if (error.response?.status === 401) {
      localStorage.removeItem('zencampus_token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default api;

// src/api/studentApi.ts — typed API functions
export const studentApi = {
  getAll: (params: { schoolId: number; page: number; search?: string }) =>
    api.get<PagedResult<StudentDto>>('/api/students', { params }).then(r => r.data),

  getById: (id: number) =>
    api.get<StudentDetailDto>(`/api/students/${id}`).then(r => r.data),

  create: (cmd: CreateStudentCommand) =>
    api.post<{ id: number }>('/api/students', cmd).then(r => r.data),

  update: (id: number, cmd: UpdateStudentCommand) =>
    api.put(`/api/students/${id}`, cmd),

  deactivate: (id: number, reason: string) =>
    api.delete(`/api/students/${id}`, { data: { reason } }),
};

🌍 useContext — Auth + School Context

// src/context/AuthContext.tsx
interface AuthUser { id: number; name: string; schoolId: number; role: string; }
interface AuthContextType {
  user: AuthUser | null;
  login: (token: string) => void;
  logout: () => void;
  isAuthenticated: boolean;
}

const AuthContext = createContext<AuthContextType>(null!);

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<AuthUser | null>(() => {
    const token = localStorage.getItem('zencampus_token');
    return token ? parseJwtPayload(token) : null;
  });

  const login = useCallback((token: string) => {
    localStorage.setItem('zencampus_token', token);
    setUser(parseJwtPayload(token));
  }, []);

  const logout = useCallback(() => {
    localStorage.removeItem('zencampus_token');
    setUser(null);
  }, []);

  return (
    <AuthContext.Provider value={{ user, login, logout, isAuthenticated: !!user }}>
      {children}
    </AuthContext.Provider>
  );
}

export const useAuth = () => useContext(AuthContext);

🎯 10 Interview Scenarios

1
What is the difference between useState and useEffect?Concept

useState: holds component state (memory). When you call setState, React re-renders the component. Like a class field that triggers UI update when changed. useEffect: runs side effects after render (data fetching, subscriptions, timers). Like OnAfterRenderAsync in Blazor. useEffect with dependency array = only re-run when specified values change. Cleanup function prevents memory leaks.

2
Why do you need a cleanup function in useEffect for API calls?Code

Race condition: user types "Basha", effect fires API call. Before it returns, user types "Ba" — second API call fires. First call returns after second: students get set to "Ba" results, then "Basha" results overwrite — wrong order. Fix: AbortController. Cleanup calls controller.abort() — when effect re-runs, previous fetch is cancelled. Also prevents setting state on unmounted component warning.

3
What is useCallback and why use it with useEffect?Code

useCallback memoizes a function — same reference returned unless dependencies change. Without useCallback: function is re-created on every render. If that function is in useEffect's dependency array, useEffect re-runs every render = infinite loop. useCallback with [user, page, search]: function only re-created when those change, useEffect only re-fires when they change. Correct for ZenCampus loadStudents.

4
What is Context and when should it be used vs prop drilling?Concept

Prop drilling: pass user info through 5 component levels just to reach a deep component. Context: global state accessible by any component without passing through intermediaries. Use Context for: auth user, theme, school selection, language. DON'T use for: frequent updates (cart item count changing) — it causes all consumers to re-render. ZenCampus: AuthContext (login/logout/user), SchoolContext (selected school).

5
How does Axios interceptor handle JWT refresh?Code

Response interceptor catches 401. Options: 1) Simple: redirect to login. 2) Advanced: try refreshToken API with the refresh token stored in httpOnly cookie. If refresh succeeds, retry the original request. If refresh fails, redirect to login. Interceptor runs globally — all API calls benefit. Without interceptor: every API function must handle 401 separately — duplicated code.

6
Explain the key prop in React lists and why it matters.Concept

React uses key to identify which items changed, added, or removed when re-rendering a list. Without key (or with index as key): React re-renders wrong elements, input values shift, animations break. Use stable, unique ID: student.id (from DB). Never use array index — if items reorder or delete, React gets confused. For ZenCampus student list: key={student.id} ensures correct DOM updates when sorting or filtering.

7
What is the difference between controlled and uncontrolled form inputs in React?Concept

Controlled: input value driven by React state. value={name} onChange={e => setName(e.target.value)}. React is the single source of truth. Easy validation, easy reset. Uncontrolled: input has its own DOM state. Read via ref.current.value. Simpler for basic cases, harder to validate. React Hook Form uses controlled + performance optimizations. ZenCampus admission form: controlled inputs for all student data fields for validation.

8
How would you protect ZenCampus routes for teacher-only pages?Code

Role-based ProtectedRoute component: check user.role from AuthContext. If user.role !== 'Teacher' and !== 'Admin', redirect to /unauthorized. Apply to routes: <Route element={<RoleGuard roles={['Teacher','Admin']}>}>. Backend MUST also enforce roles — frontend guard is UX only, not security. ZenCampus: fee collection routes = Admin/Accountant only, attendance = Teacher/Admin.

9
What is the issue with calling setState in useEffect without dependencies?Bug

Infinite loop: useEffect runs after every render. Inside, setState is called. setState triggers re-render. Re-render triggers useEffect again. Repeat forever. Fix: add a dependency array []. Empty array = run once after mount. Non-empty array = run when specific values change. If you need to react to state changes, include those state vars in the dependency array and break the cycle with conditions.

10
How does React Router v6 handle navigation and what replaced history.push?Code

React Router v6: useNavigate() hook replaces useHistory(). const navigate = useNavigate(); navigate('/students/42') — go to page. navigate(-1) — go back. navigate('/login', { replace: true }) — replace current history entry. Declarative: <Navigate to="/login" replace /> for redirect in render. <Link to="/students"> for anchor tags. For ZenCampus: after successful admission, navigate('/students', { state: { admitted: true } }).
