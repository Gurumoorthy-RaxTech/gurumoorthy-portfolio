🔍 Query — Read Data

// Setup: install Dapper NuGet package
// using Dapper;
// using System.Data;
// using Microsoft.Data.SqlClient;

public class StudentRepository
{
    private readonly string _connStr;
    private IDbConnection OpenConn() => new SqlConnection(_connStr);

    // Query<T> — returns IEnumerable<T> (multiple rows)
    public async Task<IEnumerable<Student>> GetByClassAsync(int schoolId, int classId)
    {
        const string sql = @"
            SELECT Id, Name, AdmNo, ClassId, IsActive, Fee
            FROM Students
            WHERE SchoolId = @SchoolId AND ClassId = @ClassId AND IsActive = 1
            ORDER BY Name";

        using var conn = OpenConn();
        return await conn.QueryAsync<Student>(sql, new { SchoolId = schoolId, ClassId = classId });
    }

    // QuerySingleOrDefault<T> — expects 0 or 1 result (throws if > 1)
    public async Task<Student?> GetByAdmNoAsync(string admNo, int schoolId)
    {
        const string sql = @"
            SELECT Id, Name, AdmNo, ClassId, Email, ParentMobile
            FROM Students
            WHERE AdmNo = @AdmNo AND SchoolId = @SchoolId";

        using var conn = OpenConn();
        return await conn.QuerySingleOrDefaultAsync<Student>(sql, new { AdmNo = admNo, SchoolId = schoolId });
    }

    // Query scalar value
    public async Task<int> GetActiveCountAsync(int schoolId)
    {
        using var conn = OpenConn();
        return await conn.ExecuteScalarAsync<int>(
            "SELECT COUNT(*) FROM Students WHERE SchoolId = @SchoolId AND IsActive = 1",
            new { SchoolId = schoolId });
    }

    // Query with complex JOIN and alias mapping
    public async Task<IEnumerable<StudentFeeDto>> GetStudentsWithFeesAsync(int schoolId)
    {
        const string sql = @"
            SELECT
                s.Id          AS StudentId,
                s.Name        AS StudentName,
                s.AdmNo,
                c.ClassName,
                f.PendingAmount,
                f.LastPaymentDate
            FROM Students s
            INNER JOIN Classes c ON s.ClassId = c.Id
            LEFT JOIN  Fees    f ON s.Id      = f.StudentId
            WHERE s.SchoolId = @SchoolId AND s.IsActive = 1
            ORDER BY f.PendingAmount DESC";

        using var conn = OpenConn();
        return await conn.QueryAsync<StudentFeeDto>(sql, new { SchoolId = schoolId });
    }
}

✍️ Execute — Insert, Update, Delete

// Execute — returns affected row count (no result set)
public async Task<int> CreateAsync(Student student)
{
    const string sql = @"
        INSERT INTO Students (Name, AdmNo, ClassId, SchoolId, IsActive, CreatedAt)
        VALUES (@Name, @AdmNo, @ClassId, @SchoolId, 1, GETUTCDATE())";

    using var conn = OpenConn();
    return await conn.ExecuteAsync(sql, student);  // returns 1 (rows affected)
}

// Insert and get back the new ID
public async Task<int> InsertAndGetIdAsync(Student student)
{
    const string sql = @"
        INSERT INTO Students (Name, AdmNo, ClassId, SchoolId)
        VALUES (@Name, @AdmNo, @ClassId, @SchoolId);
        SELECT CAST(SCOPE_IDENTITY() AS INT);";

    using var conn = OpenConn();
    return await conn.ExecuteScalarAsync<int>(sql, student);
}

// Update
public async Task<bool> UpdateAsync(Student student)
{
    const string sql = @"
        UPDATE Students
        SET Name = @Name, ClassId = @ClassId, UpdatedAt = GETUTCDATE()
        WHERE Id = @Id AND SchoolId = @SchoolId";

    using var conn = OpenConn();
    int rows = await conn.ExecuteAsync(sql, student);
    return rows > 0;  // true if student found and updated
}

// Soft delete
public async Task<bool> DeleteAsync(int id, int schoolId)
{
    using var conn = OpenConn();
    var rows = await conn.ExecuteAsync(
        "UPDATE Students SET IsActive = 0 WHERE Id = @Id AND SchoolId = @SchoolId",
        new { Id = id, SchoolId = schoolId });
    return rows > 0;
}

// Bulk insert — execute with list
public async Task BulkInsertAttendanceAsync(IEnumerable<Attendance> records)
{
    const string sql = @"
        INSERT INTO Attendance (StudentId, Date, Status, MarkedBy)
        VALUES (@StudentId, @Date, @Status, @MarkedBy)";

    using var conn = OpenConn();
    await conn.ExecuteAsync(sql, records);  // Dapper loops + inserts each record
}

⚙️ DynamicParameters — Dynamic SQL

// DynamicParameters — build parameters dynamically (for optional filters)
public async Task<IEnumerable<Student>> SearchAsync(
    int schoolId,
    string? name = null,
    int? classId = null,
    bool? isActive = null)
{
    var sql = new StringBuilder("SELECT * FROM Students WHERE SchoolId = @SchoolId");
    var parameters = new DynamicParameters();
    parameters.Add("SchoolId", schoolId);

    if (!string.IsNullOrWhiteSpace(name))
    {
        sql.Append(" AND Name LIKE @Name");
        parameters.Add("Name", $"%{name}%");
    }
    if (classId.HasValue)
    {
        sql.Append(" AND ClassId = @ClassId");
        parameters.Add("ClassId", classId.Value);
    }
    if (isActive.HasValue)
    {
        sql.Append(" AND IsActive = @IsActive");
        parameters.Add("IsActive", isActive.Value);
    }
    sql.Append(" ORDER BY Name");

    using var conn = OpenConn();
    return await conn.QueryAsync<Student>(sql.ToString(), parameters);
}

// Call stored procedure
public async Task<IEnumerable<StudentSummaryDto>> GetClassSummaryAsync(int schoolId, int term)
{
    var parameters = new DynamicParameters();
    parameters.Add("@SchoolId", schoolId);
    parameters.Add("@Term", term);
    parameters.Add("@TotalCount", dbType: DbType.Int32,
        direction: ParameterDirection.Output);  // OUTPUT parameter

    using var conn = OpenConn();
    var results = await conn.QueryAsync<StudentSummaryDto>(
        "GetStudentClassSummary",
        parameters,
        commandType: CommandType.StoredProcedure);

    var totalCount = parameters.Get<int>("@TotalCount");  // read OUTPUT
    return results;
}

🔀 Multi-mapping and Transactions

// Multi-mapping — map JOIN result to multiple C# classes
public async Task<IEnumerable<Student>> GetWithClassAndFeeAsync(int schoolId)
{
    const string sql = @"
        SELECT s.Id, s.Name, s.AdmNo,    -- Student columns first
               c.Id, c.ClassName,         -- then Class columns
               f.Id, f.PendingAmount      -- then Fee columns
        FROM Students s
        INNER JOIN Classes c ON s.ClassId = c.Id
        LEFT JOIN  Fees    f ON s.Id      = f.StudentId
        WHERE s.SchoolId = @SchoolId";

    using var conn = OpenConn();
    var results = await conn.QueryAsync<Student, Class, Fee?, Student>(
        sql,
        (student, cls, fee) => {         // map function
            student.Class = cls;
            student.Fee   = fee;
            return student;
        },
        new { SchoolId = schoolId },
        splitOn: "Id,Id");              // split at second and third 'Id' columns

    return results;
}

// Transactions — wrap multiple operations
public async Task AdmitStudentAsync(AdmitStudentRequest req)
{
    using var conn = OpenConn();
    conn.Open();
    using var txn = conn.BeginTransaction();

    try
    {
        // Insert student
        const string insertStudent = @"
            INSERT INTO Students (Name, AdmNo, ClassId, SchoolId)
            VALUES (@Name, @AdmNo, @ClassId, @SchoolId);
            SELECT CAST(SCOPE_IDENTITY() AS INT);";

        int studentId = await conn.ExecuteScalarAsync<int>(
            insertStudent, req, transaction: txn);  // pass transaction!

        // Insert fee record
        await conn.ExecuteAsync(
            "INSERT INTO Fees (StudentId, TotalAmount, Term) VALUES (@StudentId, @FeeAmount, 1)",
            new { StudentId = studentId, FeeAmount = req.FeeAmount },
            transaction: txn);

        // Insert parent record
        await conn.ExecuteAsync(
            "INSERT INTO Parents (StudentId, Name, Mobile) VALUES (@StudentId, @ParentName, @Mobile)",
            new { StudentId = studentId, req.ParentName, req.ParentMobile },
            transaction: txn);

        txn.Commit();  // all 3 succeed → commit
    }
    catch
    {
        txn.Rollback();  // any failure → undo all
        throw;
    }
}

// Multiple result sets with QueryMultiple
public async Task<StudentDashboard> GetStudentDashboardAsync(int studentId)
{
    const string sql = @"
        SELECT * FROM Students WHERE Id = @Id;           -- result set 1
        SELECT * FROM Fees WHERE StudentId = @Id;        -- result set 2
        SELECT * FROM Attendance WHERE StudentId = @Id
        ORDER BY Date DESC;";                         -- result set 3

    using var conn = OpenConn();
    using var multi = await conn.QueryMultipleAsync(sql, new { Id = studentId });

    var student    = await multi.ReadSingleOrDefaultAsync<Student>();
    var fees       = (await multi.ReadAsync<Fee>()).ToList();
    var attendance = (await multi.ReadAsync<Attendance>()).ToList();

    return new StudentDashboard(student, fees, attendance);
}

🎯 10 Interview Scenarios

1
What is Dapper and how is it different from EF Core?Concept

Dapper: lightweight "micro-ORM." Extension methods on IDbConnection. You write SQL, Dapper maps results to C# objects. No change tracking, no migration system, no LINQ query building. EF Core: full ORM. Change tracking, migrations, LINQ-to-SQL, lazy loading. More magic, less SQL control. ZenCampus: EF Core for standard CRUD (Student, Fee entities). Dapper for complex reports: multi-table JOINs, aggregations, stored procedures where you need exact SQL. Often used together in same project.

2
What is the difference between Query() and QuerySingleOrDefault()?Methods

Query<T>(): returns IEnumerable<T>. Reads all rows. For any number of results. QuerySingleOrDefault<T>(): expects exactly 0 or 1 result. Returns null if not found. THROWS InvalidOperationException if query returns more than 1 row. QueryFirstOrDefault<T>(): returns first row regardless of total count. Never throws. Use QuerySingleOrDefault when: looking up by unique key (PK, unique column) where multiple results would indicate a data integrity problem.

3
How does Dapper prevent SQL injection?Security

Parameterized queries. Pass C# object or anonymous type as parameters. Dapper sends SQL + parameters to DB separately — DB never concatenates user input into SQL. SAFE: conn.Query("SELECT * FROM Students WHERE Name = @Name", new { Name = nameInput }). DANGEROUS: conn.Query($"SELECT * FROM Students WHERE Name = '{nameInput}'") — string concatenation allows injection. NEVER build SQL by string concatenation. Always use @parameter syntax. Dapper does NOT protect you from string concatenation — that's your responsibility.

4
How do you use DynamicParameters for optional filter queries?Advanced

Use StringBuilder to build WHERE clause dynamically based on which filters are provided. DynamicParameters to add only the needed parameters. var p = new DynamicParameters(); p.Add("SchoolId", schoolId); if (name != null) { sql.Append(" AND Name LIKE @Name"); p.Add("Name", $"%{name}%"); } Never build: WHERE 1=1 with string-interpolated values — always parameterize. DynamicParameters also handles OUTPUT/RETURN parameters from stored procedures.

5
How does Dapper map SQL columns to C# properties?Mapping

Case-insensitive column name matching. SQL column "StudentName" maps to C# property "StudentName" or "studentname" or "STUDENTNAME". Uses column aliases in SQL for mapping: SELECT s.Name AS StudentName → maps to property StudentName. For mismatches: use column aliases in SQL (don't rename C# properties). Custom mapping: add [Column("sql_column_name")] attribute or configure TypeHandler. Most common fix: add alias in SQL — SELECT FirstName + ' ' + LastName AS FullName → maps to FullName property.

6
Explain multi-mapping in Dapper with an example.Advanced

Multi-mapping: map a single JOIN query result to multiple C# objects and combine them. conn.Query<Student, Class, Student>(sql, (s, c) => { s.Class = c; return s; }, splitOn: "ClassId"). splitOn tells Dapper where one object ends and next begins — the column name where split happens. Each type in generic params corresponds to columns in SELECT. Useful when loading parent with child objects in one query instead of N+1. ZenCampus: Get Student + Class + Fee data in one query.

7

8
How do you handle transactions in Dapper?Transactions

1) Open connection explicitly: conn.Open(). 2) BeginTransaction() to start. 3) Pass transaction to every query/execute: conn.Execute(sql, params, transaction: txn). 4) txn.Commit() on success. 5) txn.Rollback() in catch. ZenCampus: Admit student → create fee record → create parent record — all in one transaction. If fee insert fails, student insert rolled back too. No orphan students without fees.

9
What is QueryMultiple and when do you use it?Advanced

QueryMultiple: execute SQL with multiple SELECT statements in one round-trip to DB. Returns GridReader. Call Read<T>() for each result set in order. Benefits: one network round-trip instead of 3-4. Performance gain for dashboard-type pages that need data from multiple unrelated queries. ZenCampus student dashboard: SELECT student details; SELECT fees; SELECT attendance — all in one SQL, one connection round-trip. Read each in order: multi.ReadSingle(), multi.Read(), multi.Read().

10
When would you choose Dapper over EF Core in ZenCampus?Architecture

Choose Dapper when: 1) Complex reports — fee summary per class per term with multiple aggregations. 2) Performance-critical queries — attendance report for 10,000 students. 3) Stored procedures — legacy DB or DBA-managed procs. 4) Bulk operations — insert 1000 attendance records efficiently. 5) Complex JOINs that EF Core translates poorly. Choose EF Core when: standard CRUD, need change tracking, need migrations, team less comfortable with raw SQL. ZenCampus: EF Core for Student/Fee/Attendance entities; Dapper for school-wide reports and analytics.
