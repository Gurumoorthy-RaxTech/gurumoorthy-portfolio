1Reverse a String🔥 MSC interview la kettanga
"Hello" kudукка → "olleH" nu return pannuра function எழுது. Rule: .Reverse() built-in method use pannாma, manual logic வெச்சு pannумури interviewer kேлуvार்.
C#
public static string ReverseString(string input)
{
    char[] chars = input.ToCharArray();
    int left = 0, right = chars.Length - 1;

    while (left < right)
    {
        // swap left and right characters
        char temp = chars[left];
        chars[left] = chars[right];
        chars[right] = temp;

        left++;
        right--;
    }
    return new string(chars);
}

// Test
Console.WriteLine(ReverseString("Hello")); // olleH
JavaScript
function reverseString(str) {
  let arr = str.split('');
  let left = 0, right = arr.length - 1;

  while (left < right) {
    // swap left and right characters
    [arr[left], arr[right]] = [arr[right], arr[left]];
    left++;
    right--;
  }
  return arr.join('');
}

console.log(reverseString("Hello")); // olleH

// Shortcut (interviewer built-in method um sollanum na)
const reverseShort = str => str.split('').reverse().join('');

Nடுவला meetum varaikkும் idha repeat pannрோम். Example: "Hello" — H↔o swap → "oellH", apparam e↔l swap → "olleH", nடுவла "l" mattum bakiа irukку, adhu அப்படியே irukрum. 2Check Palindrome String
Oru string, front la irundhu padiсாலும் back la irundhu padiсாலும் same ah irukрा nu check panрра function ("madam", "level" ellame palindrome).
C#
public static bool IsPalindrome(string input)
{
    string clean = input.ToLower().Replace(" ", "");
    int left = 0, right = clean.Length - 1;

    while (left < right)
    {
        if (clean[left] != clean[right]) return false;
        left++;
        right--;
    }
    return true;
}

Console.WriteLine(IsPalindrome("Madam"));  // true
Console.WriteLine(IsPalindrome("Hello"));  // false
JavaScript
function isPalindrome(str) {
  const clean = str.toLowerCase().replaceAll(' ', '');
  let left = 0, right = clean.length - 1;

  while (left < right) {
    if (clean[left] !== clean[right]) return false;
    left++;
    right--;
  }
  return true;
}

console.log(isPalindrome("Madam"));  // true
console.log(isPalindrome("Hello"));  // false

4Find Duplicate Elements in an Array
[1,2,3,2,4,5,1] kудுтта, repeat aana elements ah find pannанும் — output: [1,2].
C#
public static List<int> FindDuplicates(int[] arr)
{
    var seen = new HashSet<int>();
    var duplicates = new HashSet<int>();

    foreach (int num in arr)
    {
        if (!seen.Add(num))       // Add() returns false if already present
            duplicates.Add(num);
    }
    return duplicates.ToList();
}

var result = FindDuplicates(new[] {1,2,3,2,4,5,1});
Console.WriteLine(string.Join(",", result)); // 1,2
JavaScript
function findDuplicates(arr) {
  const seen = new Set();
  const duplicates = new Set();

  for (const num of arr) {
    if (seen.has(num)) duplicates.add(num);
    else seen.add(num);
  }
  return [...duplicates];
}

console.log(findDuplicates([1,2,3,2,4,5,1])); // [1, 2]
🗣️ Thanglish Logic ExplanationHashSet/Set use pannрோम் edhуkки na, adhula "contains check" O(1) time la aagும் (array la irundha O(n) — ovvoru element ku full array search pannанும்). Ovvoru number ah "already seen" set la irukкா nु check pannрோм் — irundhа, adhу duplicate, "duplicates" set la add pannрோम்.
Time: O(n) · Space: O(n)
5Two Sum Problem
Array [2,7,11,15], target 9 kудுтта — evлाаவது rendு numbers add pannினа 9 aaganum, andha indices return pannанும் → output: [0,1] (edhуkки na 2+7=9).
C#
public static int[] TwoSum(int[] nums, int target)
{
    var map = new Dictionary<int, int>(); // value -> index

    for (int i = 0; i < nums.Length; i++)
    {
        int complement = target - nums[i];
        if (map.ContainsKey(complement))
            return new int[] { map[complement], i };

        map[nums[i]] = i;
    }
    return Array.Empty<int>();
}

var result = TwoSum(new[] {2,7,11,15}, 9);
Console.WriteLine($"[{result[0]}, {result[1]}]"); // [0, 1]
JavaScript
function twoSum(nums, target) {
  const map = new Map(); // value -> index

  for (let i = 0; i < nums.length; i++) {
    const complement = target - nums[i];
    if (map.has(complement)) {
      return [map.get(complement), i];
    }
    map.set(nums[i], i);
  }
  return [];
}

console.log(twoSum([2,7,11,15], 9)); // [0, 1]
🗣️ Thanglish Logic ExplanationBrute-force la rendு loops vेчु ovvoru pair ah check panна O(n²) aagும். Better approach — oru "Dictionary/Map" vечу, ovvoru number ah loop pannрумбोदు, "target - current number" (complement) already map la iruккா nு check pannрோम். Iருந்தа — andha rendு numbers thaan answer. Illна, current number ah map la add panни munनேறுрோм். Idhу single pass la (O(n)) solve aagும் — idha interview la mention pannினа impress aagும்.
Time: O(n) · Space: O(n)
6Check Anagram
Rendு strings kudукка, oru string oda letters ah rearrange panнा vera string vaрума nу check pannанும் ("listen" and "silent" = anagram).
C#
public static bool IsAnagram(string s1, string s2)
{
    if (s1.Length != s2.Length) return false;

    char[] arr1 = s1.ToLower().ToCharArray();
    char[] arr2 = s2.ToLower().ToCharArray();
    Array.Sort(arr1);
    Array.Sort(arr2);

    return new string(arr1) == new string(arr2);
}

Console.WriteLine(IsAnagram("listen", "silent")); // true
JavaScript
function isAnagram(s1, s2) {
  if (s1.length !== s2.length) return false;

  const sort = s => s.toLowerCase().split('').sort().join('');
  return sort(s1) === sort(s2);
}

console.log(isAnagram("listen", "silent")); // true
🗣️ Thanglish Logic ExplanationSimplest approach — rendайும் letters ah alphabetical order la sort pannிட்டு compare pannрது. Sort pannினா, "listen" ah "eilnst" aaguম், "silent" um "eilnst" thaan aaguম் — match aagும். Vera approach — character frequency count (Dictionary la ovvoru letter evлो тடवा vарутு nु count pannி compare pannрது) — idhu O(n), sorting approach O(n log n).
Time: O(n log n) with sorting

8Fibonacci Series

public static void PrintFibonacci(int n)
{
    int a = 0, b = 1;
    for (int i = 0; i < n; i++)
    {
        Console.Write(a + " ");
        int next = a + b;
        a = b;
        b = next;

JavaScript
function printFibonacci(n) {
  let a = 0, b = 1;
  const result = [];
  for (let i = 0; i < n; i++) {
    result.push(a);
    [a, b] = [b, a + b];
  }
  return result;
}

console.log(printFibonacci(8)); // [0,1,1,2,3,5,8,13]
🗣️ Thanglish Logic ExplanationIdhu "iterative" approach — rendு variables (a, b) vechु current um next number-ум track pannрோம், loop la roll pannрோम். 10Reverse an Array (In-Place)
[1,2,3,4,5] → [5,4,3,2,1] extra array create pannаma, same array la reverse pannанум்.
C#
public static void ReverseArray(int[] arr)
{
    int left = 0, right = arr.Length - 1;
    while (left < right)
    {
        (arr[left], arr[right]) = (arr[right], arr[left]); // tuple swap
        left++;
        right--;
    }
}

ReverseArray(nums);
Console.WriteLine(string.Join(",", nums)); // 5,4,3,2,1

    left++;
    right--;
  }
  return arr;
}

13String Compression
"aabcccccaaa" → "a2b1c5a3" — repeat aana characters ah count vechа compress pannанும்.
C#
public static string CompressString(string input)
{
    var sb = new System.Text.StringBuilder();
    int count = 1;

    for (int i = 1; i <= input.Length; i++)
    {
        if (i < input.Length && input[i] == input[i - 1])
        {
            count++;
        }
        else
        {
            sb.Append(input[i - 1]);
            sb.Append(count);
            count = 1;
        }
    }
    return sb.ToString();
}

Console.WriteLine(CompressString("aabcccccaaa")); // a2b1c5a3
JavaScript
function compressString(str) {
  let result = '';
  let count = 1;

  for (let i = 1; i <= str.length; i++) {
    if (i < str.length && str[i] === str[i - 1]) {
      count++;
    } else {
      result += str[i - 1] + count;
      count = 1;
    }
  }
  return result;
}

console.log(compressString("aabcccccaaa")); // a2b1c5a3
🗣️ Thanglish Logic ExplanationString ah loop pannி, current character munNadi character kூடa same-a nу check pannрோम் — same na count increment, differenta — "character + count" ah result la add panни, count ah 1 ku reset pannрோм். Loop i <= length varaikkும் pogрது mукkийam — last character group ah kூடa append pannанும்нала.
Time: O(n) · Space: O(n)
14Find the Missing Number (1 to N)
[1,2,4,5,6] — 1 to 6 la, oru number missing (idhula 3 missing). Adha kandுpiдிanум்.
C#
public static int FindMissingNumber(int[] arr, int n)
{
    int expectedSum = n * (n + 1) / 2;   // 1 to n sum formula
    int actualSum = arr.Sum();
    return expectedSum - actualSum;
}

Console.WriteLine(FindMissingNumber(new[] {1,2,4,5,6}, 6)); // 3
JavaScript
function findMissingNumber(arr, n) {
  const expectedSum = (n * (n + 1)) / 2;
  const actualSum = arr.reduce((a, b) => a + b, 0);
  return expectedSum - actualSum;
}

console.log(findMissingNumber([1,2,4,5,6], 6)); // 3
🗣️ Thanglish Logic ExplanationIdhu oru "math trick" — 1 to N varaikkும் sum ku formula irukку: n(n+1)/2. 15Find the Second Largest Number in an Array
[10, 5, 20, 8, 15] — largest 20, second largest 15. Sort panнாma find pannанум் (interview la sort panна koodathu nு kekkலाம்).
C#
public static int SecondLargest(int[] arr)
{
    int first = int.MinValue, second = int.MinValue;

    foreach (int num in arr)
    {
        if (num > first)
        {
            second = first;
            first = num;
        }
        else if (num > second && num != first)
        {
            second = num;
        }
    }
    return second;
}

Console.WriteLine(SecondLargest(new[] {10,5,20,8,15})); // 15
JavaScript
function secondLargest(arr) {
  let first = -Infinity, second = -Infinity;

  for (const num of arr) {
    if (num > first) {
      second = first;
      first = num;
    } else if (num > second && num !== first) {
      second = num;
    }
  }
  return second;
}

console.log(secondLargest([10,5,20,8,15])); // 15
🗣️ Thanglish Logic ExplanationRendு variables (first, second) vечу track pannрோm். Single pass la — puthु number, current largest (first) vida periyа na, pazhaya first ah second ku push pannிட்டு, puthу number ah first ah aакрோm். Illना, second vida periyа (aana first ku equal illama) na, second ah update pannрோm். Sort pannினа O(n log n) aagும், idhу single pass O(n) — performance point ah interview la sollлाம்.
Time: O(n) · Space: O(1)
16Sum of Digits of a Number

public static int SumOfDigits(int n)
{
    int sum = 0;
    n = Math.Abs(n);
    while (n > 0)
    {
        sum += n % 10;   // last digit eduthu sum ku add pannrom
        n /= 10;          // last digit ah remove pannrom
    }
    return sum;
}

    n = Math.floor(n / 10);
  }
  return sum;

public static void BubbleSort(int[] arr)
{
    int n = arr.Length;
    for (int i = 0; i < n - 1; i++)
    {
        for (int j = 0; j < n - i - 1; j++)
        {
            if (arr[j] > arr[j + 1])
            {
                (arr[j], arr[j + 1]) = (arr[j + 1], arr[j]); // swap
            }
        }
    }
}

public static int BinarySearch(int[] arr, int target)
{
    int left = 0, right = arr.Length - 1;

    while (left <= right)
    {
        int mid = left + (right - left) / 2;
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }

}

public static string TitleCase(string sentence)
{
    string[] words = sentence.Split(' ');
    for (int i = 0; i < words.Length; i++)
    {
        if (words[i].Length > 0)
            words[i] = char.ToUpper(words[i][0]) + words[i].Substring(1).ToLower();
    }
    return string.Join(" ", words);
}

Console.WriteLine(TitleCase("hello world from chennai")); // Hello World From Chennai
