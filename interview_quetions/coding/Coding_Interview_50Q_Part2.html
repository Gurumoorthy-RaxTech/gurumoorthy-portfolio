int CountOccurrences(string text, string pattern)
{
    int count = 0, index = 0;
    while ((index = text.IndexOf(pattern, index)) != -1)
    {
        count++;
        index += pattern.Length;
    }
    return count;
}

bool IsRotation(string s1, string s2)
{
    if (s1.Length != s2.Length) return false;

6Check if brackets are balanced: "({[]})" → valid, "([)]" → invalidLogic▼
bool IsBalanced(string s)
{
    var stack = new Stack<char>();
    var pairs = new Dictionary<char,char>
    {
        [')'] = '(',
        [']'] = '[',
        ['}'] = '{'
    };
    foreach (char c in s)
    {
        if ("([{".Contains(c))
            stack.Push(c);
        else if (pairs.ContainsKey(c))
        {
            if (stack.Count == 0 || stack.Pop() != pairs[c])
                return false;
        }
    }
    return stack.Count == 0;
}

Console.WriteLine(IsBalanced("({[]})")); // True
Console.WriteLine(IsBalanced("([)]"));   // False
True / False
💡 Very common in senior interviews. Uses Stack data structure.
7Compress a string: "aaabbbcc" → "a3b3c2"C#▼
string Compress(string s)
{
    var sb = new System.Text.StringBuilder();
    int i = 0;
    while (i < s.Length)
    {
        char curr = s[i];
        int count = 1;
        while (i + count < s.Length && s[i + count] == curr)
            count++;
        sb.Append(curr);
        if (count > 1) sb.Append(count);
        i += count;
    }
    return sb.ToString();
}

Console.WriteLine(Compress("aaabbbcc"));  // a3b3c2
Console.WriteLine(Compress("abcd"));      // abcd (no compression needed)

8Convert string number to integer without int.Parse (implement atoi)Logic▼
int MyAtoi(string s)
{
    s = s.Trim();
    int sign = 1, result = 0, i = 0;

    if (i < s.Length && (s[i] == '-' || s[i] == '+'))
        sign = s[i++] == '-' ? -1 : 1;

    while (i < s.Length && char.IsDigit(s[i]))
    {
        result = result * 10 + (s[i] - '0');
        i++;
    }
    return sign * result;
}

Console.WriteLine(MyAtoi("  -123")); // -123
Console.WriteLine(MyAtoi("42abc"));  // 42
-123 / 42

9Find maximum sum of contiguous subarray (Kadane's Algorithm)Logic▼
int MaxSubarraySum(int[] arr)
{
    int maxSoFar = arr[0];
    int maxEndingHere = arr[0];

    for (int i = 1; i < arr.Length; i++)
    {
        maxEndingHere = Math.Max(arr[i], maxEndingHere + arr[i]);
        maxSoFar = Math.Max(maxSoFar, maxEndingHere);
    }
    return maxSoFar;
}

int[] arr = { -2,1,-3,4,-1,2,1,-5,4 };
Console.WriteLine(MaxSubarraySum(arr));
6   (subarray: 4,-1,2,1)
💡 Very famous algorithm. If asked "max sum subarray" — this is the answer.
10Move all zeros to end of array (keep order of non-zeros)C#▼
void MoveZerosToEnd(int[] arr)
{
    int insertPos = 0;
    // Move all non-zero to front
    foreach (int n in arr)
        if (n != 0) arr[insertPos++] = n;

    // Fill rest with zeros
    while (insertPos < arr.Length)
        arr[insertPos++] = 0;
}

int[] arr = { 1,0,3,0,5,0,7 };
MoveZerosToEnd(arr);

1, 3, 5, 7, 0, 0, 0
11Rotate an array to the right by K positionsC#▼
int[] RotateRight(int[] arr, int k)
{
    int n = arr.Length;
    k = k % n; // handle k > n
    var result = new int[n];
    for (int i = 0; i < n; i++)
        result[(i + k) % n] = arr[i];
    return result;
}

int[] arr = { 1,2,3,4,5 };
Console.WriteLine(string.Join(", ", RotateRight(arr, 2)));
4, 5, 1, 2, 3
12Binary Search – find index of element in sorted arrayLogic▼
int BinarySearch(int[] arr, int target)
{
    int left = 0, right = arr.Length - 1;
    while (left <= right)
    {
        int mid = left + (right - left) / 2; // avoids overflow
        if      (arr[mid] == target) return mid;
        else if (arr[mid] < target)  left  = mid + 1;
        else                         right = mid - 1;
    }
    return -1; // not found
}

int[] sorted = { 2,5,8,12,16,23,38 };
Console.WriteLine(BinarySearch(sorted, 23)); // index 5
Console.WriteLine(BinarySearch(sorted, 10)); // -1
5 / -1
💡 Binary search = O(log n). Linear search = O(n). Always mention time complexity.

15Find GCD and LCM of two numbersLogic▼
int GCD(int a, int b)
{
    while (b != 0)
    {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

int LCM(int a, int b)
    => (a / GCD(a, b)) * b; // divide first to avoid overflow

Console.WriteLine(GCD(48, 18)); // 6
Console.WriteLine(LCM(4, 6));   // 12

16Power of a number using recursion: 2^10Recursion▼
// Fast power: O(log n) — divide exponent by 2 each time
double Power(double base_, int exp)
{
    if (exp == 0) return 1;
    if (exp < 0)  return 1 / Power(base_, -exp);

    if (exp % 2 == 0)
    {
        double half = Power(base_, exp / 2);
        return half * half; // calculate once, square it
    }
    return base_ * Power(base_, exp - 1);
}

Console.WriteLine(Power(2, 10)); // 1024
Console.WriteLine(Power(3, 4));  // 81
1024 / 81

void Hanoi(int n, string from, string to, string aux)
{

    {
        Console.WriteLine($"Move disk 1 from {from} to {to}");

26Get Nth highest salary (generic — works for any N)SQL▼
-- N = 3 (3rd highest)
SELECT Salary FROM (
    SELECT DISTINCT Salary,
           DENSE_RANK() OVER (ORDER BY Salary DESC) AS Rnk
    FROM Employees
) t
WHERE Rnk = 3; -- change 3 to any N

-- Or with OFFSET
SELECT DISTINCT Salary
FROM Employees
ORDER BY Salary DESC
OFFSET 2 ROWS FETCH NEXT 1 ROWS ONLY; -- OFFSET N-1

    s.Phone,

31Difference between WHERE and HAVINGSQL▼
-- WHERE: filters ROWS before grouping
SELECT Department, AVG(Salary) AS AvgSal
FROM Employees
WHERE IsActive = 1          -- filters rows first
GROUP BY Department;

-- HAVING: filters GROUPS after aggregation
SELECT Department, AVG(Salary) AS AvgSal
FROM Employees
GROUP BY Department
HAVING AVG(Salary) > 60000;   -- filters groups

-- Combined
SELECT Department, COUNT(*) AS EmpCount, AVG(Salary) AS AvgSal
FROM Employees
WHERE IsActive = 1             -- WHERE first
GROUP BY Department
HAVING COUNT(*) >= 5;         -- HAVING after
32Write a trigger – auto log when salary is updatedSQL▼
-- Create audit table first
CREATE TABLE SalaryAudit (
    AuditId    INT PRIMARY KEY IDENTITY,
    EmployeeId INT,
    OldSalary  DECIMAL(10,2),
    NewSalary  DECIMAL(10,2),
    ChangedOn  DATETIME DEFAULT GETDATE(),
    ChangedBy  NVARCHAR(100) DEFAULT SYSTEM_USER
)

-- Trigger
CREATE TRIGGER trg_SalaryUpdate
ON Employees
AFTER UPDATE
AS
BEGIN
    IF UPDATE(Salary) -- only when salary column changed
    BEGIN
        INSERT INTO SalaryAudit (EmployeeId, OldSalary, NewSalary)
        SELECT d.EmployeeId, d.Salary, i.Salary
        FROM DELETED d  -- old values
        JOIN  INSERTED i ON d.EmployeeId = i.EmployeeId
        WHERE d.Salary != i.Salary;
    END
END
33Explain indexes – when NOT to use an indexSQL▼

When indexes HELP:

  • Columns used in WHERE, JOIN, ORDER BY frequently
  • High cardinality columns (many unique values)
  • Large tables with frequent SELECT

When indexes HURT (don't add index):

  • Small tables — full scan is faster
  • Columns with very few distinct values (IsActive: only 0/1)
  • Tables with heavy INSERT/UPDATE/DELETE — indexes slow writes
  • Columns rarely used in queries

35Implement retry logic for failed API callsC#▼
async Task<T> RetryAsync<T>(Func<Task<T>> operation, int maxRetries = 3)
{
    int attempt = 0;
    while (true)
    {
        try
        {
            return await operation();
        }
        catch (Exception ex) when (attempt < maxRetries - 1)
        {
            attempt++;
            int delay = (int)Math.Pow(2, attempt) * 1000; // 2s, 4s, 8s
            Console.WriteLine($"Attempt {attempt} failed. Retrying in {delay}ms...");
            await Task.Delay(delay);
        }
    }
}

// Usage
var result = await RetryAsync(() => _smsService.SendOtpAsync(phone));
💡 This is exactly what your OTP/SMS system should use. Mention exponential backoff.

47What is StringBuilder vs string concatenation? When to use?C#▼
// BAD — creates new string object every iteration (N strings in memory)

for (int i = 0; i < 10000; i++)
    result += i.ToString(); // 10000 string objects created!

for (int i = 0; i < 10000; i++)
    sb.Append(i);

49Implement a simple cache using Dictionary with expiryC#▼
public class SimpleCache<TKey, TValue>
{
    private class CacheItem
    {
        public TValue Value    { get; set; }
        public DateTime Expiry { get; set; }
    }

    private readonly Dictionary<TKey, CacheItem> _cache = new();

    public void Set(TKey key, TValue value, TimeSpan ttl)
        => _cache[key] = new CacheItem { Value = value, Expiry = DateTime.UtcNow + ttl };

    public bool TryGet(TKey key, out TValue value)
    {
        if (_cache.TryGetValue(key, out var item) && item.Expiry > DateTime.UtcNow)
        {
            value = item.Value;
            return true;
        }
        value = default;
        return false;
    }
}

// Usage
var cache = new SimpleCache<string, string>();
cache.Set("student_1", "Ravi Kumar", TimeSpan.FromMinutes(5));

if (cache.TryGet("student_1", out var name))
    Console.WriteLine($"From cache: {name}");
50Given a list of transactions, find which day had highest total spend using LINQ + GroupByLINQ▼
var transactions = new[]

    Console.WriteLine($"  {d.Date}: ₹{d.Total}"));
