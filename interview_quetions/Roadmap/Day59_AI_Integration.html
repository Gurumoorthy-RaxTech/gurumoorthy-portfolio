// Simple chat completion
public class ZenCampusAiService(AzureOpenAIClient _ai,
    IStudentRepository _students)
{
    public async Task<string> AnswerParentQueryAsync(
        int studentId, string question, CancellationToken ct = default)
    {
        // 1. Fetch student context from DB
        var student = await _students.GetByIdAsync(studentId, ct);
        var context  = $"""
            Student: {student!.Name}
            Class: {student.ClassName}
            Total Fees: {student.TotalFees:C}
            Paid: {student.PaidAmount:C}
            Pending: {(student.TotalFees - student.PaidAmount):C}
            Attendance: {student.AttendancePercentage}%
            """;

        // 2. Send to GPT-4 with context (RAG principle)
        var chatClient = _ai.GetChatClient("gpt-4o");
        var messages = new []
        {
            new SystemChatMessage("""
                You are ZenCampus AI assistant. Answer parent queries about their child.
                Be concise, accurate, and friendly. Respond in the same language as the question.
                Only use the provided context. Never make up information.
                """),
            new UserChatMessage($"Context:\n{context}\n\nQuestion: {question}")
        };

        var response = await chatClient.CompleteChatAsync(messages,
            new ChatCompletionOptions { MaxOutputTokenCount = 500 }, ct);
        return response.Value.Content[0].Text;
    }
}

// Program.cs — Register Semantic Kernel
var kernel = Kernel.CreateBuilder()
    .AddAzureOpenAIChatCompletion(
        deploymentName: "gpt-4o",
        endpoint:       config["AzureOpenAI:Endpoint"]!,
        credential:     new DefaultAzureCredential())
    .Build();

builder.Services.AddSingleton(kernel);

// Semantic Kernel Plugin — native functions the AI can call
public class ZenCampusPlugin
{
    private readonly IStudentRepository _students;
    private readonly IFeeRepository     _fees;

    [KernelFunction, Description("Get student attendance percentage for current month")]
    public async Task<string> GetStudentAttendanceAsync(
        [Description("Student admission number")] string admissionNo)
    {
        var student = await _students.GetByAdmissionNoAsync(admissionNo);
        if (student is null) return "Student not found";
        return $"{student.Name} attendance: {student.AttendancePercentage}% this month";
    }

    [KernelFunction, Description("Get pending fee amount for a student")]
    public async Task<string> GetPendingFeesAsync(
        [Description("Student admission number")] string admissionNo)
    {
        var fee = await _fees.GetPendingByAdmissionNoAsync(admissionNo);
        return $"Pending fee: Rs {fee?.PendingAmount ?? 0}";
    }
}

// Use with auto function calling (SK calls tools automatically)
var kernel = app.Services.GetRequiredService<Kernel>();
kernel.Plugins.AddFromType<ZenCampusPlugin>("ZenCampus");

var result = await kernel.InvokePromptAsync(
    "What is the pending fee for student ADM2026001?",
    new KernelArguments(new OpenAIPromptExecutionSettings
    {
        FunctionChoiceBehavior = FunctionChoiceBehavior.Auto() // SK calls tools automatically
    }));
// SK: calls GetPendingFeesAsync("ADM2026001") → returns result to GPT → natural language answer

// Generate embedding for semantic search
public class ZenCampusKnowledgeBase(AzureOpenAIClient _ai,
    ZenCampusDbContext _db)
{
    // Store: embed each FAQ/policy document
    public async Task IndexDocumentAsync(string title, string content)
    {
        var embeddingClient = _ai.GetEmbeddingClient("text-embedding-3-small");
        var embedding = await embeddingClient.GenerateEmbeddingAsync(content);

        _db.KnowledgeDocuments.Add(new KnowledgeDocument
        {
            Title     = title,
            Content   = content,
            Embedding = embedding.Value.ToFloats().ToArray()  // 1536 dimensions
        });
        await _db.SaveChangesAsync();
    }

    // Retrieve: find most similar documents to a query
    public async Task<List<string>> SearchAsync(string query, int topK = 3)
    {
        var embeddingClient = _ai.GetEmbeddingClient("text-embedding-3-small");
        var queryEmbedding = await embeddingClient.GenerateEmbeddingAsync(query);
        var queryVec = queryEmbedding.Value.ToFloats();

        // Azure SQL vector search (or pgvector, or Azure AI Search)
        var docs = await _db.KnowledgeDocuments
            .OrderByDescending(d => VectorFunctions.CosineSimilarity(d.Embedding, queryVec))
            .Take(topK)
            .Select(d => d.Content)
            .ToListAsync();
        return docs;
    }
}

💬 ZenCampus AI Chatbot API Endpoint

// ChatController.cs
[ApiController]
[Route("api/[controller]")]
[Authorize]
public class ChatController(ZenCampusAiService _ai,
    ZenCampusKnowledgeBase _kb) : ControllerBase
{
    [HttpPost("parent-query")]
    public async Task<IActionResult> ParentChatAsync(ParentChatRequest req)
    {
        var parentId  = User.GetParentId();   // JWT claim
        var studentId = User.GetStudentId();  // parent's child

        // RAG: find relevant policies/FAQs
        var relevantDocs = await _kb.SearchAsync(req.Question);
        var answer = await _ai.AnswerParentQueryAsync(studentId, req.Question, relevantDocs);

        return Ok(new { Answer = answer, Source = "ZenCampus AI" });
    }

    // Streaming response (typing effect)
    [HttpPost("stream")]
    public async Task StreamChatAsync(ParentChatRequest req)
    {
        Response.Headers.ContentType = "text/event-stream";
        await foreach (var chunk in _ai.StreamAnswerAsync(req.Question))
        {
            await Response.WriteAsync($"data: {chunk}\n\n");
            await Response.Body.FlushAsync();
        }
    }
}

public record ParentChatRequest(string Question);

🎯 10 Interview Scenarios

1
What is RAG and why is it better than fine-tuning for ZenCampus?Concept

RAG (Retrieval Augmented Generation): retrieve relevant data from your own sources, inject into the LLM prompt. The model generates answers grounded in your data. Better than fine-tuning for ZenCampus because: student data changes daily (fine-tuning is static). Answers must be accurate and school-specific (RAG pulls live DB data). Fine-tuning: expensive, takes hours, still hallucinates. RAG: real-time, accurate, auditable (can show source).

2
What is Semantic Kernel and how does it differ from calling OpenAI API directly?Concept

Direct OpenAI API: you send messages, get text back. Manual function calling. No orchestration. Semantic Kernel: Microsoft's AI orchestration SDK. Register your C# methods as [KernelFunction]. SK automatically calls your functions when the AI needs data. Handles: function calling, chaining multiple AI calls, prompt templates, memory/context, multiple AI providers. For ZenCampus: SK calls GetPendingFees() automatically when parent asks about fees — no manual parsing.

3
What are embeddings and what makes them useful for semantic search?Concept

Embeddings: convert text to a vector of numbers (e.g., 1536 dimensions for text-embedding-3-small). Semantically similar texts → vectors that are numerically close (cosine similarity ≈ 1). "Fee payment" and "outstanding dues" → similar vectors even though different words. For ZenCampus: parent asks "what money does my son owe?" — embedding search finds fee FAQ documents even without exact keyword match. Traditional SQL LIKE '%fee%' would miss this.

4
How do you prevent the AI chatbot from making up information about students?Practical

System prompt guardrail: "Only use the provided context. Never make up information. If you don't know, say 'I don't have that information.'" Inject actual DB data as context in every prompt. Grounding = hallucination prevention. Additionally: fact-check critical answers programmatically (fee amounts should match DB). Log all AI interactions. User feedback loop to catch wrong answers. Temperature = 0 (deterministic, not creative) for factual queries.

5
How would you handle multi-turn conversation (chat history) in ZenCampus chatbot?Code

Store conversation history in Redis per session (sessionId from JWT). Each turn: load previous messages from Redis, append new user message, send full history to GPT, append AI response, save back to Redis. Limit history to last 10 turns (token budget). Redis TTL = 30 minutes (session timeout). For ZenCampus: parent asks "what's his fee?" → "how much did he pay last month?" — context maintained across turns.

6
What is function calling / tool use in OpenAI API?Code

Define functions (tools) with JSON Schema description. Send to GPT. GPT decides which function to call with what arguments. Your app executes the function. Return result to GPT. GPT generates final response. For ZenCampus: define GetStudentFees(admissionNo), GetAttendance(admissionNo), GetExamResults(admissionNo) as tools. Parent asks "how is Basha doing?" → GPT calls all three tools → generates comprehensive answer. Semantic Kernel automates this.

7
How do you manage OpenAI API costs for a 50,000-student ZenCampus chatbot?Practical

Token budgeting: limit context to relevant data (not all 50,000 students). Cache common questions/answers in Redis. Rate limiting per parent (max 20 queries/day). Use cheaper models for simple queries (GPT-4o mini for FAQ, GPT-4o only for complex analysis). Log token usage per request. Set Azure OpenAI spending limits. Async processing for non-urgent queries. Pre-generate common reports instead of real-time AI generation.

8
What is the system prompt and why is it critical for ZenCampus AI?Concept

System prompt: persistent instructions given to the AI at the start of every conversation. Sets persona, constraints, language, behavior. For ZenCampus: "You are ZenCampus AI for parents. Only discuss student information provided in context. Respond in the language the parent uses (Tamil/English). Never discuss other schools' data. Never make up fees or attendance. Escalate to school admin for urgent issues." Weak system prompt = jailbreak risk, hallucinations, privacy violations.

9
How would you test an AI chatbot integration?Testing

Integration test with real API (mark as [Trait("Category","Integration")]): send test question, verify response contains expected data (student name, fee amount). Unit test the prompt building logic (string construction, context injection). Mock the AI client for unit tests of the controller. Evaluation: golden dataset of questions + expected answers, measure accuracy. Monitor in production: log all Q&A, review flagged responses, user satisfaction ratings.

10
What is the difference between GPT-4o and text-embedding-3-small — when to use each?Concept

GPT-4o: generative model, text in → text out. Understands context, answers questions, reasons, writes. Use for: chatbot responses, summarization, translation, analysis. text-embedding-3-small: converts text to vector (numbers). No natural language output. Use for: semantic search (find similar documents), RAG (find relevant context), content recommendation. Never use embedding model for generating answers — it doesn't generate text. Never use GPT-4o for search — it's too expensive and doesn't produce vectors.
