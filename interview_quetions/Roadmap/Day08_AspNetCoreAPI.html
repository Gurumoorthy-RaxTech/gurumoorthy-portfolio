1 Controllers + Action Results — ZenCampus API
🎬 Vikram — Each agent handles specific mission, reports back

Controller = agent. Each action method = specific mission. Returns result to HQ (client). [ApiController] attribute = automatic model validation, binding source inference — ஒரு attribute, நிறைய power!

// ZenCampus StudentController — production-ready pattern
[ApiController]
[Route("api/v{version:apiVersion}/[controller]")]
[ApiVersion("1.0")]
[Produces("application/json")]
public class StudentsController : ControllerBase  // NOT Controller — no Views needed!
{
    private readonly IStudentService _svc;
    private readonly ILogger<StudentsController> _log;

    public StudentsController(IStudentService svc, ILogger<StudentsController> log)
        => (_svc, _log) = (svc, log);

    // GET api/v1/students?page=1&size=20
    [HttpGet]
    [ProducesResponseType(typeof(PagedResult<StudentDto>), 200)]
    public async Task<ActionResult<PagedResult<StudentDto>>> GetStudents(
        [FromQuery] int page = 1,
        [FromQuery] int size = 20)
    {
        var result = await _svc.GetPagedAsync(page, size);
        return Ok(result);
    }

    // GET api/v1/students/42
    [HttpGet("{id:int}")]
    [ProducesResponseType(typeof(StudentDto), 200)]
    [ProducesResponseType(404)]
    public async Task<ActionResult<StudentDto>> GetStudent(int id)
    {
        var student = await _svc.GetByIdAsync(id);
        return student is null ? NotFound() : Ok(student);
    }

    // POST api/v1/students
    [HttpPost]
    [ProducesResponseType(typeof(StudentDto), 201)]
    [ProducesResponseType(typeof(ValidationProblemDetails), 400)]
    public async Task<ActionResult<StudentDto>> CreateStudent(
        [FromBody] CreateStudentDto dto)
    {
        var created = await _svc.CreateAsync(dto);
        return CreatedAtAction(
            nameof(GetStudent),
            new { id = created.Id },
            created); // 201 + Location header!
    }

    // DELETE api/v1/students/42
    [HttpDelete("{id:int}")]
    [ProducesResponseType(204)]
    [ProducesResponseType(404)]
    public async Task<IActionResult> DeleteStudent(int id)
    {
        var deleted = await _svc.DeleteAsync(id);
        return deleted ? NoContent() : NotFound();
    }
}

// Custom Middleware — ZenCampus request logging
public class RequestLoggingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<RequestLoggingMiddleware> _log;

    public RequestLoggingMiddleware(RequestDelegate next,
        ILogger<RequestLoggingMiddleware> log)
        => (_next, _log) = (next, log);

    public async Task InvokeAsync(HttpContext ctx)
    {
        var sw = Stopwatch.StartNew();
        _log.LogInformation("→ {method} {path}",
            ctx.Request.Method, ctx.Request.Path);

        await _next(ctx); // call next middleware!

        sw.Stop();
        _log.LogInformation("← {status} {ms}ms",
            ctx.Response.StatusCode, sw.ElapsedMilliseconds);
    }
}

// Extension method — clean registration
public static class MiddlewareExtensions
{
    public static IApplicationBuilder UseRequestLogging(
        this IApplicationBuilder app)
        => app.UseMiddleware<RequestLoggingMiddleware>();
}

// Program.cs
app.UseRequestLogging(); // register!
💡 Use vs Run vs Map

Use — calls next middleware (normal). Run — terminal, no next (short-circuit). Map — branch by path. MapWhen — branch by condition.

// Short-circuit — maintenance mode

        await ctx.Response.WriteAsJsonAsync(new { message = "ZenCampus maintenance mode" });
        return; // short-circuit — next NOT called!
    }
    await next(ctx); // normal flow

// Action Filter — ZenCampus audit logging
public class AuditLogFilter : IAsyncActionFilter
{
    private readonly IAuditService _audit;

    public AuditLogFilter(IAuditService audit) => _audit = audit;

    public async Task OnActionExecutionAsync(
        ActionExecutingContext context,
        ActionExecutionDelegate next)
    {
        var userId  = context.HttpContext.User.Identity?.Name;
        var action  = context.ActionDescriptor.DisplayName;
        var sw      = Stopwatch.StartNew();

        var executed = await next(); // execute action!

        await _audit.LogAsync(new AuditEntry
        {
            UserId     = userId,
            Action     = action,
            StatusCode = (executed.Result as ObjectResult)?.StatusCode,
            DurationMs = sw.ElapsedMilliseconds
        });
    }
}

// Register globally — all controllers!
builder.Services.AddControllers(opts =>
    opts.Filters.Add<AuditLogFilter>());

// OR selective — attribute on controller/action
[ServiceFilter(typeof(AuditLogFilter))]  // uses DI!
public class FeeController : ControllerBase { }
// Exception Filter — global error format
public class ZenCampusExceptionFilter : IExceptionFilter
{
    private readonly ILogger<ZenCampusExceptionFilter> _log;

    public void OnException(ExceptionContext ctx)
    {
        _log.LogError(ctx.Exception, "Unhandled exception");

        ctx.Result = ctx.Exception switch
        {
            StudentNotFoundException ex => new NotFoundObjectResult(
                new { ex.Message, ex.ErrorCode }),

            UnauthorizedAccessException => new UnauthorizedObjectResult(
                new { Message = "Access denied" }),

            _ => new ObjectResult(new { Message = "Internal server error" })
                { StatusCode = 500 }
        };

        ctx.ExceptionHandled = true; // mark handled — no re-throw!
    }
}
💡 Filter vs Middleware — when to use which?

Middleware: Applies to ALL requests (static files, 404, etc). Good for logging, HTTPS, CORS.
Filter: Applies only to MVC/API controller actions. Good for auth, audit, validation — knows about controller context.

public class FutureDateAttribute : ValidationAttribute
{
    protected override ValidationResult? IsValid(object? value, ValidationContext ctx)
    {
        if (value is DateTime date && date > DateTime.Now)
            return ValidationResult.Success;
        return new ValidationResult("Exam date must be in the future");
    }
}

// IValidatableObject — cross-field validation
public class CreateExamDto : IValidatableObject
{
    public DateTime StartDate { get; set; }
    public DateTime EndDate   { get; set; }

    public IEnumerable<ValidationResult> Validate(ValidationContext ctx)
    {
        if (EndDate <= StartDate)
            yield return new ValidationResult(
                "EndDate must be after StartDate",
                [nameof(EndDate)]);
    }
}
6 Global Error Handling — Problem Details Standard
// UseExceptionHandler — global error middleware (preferred over Exception Filter)
app.UseExceptionHandler(errApp =>
    errApp.Run(async ctx =>
    {
        var ex = ctx.Features.Get<IExceptionHandlerFeature>()?.Error;

        (int status, string title) = ex switch
        {
            StudentNotFoundException  => (404, "Student not found"),
            ValidationException      => (400, "Validation failed"),
            UnauthorizedAccessException => (401, "Unauthorized"),
            _                        => (500, "An error occurred")
        };

        ctx.Response.StatusCode = status;
        await ctx.Response.WriteAsJsonAsync(new ProblemDetails
        {
            Title    = title,
            Status   = status,
            Detail   = ex?.Message,
            Instance = ctx.Request.Path
        });
    }));

// .NET 8 — AddProblemDetails() built-in
builder.Services.AddProblemDetails(opts =>
    opts.CustomizeProblemDetails = ctx =>
    {
        ctx.ProblemDetails.Extensions["traceId"] =
            ctx.HttpContext.TraceIdentifier; // for support tickets!
    });
app.UseExceptionHandler();
app.UseStatusCodePages();
💡 ProblemDetails — RFC 7807 standard error format

All errors in one consistent JSON format: {"type":"...","title":"...","status":404,"detail":"...","instance":"/api/students/99"}. Clients can parse predictably. Swagger auto-documents. ALWAYS use this in production APIs!

Middleware order wrong — Auth bypass aagudhu Real Interview Hard
📋 Scenario: "Production-la protected API-endpoints anonymous users access பண்றாங்க! [Authorize] attribute இருக்கு. Code review-la இந்த order பாத்தோம்:" 
app.UseRouting();
app.UseAuthorization();   // ← authorization
app.UseAuthentication(); // ← authentication AFTER?!
app.MapControllers();
"Why is this wrong? Fix it."
✅ ANSWER

Root Cause: Authentication must run BEFORE Authorization. Auth middleware sets HttpContext.User from JWT/cookie. Authorization reads HttpContext.User to check claims. If Auth runs after Authz — User is always anonymous → Authz always allows!

// ✅ CORRECT ORDER:
app.UseRouting();
app.UseAuthentication(); // WHO are you? → sets User
app.UseAuthorization();  // WHAT can you do? → reads User
app.MapControllers();

// Interview answer: "Authentication populates HttpContext.User.
// Authorization inspects it. Wrong order = User always null/anonymous
// = [Authorize] sees no claims = bypass. Order is CRITICAL."
⭐ Also catch: CORS before Auth!

CORS preflight (OPTIONS) must respond before auth check. Put UseCors() BEFORE UseAuthentication() and UseAuthorization().

Every API call-la School permission check — DRY panna ena? Medium
📋 Scenario: "ZenCampus-la 50+ controller actions இருக்கு. எல்லாத்திலயும் 'does user belong to this school?' check பண்றோம். Copy-paste 50 places. DRY panna ena approach?"

// Custom requirement
public record SchoolAccessRequirement(string RouteParam = "schoolId")
    : IAuthorizationRequirement;

// Handler — runs once for all [Authorize("SchoolAccess")] endpoints
public class SchoolAccessHandler
    : AuthorizationHandler<SchoolAccessRequirement>
{
    private readonly IHttpContextAccessor _http;

    protected override Task HandleRequirementAsync(
        AuthorizationHandlerContext ctx,
        SchoolAccessRequirement     req)
    {
        var httpCtx   = _http.HttpContext!;
        var schoolId  = int.Parse(httpCtx.GetRouteValue(req.RouteParam)!.ToString()!);
        var userSchool = int.Parse(ctx.User.FindFirst("SchoolId")!.Value);

        if (schoolId == userSchool)
            ctx.Succeed(req);

        return Task.CompletedTask;
    }
}

// Register
builder.Services.AddAuthorization(opts =>
    opts.AddPolicy("SchoolAccess", p =>
        p.AddRequirements(new SchoolAccessRequirement())));
builder.Services.AddScoped<IAuthorizationHandler, SchoolAccessHandler>();

// Usage — one attribute replaces 50 copy-pastes!
[Authorize(Policy = "SchoolAccess")]
[HttpGet("schools/{schoolId}/students")]
public IActionResult GetStudents(int schoolId) { ... }

// Structured logging middleware — captures everything
public class StructuredLoggingMiddleware
{
    public async Task InvokeAsync(HttpContext ctx)
    {
        var sw        = Stopwatch.StartNew();
        var correlationId = Guid.NewGuid().ToString("N")[..8];
        ctx.Response.Headers["X-Correlation-Id"] = correlationId;

            "HTTP {method} {path} {status} {ms}ms",
            ctx.Request.Method,
            ctx.Request.Path,
            ctx.Response.StatusCode,
            sw.ElapsedMilliseconds);
    }
}

Rate limiting — SMS API abuse panna someone Hard
📋 Scenario: "ZenCampus-la /api/forgot-password endpoint இருக்கு. Someone script போட்டு 1000 requests/minute அடிக்கிறாங்க. SMS provider bill ரொம்ப வந்துச்சு! Fix?"
✅ ANSWER — Per-IP Rate Limiting
builder.Services.AddRateLimiter(opts =>
{
    // Sliding window — per IP per minute
    opts.AddSlidingWindowLimiter("sms", lim =>
    {
        lim.PermitLimit       = 5;                           // 5 requests
        lim.Window            = TimeSpan.FromMinutes(1);        // per minute
        lim.SegmentsPerWindow  = 6;                           // 10-sec segments
        lim.QueueLimit         = 0;                           // no queue — reject immediately
    });

    // Per IP partitioning
    opts.AddPolicy("sms-per-ip", ctx =>
        RateLimitPartition.GetSlidingWindowLimiter(
            partitionKey: ctx.Connection.RemoteIpAddress?.ToString() ?? "anon",
            factory: _ => new SlidingWindowRateLimiterOptions
            {
                PermitLimit      = 5,
                Window           = TimeSpan.FromMinutes(1),
                SegmentsPerWindow = 6
            }));

    opts.OnRejected = async (ctx, _) =>
    {
        ctx.HttpContext.Response.StatusCode = 429;
        await ctx.HttpContext.Response.WriteAsJsonAsync(
            new { Message = "Too many requests. Try after 1 minute." });
    };
});

"ASP.NET Core API experience sollu" — Complete Answer Senior
📋 Scenario: "ASP.NET Core Web API-la real experience பண்ணிருக்கீங்களா? Architecture, middleware, filters — ZenCampus example குடு."
✅ COMPLETE INTERVIEW ANSWER

"Yes — ZenCampus: 50,000 students, 200 schools, ASP.NET Core 8 Web API."

Architecture: Thin controllers — only parse request and return response. Business logic in service layer. Repository for data access. DTOs for API contract — entities never exposed directly.

Middleware pipeline: Exception handler first, then HTTPS redirect, CORS for our React frontend, then authentication + authorization, then rate limiting on sensitive endpoints, then controllers. Order was critical — had a bug where auth ran before auth, fixed by reordering.

Filters: Global AuditLogFilter for every write operation (who, what, when, how long). Global exception filter mapping custom exceptions to ProblemDetails format. Per-controller SchoolAccess authorization policy replacing 50 copy-pasted checks.

Versioning: URL segment versioning (v1/v2). V1 unchanged for existing clients. V2 with richer StudentDto. Sunset header on deprecated versions.

Pain point solved: SMS abuse — rate limiter per IP, 5 req/min on forgot-password. Cut SMS bill by 95%.

Q2: Middleware vs Filter — key difference?

A) Middleware is faster
B) Middleware = all requests (pipeline level). Filter = MVC/API actions only, has action context.
C) Filter supports async, middleware doesn't
D) No difference
✅ Middleware handles ALL HTTP requests (including static files, 404 pages). Filters run only for controller actions and have access to ActionContext, route data, etc. Choose middleware for cross-cutting HTTP concerns, filter for action-specific logic!
❌ Middleware = pipeline level, ALL requests. Filter = MVC/API only, knows about controllers. Middleware for CORS/HTTPS, Filter for audit/authorization on actions!

Q3: AllowAnyOrigin() + AllowCredentials() — what happens?

A) Works fine — allows all origins with credentials
B) Runtime InvalidOperationException — security risk, not allowed
C) Compile error
D) Just ignores AllowCredentials
✅ ASP.NET Core throws at runtime: "The CORS protocol does not allow specifying a wildcard origin with credentials." Use WithOrigins("specific-url") when AllowCredentials() is needed — security by design!
❌ Runtime exception! AllowAnyOrigin + AllowCredentials = security hole + ASP.NET Core blocks it. Must use WithOrigins("specific") when credentials involved!

Q4: ControllerBase vs Controller — Web API-la ena use pananum?

A) ControllerBase — no View support needed, lighter
B) Controller — always use
C) No difference
D) Controller for async, ControllerBase for sync
✅ ControllerBase = base for API (no View methods). Controller extends ControllerBase and adds View(), PartialView(), etc. Web API never serves Views — ControllerBase is correct and lighter!
❌ ControllerBase for Web API! Controller inherits ControllerBase and adds View-related methods (View(), PartialView()) — unnecessary bloat for APIs. Always ControllerBase for REST APIs!
