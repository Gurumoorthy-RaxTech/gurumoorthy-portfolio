2 Dapper — Raw SQL Speed for Reports
🎬 Roja — Speed matters in a crisis. EF Core = thorough but sometimes slow for complex reports. Dapper = raw SQL speed, maps to C# objects. ZenCampus fee report: EF 1.2s → Dapper 85ms. Same data, 14x faster! 
// NuGet: Dapper
// Use Dapper for: complex reports, aggregates, multi-table JOINs, stored procs
// Use EF Core for: CRUD, simple queries where change tracking helps

public class FeeReportRepository(IDbConnection db)
{
    // Dapper: QueryAsync maps result rows → FeeReportDto automatically
    public async Task<IEnumerable<FeeReportDto>> GetMonthlyFeeReportAsync(
        int schoolId, int month, int year)
    {
        const string sql = @"
            SELECT
                s.Id         AS StudentId,
                s.Name       AS StudentName,
                s.Class,
                fr.Amount,
                fr.PaidDate,
                fr.ReceiptNo,
                DATEDIFF(day, fr.DueDate, fr.PaidDate) AS DaysLate
            FROM Students s
            INNER JOIN FeeRecords fr ON fr.StudentId = s.Id
            WHERE s.SchoolId = @SchoolId
              AND MONTH(fr.PaidDate) = @Month
              AND YEAR(fr.PaidDate)  = @Year
            ORDER BY s.Class, s.Name";

        return await db.QueryAsync<FeeReportDto>(sql, new { SchoolId = schoolId, Month = month, Year = year });
    }

    // Multi-mapping: one query, map to two objects (parent + child)
    public async Task<IEnumerable<SchoolFeeDto>> GetSchoolFeesSummaryAsync(int schoolId)
    {
        const string sql = @"
            SELECT sc.Id, sc.Name AS SchoolName,
                   SUM(fr.Amount) AS TotalCollected, COUNT(fr.Id) AS ReceiptCount
            FROM Schools sc
            LEFT JOIN FeeRecords fr ON fr.SchoolId = sc.Id
            WHERE sc.Id = @SchoolId
            GROUP BY sc.Id, sc.Name";

        return await db.QueryAsync<SchoolFeeDto>(sql, new { SchoolId = schoolId });
    }

    // QueryFirstOrDefaultAsync — single row
    public async Task<StudentFeeStatusDto?> GetStudentFeeStatusAsync(int studentId)
    {
        return await db.QueryFirstOrDefaultAsync<StudentFeeStatusDto>(
            "SELECT s.Name, SUM(fr.Amount) AS TotalPaid, MAX(fr.PaidDate) AS LastPayment FROM Students s LEFT JOIN FeeRecords fr ON fr.StudentId = s.Id WHERE s.Id = @Id GROUP BY s.Name",
            new { Id = studentId });
    }

    // ExecuteAsync — for INSERT/UPDATE/DELETE via Dapper (bulk inserts especially)
    public async Task BulkInsertAttendanceAsync(IEnumerable<AttendanceRecord> records)
    {
        await db.ExecuteAsync(
            "INSERT INTO Attendance (StudentId, Date, Status) VALUES (@StudentId, @Date, @Status)",
            records);  // Dapper sends one batch for all records — much faster than EF SaveChanges per-record!
    }
}

3 Caching — Memory Cache + Redis + Output Cache
🎬 Thuppakki — Vijay uses intel before action. Cache = same. School list doesn't change every second — query once, cache 5 minutes, serve from memory. DB query avoided 1000 times! 
// IMemoryCache — in-process cache (single server)
builder.Services.AddMemoryCache();

public class SchoolService(IMemoryCache cache, ISchoolRepository repo)
{
    public async Task<List<SchoolDto>> GetAllSchoolsAsync()
    {
        const string cacheKey = "all-schools";

        // GetOrCreateAsync — atomic check + create (no race condition)
        return await cache.GetOrCreateAsync(cacheKey, async entry =>
        {
            entry.AbsoluteExpirationRelativeToNow = TimeSpan.FromMinutes(5); // expire in 5 min
            entry.SlidingExpiration                = TimeSpan.FromMinutes(2); // reset if accessed
            entry.Size                             = 1;                         // for SizeLimit
            return await repo.GetAllAsync();
        }) ?? [];
    }

    // Invalidate when school added/updated
    public async Task AddSchoolAsync(School school)
    {
        await repo.AddAsync(school);
        cache.Remove("all-schools"); // bust cache after write!
    }
}

public async Task<StudentDto?> GetStudentAsync(int id)
{
    var key   = $"student:{id}";
    var bytes = await _cache.GetAsync(key);

    if (bytes is not null)
        return JsonSerializer.Deserialize<StudentDto>(bytes);

    var student = await _repo.GetByIdAsync(id);

        new DistributedCacheEntryOptions { AbsoluteExpirationRelativeToNow = TimeSpan.FromMinutes(10) });

    return student;
}

4 Async Best Practices — Don't Block Threads
// DEADLY MISTAKE: .Result or .Wait() in async context → deadlock!
var student = _repo.GetAsync(101).Result;  // ⚠️ DEADLOCK in ASP.NET Core!
_repo.SaveAsync(student).Wait();           // ⚠️ DEADLOCK!
// ASP.NET Core thread pool + SynchronizationContext = .Result deadlock
// Fix: always await, never .Result/.Wait() in non-console code

// ConfigureAwait(false) — library code only (not ASP.NET Core apps)
var data = await _repo.GetAsync(101).ConfigureAwait(false);
// In library code (NuGet packages): ConfigureAwait(false) = don't capture sync context
// In ASP.NET Core app code: not needed (no sync context in ASP.NET Core)

// Parallel async — run INDEPENDENT operations concurrently!
// SLOW: sequential awaits (student + fees run one after another)
var student  = await _studentRepo.GetAsync(id);
var fees     = await _feeRepo.GetAsync(id);
var schedule = await _scheduleRepo.GetAsync(id);
// Total: 100ms + 80ms + 60ms = 240ms

// FAST: Task.WhenAll (all 3 run in parallel)
var (studentTask, feesTask, scheduleTask) = (
    _studentRepo.GetAsync(id),
    _feeRepo.GetAsync(id),
    _scheduleRepo.GetAsync(id));

await Task.WhenAll(studentTask, feesTask, scheduleTask);
var student  = studentTask.Result;   // safe AFTER WhenAll — task is completed!
var fees     = feesTask.Result;
var schedule = scheduleTask.Result;
// Total: max(100, 80, 60) = 100ms. 2.4x faster!

// Parallel.ForEachAsync — controlled parallelism for bulk operations
var schools = await _ctx.Schools.ToListAsync();

await Parallel.ForEachAsync(schools,
    new ParallelOptions { MaxDegreeOfParallelism = 5 },  // max 5 concurrent DB calls
    async (school, ct) =>
    {
        await _reportService.GenerateMonthlyReportAsync(school.Id, ct);
    });
// 200 schools processed 5 at a time. Total: ~200/5 × 2s = 80s instead of 200 × 2s = 400s!

// ValueTask — avoid heap allocation for hot paths (cache hits)
public ValueTask<Student?> GetStudentAsync(int id)
{
    // Cache hit: synchronous return — no heap allocation!
    if (_localCache.TryGetValue(id, out var cached))
        return ValueTask.FromResult(cached);

    // Cache miss: async DB call
    return new ValueTask<Student?>(_repo.GetAsync(id));
}
// Task always allocates on heap. ValueTask = zero alloc on synchronous path!

// Cursor-based pagination (fast) — for large tables
public async Task<PagedResult<StudentDto>> GetStudentsPagedAsync(
    int schoolId, int lastId = 0, int pageSize = 50)
{
    var items = await _ctx.Students
        .Where(s => s.SchoolId == schoolId && s.Id > lastId)  // cursor: last seen Id
        .OrderBy(s => s.Id)
        .Take(pageSize + 1)  // take one extra to check if more pages exist
        .Select(s => new StudentDto { Id = s.Id, Name = s.Name, Class = s.Class })
        .AsNoTracking()
        .ToListAsync();

    var hasMore  = items.Count > pageSize;
    if (hasMore) items.RemoveAt(items.Count - 1);

    return new PagedResult<StudentDto>
    {
        Items      = items,
        HasMore    = hasMore,
        NextCursor = hasMore ? items.Last().Id : (int?)null
    };
    // No COUNT(*) = no full table scan! Cursor-based = consistent O(log n) regardless of page number
}

// Offset pagination (simpler, but slower for deep pages)
.Skip((pageNumber - 1) * pageSize).Take(pageSize)
// Page 1000 of 50 records = skip 49,950 rows → SQL Server scans 50,000 rows → SLOW!
// Compiled Queries — pre-compile EF LINQ → 40% faster for hot paths
public class StudentQueries
{
    // Compiled once at startup, reused for every call
    private static readonly Func<ZenCampusDbContext, int, IAsyncEnumerable<StudentDto>>
        GetBySchoolCompiled = EF.CompileAsyncQuery(
            (ZenCampusDbContext ctx, int schoolId) =>
                ctx.Students
                   .Where(s => s.SchoolId == schoolId && s.Status == "Active")
                   .Select(s => new StudentDto { Id = s.Id, Name = s.Name })
                   .OrderBy(s => s.Name));

    public IAsyncEnumerable<StudentDto> GetActiveStudentsAsync(ZenCampusDbContext ctx, int schoolId)
        => GetBySchoolCompiled(ctx, schoolId);
    // Called thousands of times/sec for attendance dashboard — 40% less CPU overhead!
}

When to use Dapper vs EF Core? Medium Real Interview
📋 "You use both EF Core and Dapper in ZenCampus. How do you decide which one for a given query?"
✅ CLEAR DECISION RULES

EF Core: Student/Fee/Attendance CRUD — create, update, delete, simple filters. Where change tracking helps: load student, modify, SaveChanges(). Migrations for schema management. Simple WHERE + SELECT for single entity.

Dapper: Fee collection monthly report (5 table JOIN + GROUP BY + DATEDIFF calculations). Complex aggregates across multiple tables. Stored procedure calls. Bulk insert (10,000 attendance records — EF one-by-one SaveChanges is 30x slower). School-level analytics dashboards. Any query where EF generates ugly/slow SQL that you'd rather write yourself.

Hybrid pattern in ZenCampus: Use EF Core for all writes (Student, Fee, Attendance repositories). Use Dapper for all report queries (FeeReportRepository, AttendanceReportRepository). Same connection string, different libraries. EF handles schema (migrations), Dapper handles read performance.

public async Task UpdateSchoolAsync(School school)
{

    // Invalidate ALL related cache keys
    _cache.Remove("all-schools");                    // list cache

// Pattern: cache tags group related keys

// EvictByTag("school-42") = invalidate ALL at once after any school 42 update

"Cache invalidation and naming things. In ZenCampus: every write method explicitly removes affected cache keys. Pattern: UpdateAsync() → RemoveAsync(key). For multi-level cache (MemoryCache + Redis + OutputCache), invalidate all three. Missing one = stale data bug."

ValueTask vs Task — when does it matter? Medium
📋 "Should you always return ValueTask instead of Task for better performance?"
✅ VALUETASK — ONLY FOR HOT SYNC-SOMETIMES PATHS
// Task = always allocates on heap (even when result is immediate)
// ValueTask = zero alloc when result is synchronous (cache hit)

// Use ValueTask ONLY when:
// 1. Method often completes synchronously (cache hit 80% of time)
// 2. Called thousands of times per second (hot path)
// 3. Avoiding allocation matters for GC pressure

public ValueTask<Student?> GetStudentAsync(int id)
{
    // Synchronous path (cache hit): zero heap allocation
    if (_localCache.TryGetValue(id, out var cached))
        return ValueTask.FromResult(cached);   // ValueTask<T> is a struct!

    // Async path (cache miss): one Task allocation (same as before)
    return new ValueTask<Student?>(_db.GetByIdAsync(id));
}

// Don't use ValueTask when:
// 1. Method always awaits (always async) → Task is fine
// 2. Caller needs to await multiple times → ValueTask can't be awaited twice!
// 3. Result needs to be stored and awaited later → use Task

// DANGER: never await ValueTask twice!
var vt = GetStudentAsync(101);
var s1 = await vt;  // OK
var s2 = await vt;  // ❌ UNDEFINED BEHAVIOR — await twice!

7 ⚡ Quick Quiz

Q1: AsNoTracking() use பண்ணும்போது என்ன happen ஆகும்?

A) Query doesn't hit the database — returns from memory cache
B) EF Core doesn't create ChangeTracker entries — 30-40% faster for reads, but SaveChanges() won't detect changes to those entities
C) Query runs asynchronously in the background
D) Disables lazy loading permanently
✅ AsNoTracking() skips the EF Core Change Tracker. Every tracked entity needs a ChangeTracker entry (memory + CPU overhead). For read-only endpoints, this is wasted overhead. 30-40% faster and less memory. NEVER use for entities you'll update — changes won't be detected by SaveChanges()!
❌ AsNoTracking() tells EF Core: "don't monitor this entity for changes." Still hits DB. Still materializes objects. Just skips ChangeTracker overhead (the dictionary that tracks Added/Modified/Deleted state). Use for all read-only queries. Avoid for anything you'll SaveChanges() on!

Q2: N+1 problem — which code causes it?

A) Using Include() with ThenInclude() on nested navigation properties
B) Loading a list, then accessing a navigation property inside a foreach loop (without Include)
C) Using AsNoTracking() with pagination
D) Calling ToListAsync() inside a service layer
✅ N+1: 1 query to get the list + N queries (one per item) to load related data. Caused by accessing navigation properties (school.Students) in a foreach when they weren't Include()'d. EF lazy loads each access = N separate SQL queries. Fix: Include() before ToListAsync(), or Select() projections that compute everything server-side.
❌ N+1 = "1 query for the list + N queries for related data." Happens when you load entities without Include(), then access a navigation property in a loop. EF Core lazy loads each .Students access → one SQL per iteration. 200 schools = 201 SQL queries! Fix: .Include(s => s.Students) or .Select(projecting what you need).

Q3: Task.WhenAll vs sequential await — key difference?

A) WhenAll uses multiple threads; sequential uses one thread
B) WhenAll starts all tasks concurrently and waits for all to finish; sequential awaits each one fully before starting the next
C) WhenAll only works with CPU-bound operations
D) No functional difference — WhenAll is just syntactic sugar
✅ Sequential: await A (100ms) → then start B (80ms) → then start C (60ms) = 240ms total. WhenAll: START all three simultaneously → wait for slowest = 100ms. For independent DB calls (student + fees + timetable), WhenAll is 2.4x faster! Important: they must be INDEPENDENT (no B depends on A's result).
❌ Task.WhenAll starts all tasks IMMEDIATELY (no waiting between starts), then waits for ALL to complete. Sequential await starts each task only after the previous finishes. For 3 independent 100ms DB calls: sequential = 300ms, WhenAll = 100ms. Use WhenAll for independent operations. Don't use if B needs A's result!

Q4: Cache invalidation — fee paid, which caches must be busted?

A) No cache needs busting — wait for TTL to expire naturally
B) Only the student's individual cache key
C) All related keys: student fee status cache, school fee summary cache, and any output-cached API responses that show fee data
D) Clear all caches on every write operation
✅ After fee payment: student's fee status changes, school's total collection changes, any dashboard showing "fees due" changes. Must bust: student:{id} key, school-fee-summary:{schoolId}, any OutputCache tagged with "fees". Missing one = stale data. Clearing ALL caches = too aggressive, hurts performance. Target only affected keys.
❌ After any write, bust ALL caches that display that data. Fee paid → student fee status cache (student:101:fee-status), school fee summary cache (school:42:fee-summary), and HTTP output-cache responses showing fee data. Don't just wait for TTL — admin will see wrong totals until expiry. Write-through: always remove on write!
