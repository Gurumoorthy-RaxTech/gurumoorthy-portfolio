// Token generation service
public class JwtTokenService(IConfiguration config)
{
    public LoginResponse GenerateTokens(User user)
    {
        var claims = new[]
        {
            new Claim(ClaimTypes.NameIdentifier, user.Id.ToString()),
            new Claim(ClaimTypes.Name,             user.Name),
            new Claim(ClaimTypes.Role,             user.Role),     // Admin/Teacher/Parent
            new Claim("schoolId",                  user.SchoolId.ToString()),
            new Claim(JwtRegisteredClaimNames.Jti,  Guid.NewGuid().ToString()), // unique token ID
        };

        var key   = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(config["Jwt:Secret"]!));
        var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
        var token = new JwtSecurityToken(
            issuer:             "https://zencampus.com",
            audience:           "zencampus-api",
            claims:             claims,
            expires:            DateTime.UtcNow.AddHours(1),  // short-lived!
            signingCredentials: creds);

        return new LoginResponse
        {
            AccessToken  = new JwtSecurityTokenHandler().WriteToken(token),
            RefreshToken = GenerateRefreshToken(),  // opaque, stored in DB
            ExpiresAt    = DateTime.UtcNow.AddHours(1)
        };
    }

    private static string GenerateRefreshToken()
    {
        var bytes = RandomNumberGenerator.GetBytes(64);
        return Convert.ToBase64String(bytes);  // 64 crypto-random bytes → opaque token
    }
}
// Refresh token flow — when access token expires (1hr), use refresh token (7 days)
[HttpPost("auth/refresh")]
public async Task<IActionResult> RefreshAsync([FromBody] RefreshRequest req)
{
    var stored = await _tokenRepo.GetRefreshTokenAsync(req.RefreshToken);

    if (stored is null || stored.ExpiresAt < DateTime.UtcNow || stored.IsRevoked)
        return Unauthorized("Invalid or expired refresh token");

    // Rotate refresh token — old one invalid, new one issued
    await _tokenRepo.RevokeAsync(req.RefreshToken);
    var user     = await _userRepo.GetAsync(stored.UserId);
    var response = _jwtService.GenerateTokens(user!);

    await _tokenRepo.SaveRefreshTokenAsync(user!.Id, response.RefreshToken,
        DateTime.UtcNow.AddDays(7));

    return Ok(response);
}
// Refresh token rotation: old refresh token revoked on each use.
// If attacker steals refresh token and uses it → original user's next refresh fails → detects theft!

// Policy-based authorization (complex rules)
builder.Services.AddAuthorization(opts =>
{
    opts.AddPolicy("SameSchool", p => p
        .RequireAuthenticatedUser()
        .RequireClaim("schoolId")
        .AddRequirements(new SameSchoolRequirement()));

    opts.AddPolicy("CanCollectFee", p => p
        .RequireRole("Admin", "Accountant")
        .RequireClaim("schoolId"));   // must have schoolId AND be Admin or Accountant

    opts.AddPolicy("ParentOnly", p => p
        .RequireRole("Parent")
        .RequireClaim("studentIds"));  // parent must have linked studentIds
});

// Custom requirement + handler
public class SameSchoolRequirement : IAuthorizationRequirement { }

public class SameSchoolHandler : AuthorizationHandler<SameSchoolRequirement, Student>
{
    protected override Task HandleRequirementAsync(
        AuthorizationHandlerContext ctx,
        SameSchoolRequirement req,
        Student resource)  // resource = the student being accessed
    {
        var userSchoolId = int.Parse(ctx.User.FindFirst("schoolId")!.Value);

        if (resource.SchoolId == userSchoolId)
            ctx.Succeed(req);           // teacher can access this student
        // else: implicit Fail — teacher from school 42 can't touch school 99's students!

        return Task.CompletedTask;
    }
}

// Resource-based authorization — check in controller action
[HttpGet("students/{id}")]
[Authorize(Roles = "Admin,Teacher,Parent")]
public async Task<IActionResult> GetStudent(int id)
{
    var student = await _repo.GetAsync(id);
    if (student is null) return NotFound();

    // Resource-based: can THIS user access THIS specific student?
    var authResult = await _authService.AuthorizeAsync(User, student, "SameSchool");
    if (!authResult.Succeeded) return Forbid(); // 403 — authenticated but not authorized

    // For Parent role: extra check — is this their child?
    if (User.IsInRole("Parent"))
    {
        var parentId  = int.Parse(User.FindFirst(ClaimTypes.NameIdentifier)!.Value);
        var isMyChild = await _repo.IsParentOfStudentAsync(parentId, id);
        if (!isMyChild) return Forbid(); // Parent A cannot see Parent B's child!
    }

    return Ok(student);
}

6 Rate Limiting — Brute Force + DDoS Prevention
🎬 Theri — Vijay controls the chaotic crowd methodically. Rate limiting = same. Login endpoint without limit: attacker tries 10,000 passwords in 10 seconds. With limit: 5 attempts per minute, locked for 15 min = brute force blocked! 
// ASP.NET Core 8 — built-in rate limiting (no 3rd party needed!)
// NuGet: already in framework

builder.Services.AddRateLimiter(opts =>
{
    // Fixed window: 5 requests per minute per IP on login
    opts.AddFixedWindowLimiter("login", o =>
    {
        o.Window           = TimeSpan.FromMinutes(1);
        o.PermitLimit      = 5;              // 5 attempts per minute
        o.QueueLimit       = 0;              // no queuing — reject immediately
        o.QueueProcessingOrder = QueueProcessingOrder.OldestFirst;
    });

    // Sliding window: 100 API calls per minute per user
    opts.AddSlidingWindowLimiter("api", o =>
    {
        o.Window            = TimeSpan.FromMinutes(1);
        o.SegmentsPerWindow = 6;              // check every 10 seconds
        o.PermitLimit       = 100;
    });

    // Per-user partitioned limiter (different limits per school)
    opts.AddPolicy("per-school", httpCtx =>
        RateLimitPartition.GetFixedWindowLimiter(
            partitionKey: httpCtx.User.FindFirst("schoolId")?.Value ?? "anonymous",
            factory: _ => new FixedWindowRateLimiterOptions
            {
                Window      = TimeSpan.FromMinutes(1),
                PermitLimit = 200  // 200 requests per school per minute
            }));

    // 429 response when limit exceeded
    opts.OnRejected = async (ctx, ct) =>
    {
        ctx.HttpContext.Response.StatusCode = 429;
        ctx.HttpContext.Response.Headers.Append("Retry-After", "60");
        await ctx.HttpContext.Response.WriteAsync("Too many requests. Try after 60 seconds.", ct);
    };
});

app.UseRateLimiter(); // must be before UseAuthentication!

Password hashing — developer used MD5! Medium
📋 Scenario: "Legacy ZenCampus stores passwords as MD5 hashes. Security audit found this. What's wrong and how to migrate to BCrypt?"
✅ BCRYPT MIGRATION — SAFE TRANSITION
// Why MD5 is BAD for passwords:
// 1. Fast: GPU can crack 10 BILLION MD5/second
// 2. Rainbow tables: precomputed hash → password in seconds
// 3. No salt: same password = same hash (identifies duplicates)
// MD5("password123") = 482c811da5d5b4bc6d497ffa98491e38 → cracked in 0.01ms!

// BCrypt advantages:
// 1. Adaptive cost factor (work factor) — intentionally slow
// 2. Built-in salt — same password → different hash every time
// 3. Cost 12 = ~250ms to hash — 250ms × 10B guesses = 793 years!
// NuGet: BCrypt.Net-Next

var hash = BCrypt.Net.BCrypt.HashPassword(password, workFactor: 12);
// $2a$12$N9qo8uLOickgx2ZMRZoMyeIjZAgcfl7p92ldGxad68LJZdL17lhWy
//  ↑   ↑                                               ↑ 22-char salt embedded!

var isValid = BCrypt.Net.BCrypt.Verify(plaintext, hash);

// Safe migration — NO forced password reset needed!
// 1. Add column: PasswordHashBcrypt (nullable)
// 2. On login: if BCrypt column null, verify MD5, re-hash with BCrypt, store BCrypt
// 3. After 30 days: force-reset remaining MD5 users (never logged in)
// 4. Remove MD5 column
public async Task<LoginResponse?> LoginAsync(string email, string password)
{
    var user = await _repo.GetByEmailAsync(email);
    if (user is null) return null;

    bool valid;
    if (user.PasswordHashBcrypt is not null)
    {
        valid = BCrypt.Net.BCrypt.Verify(password, user.PasswordHashBcrypt);
    }
    else
    {
        var md5Hash = ComputeMd5(password);
        valid = md5Hash == user.PasswordHashMd5;
        if (valid)  // upgrade on login!
        {
            user.PasswordHashBcrypt = BCrypt.Net.BCrypt.HashPassword(password, 12);
            await _repo.UpdateAsync(user);
        }
    }
    return valid ? _jwtService.GenerateTokens(user) : null;
}

Refresh token theft — detect and revoke Hard
📋 Scenario: "Attacker somehow gets parent's refresh token (phishing, XSS, device theft). How does refresh token rotation detect this and protect the account?"
✅ REFRESH TOKEN ROTATION — THEFT DETECTION
// Normal flow:
// User logs in → RefreshToken_A issued, stored in DB
// After 1hr: user sends RefreshToken_A → gets new AccessToken + RefreshToken_B
// RefreshToken_A is REVOKED in DB

// Attack detected:
// Attacker steals RefreshToken_A (before user uses it)
// User uses RefreshToken_A → gets RefreshToken_B (normal)
// Attacker later uses RefreshToken_A → DB shows it's REVOKED → 401!
// Attacker detected — didn't get a valid token!

// But what if attacker uses it FIRST?
// Attacker uses RefreshToken_A → gets RefreshToken_B (attacker has it now)
// User uses RefreshToken_A → REVOKED → 401 → user calls support
// Support sees: same token used twice from different IPs → compromise confirmed!

// BEST PRACTICE: when revoked token reuse detected → revoke ALL user's tokens
public async Task<IActionResult> RefreshAsync(RefreshRequest req)
{
    var token = await _repo.GetRefreshTokenAsync(req.RefreshToken);

    if (token?.IsRevoked == true)
    {
        // Reuse detected! Revoke ALL tokens for this user immediately
        await _repo.RevokeAllUserTokensAsync(token.UserId);
        _log.LogWarning("Refresh token reuse detected for user {UserId}", token.UserId);
        // Optionally: send security alert email to user
        return Unauthorized("Security violation detected. Please log in again.");
    }
    // ... normal rotation flow
}

📋 Scenario: "Describe the security measures you've implemented in ZenCampus. How do you approach application security?"
✅ COMPLETE SECURITY POSTURE ANSWER

Authentication: JWT with 1-hour access tokens, 7-day refresh tokens with rotation. Tokens generated with HMACSHA256 and 64-byte secret from Azure Key Vault. Refresh token reuse detection — revokes all user tokens on suspicious reuse. BCrypt (work factor 12) for password hashing after migrating from legacy MD5.

Authorization: Three-layer model: Role-based (Admin/Teacher/Parent/Student), Policy-based (CanCollectFee = Admin+Accountant+schoolId claim), Resource-based (teacher can only access their school's students via IAuthorizationService). Never trust client-provided schoolId — always from JWT claim. Parent checked against IsParentOfStudent() for every student data endpoint — prevents IDOR/OWASP A01.

Injection prevention: EF Core for all queries (parameterized by default). Dapper uses @param syntax for raw SQL. FluentValidation on all input DTOs — name must match [a-zA-Z\s\.\-']+ regex. Never concatenate user input into SQL strings.

XSS: Razor auto-encodes all output. CSP header blocks inline scripts. JWT stored in httpOnly cookie (not localStorage) — XSS can't steal it. Input validation rejects HTML characters in names.

Rate limiting: ASP.NET Core 8 built-in. Login endpoint: 5 per minute per IP. API endpoints: 100 per minute per user. Account lockout after 5 failed logins (15 min). Azure Application Insights alert for login spike.

Secrets: Azure Key Vault + Managed Identity. Zero credentials in code or config files. Swagger disabled in production. Security headers on all responses.

8 ⚡ Quick Quiz

Q1: JWT payload-la sensitive data (password, address) போடலாமா?

A) Yes — JWT is encrypted, so it's safe
B) No — JWT payload is only Base64-encoded (not encrypted). Anyone with the token can decode it. Only put identity and authorization claims.
C) Yes — only the signature part is readable
D) Safe only if using RS256 algorithm
✅ JWT is signed (HMAC/RSA), NOT encrypted. Payload = Base64URL encoded. Decode at jwt.io → see everything. Never put passwords, SSN, salary, address in JWT. Only: userId, role, schoolId (for routing/auth). Sensitive data → fetch fresh from DB per request.
❌ JWT payload is BASE64 ENCODED (not encrypted). Anyone can decode: atob(payload.replace(/-/g,'+').replace(/_/g,'/')) → JSON object. The signature proves authenticity but doesn't hide content. Never put sensitive data in JWT — only identity claims (userId, role, schoolId).

Q2: OWASP A01 Broken Access Control — ZenCampus example?

A) Admin uses weak password "admin123"
B) SQL injection in student search
C) Parent calls GET /api/students/102 (not their child) and successfully sees the student's data because only role check (Parent) exists, not ownership check
D) JWT token not expiring
✅ IDOR (Insecure Direct Object Reference) = OWASP A01. Parent is authenticated (role=Parent) but accesses resource they don't own (student 102 belongs to different parent). Role check alone is insufficient. Must check: IsParentOfStudent(parentId, studentId). Missing ownership check = broken access control.
❌ OWASP A01 Broken Access Control = authorization check is missing or insufficient. Classic example: [Authorize(Roles="Parent")] only verifies role. Parent 50 can iterate student IDs 1-50000 and see all of them. Fix: check ownership in every resource endpoint — IsParentOfStudent() for parents, SameSchool for teachers.

Q3: MD5 vs BCrypt password hashing — key difference?

A) MD5 is longer, so more secure
B) BCrypt is intentionally slow (250ms at work factor 12) and includes built-in salt. MD5 is fast (cracked in milliseconds by GPU) and has no built-in salt.
C) Both are equally secure for passwords
D) MD5 is better for short passwords, BCrypt for long ones
✅ MD5 design goal: FAST (for checksums). BCrypt design goal: SLOW (for passwords). GPU: 10 billion MD5/second vs 0.004 BCrypt/second (at cost 12). Also: BCrypt embeds random salt in output → same password = different hash every time → no rainbow tables. BCrypt is the minimum for password storage (Argon2id is even better).
❌ MD5 is cryptographically broken AND fast — attacker can try 10 billion passwords per second. BCrypt is intentionally slow (adaptive cost factor). Cost 12 = ~250ms per hash. 10B guesses = 793 years! BCrypt also has built-in salt (in the output string) — same password always hashes differently, defeating rainbow tables.

Q4: Why is storing JWT in localStorage dangerous vs httpOnly cookie?

A) localStorage is slower than cookies
B) localStorage is accessible by JavaScript — any XSS attack can read and steal the JWT. httpOnly cookies cannot be read by JavaScript — XSS cannot steal them.
C) localStorage tokens expire automatically, cookies don't
D) Cookies are encrypted by the browser, localStorage is not
✅ localStorage: JavaScript can read with localStorage.getItem('jwt'). XSS attack: inject script → steal JWT → send to attacker server → attacker logs in as you. httpOnly cookie: Set-Cookie: token=...; HttpOnly — browser never lets JS read it. Even successful XSS cannot steal the JWT. Defense in depth: httpOnly + CSP + input validation.
❌ localStorage is accessible via JavaScript (any JS on the page). XSS injects malicious JS → reads localStorage.getItem('jwt') → sends to evil server → attacker has your JWT! httpOnly cookies: the browser's Set-Cookie API with HttpOnly flag means NO JavaScript can read that cookie. XSS can't steal what it can't read!
