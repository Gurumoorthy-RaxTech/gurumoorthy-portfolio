// NuGet: xunit, xunit.runner.visualstudio, Moq, FluentAssertions

public class CreateStudentCommandHandlerTests
{
    private readonly Mock<IStudentRepository> _mockRepo;
    private readonly Mock<IUnitOfWork>        _mockUow;
    private readonly CreateStudentCommandHandler _sut; // System Under Test

    public CreateStudentCommandHandlerTests()
    {
        _mockRepo = new Mock<IStudentRepository>();
        _mockUow  = new Mock<IUnitOfWork>();
        _sut = new CreateStudentCommandHandler(
            _mockRepo.Object, _mockUow.Object,
            NullLogger<CreateStudentCommandHandler>.Instance);
    }

    [Fact]
    public async Task Handle_NewStudent_ReturnsSuccessWithId()
    {
        // Arrange
        var cmd = new CreateStudentCommand("Basha Kumar", "ADM2026001",
            DateOnly.Parse("2010-06-15"), ClassId: 5, SchoolId: 1, ParentPhone: "9876543210");
        _mockRepo.Setup(r => r.ExistsAsync(1, "ADM2026001", default)).ReturnsAsync(false);
        _mockRepo.Setup(r => r.AddAsync(It.IsAny<Student>(), default))
                 .Callback<Student, CancellationToken>((s, _) => s.SetId(42)) // simulate DB id
                 .Returns(Task.CompletedTask);

        // Act
        var result = await _sut.Handle(cmd, default);

        // Assert (FluentAssertions)
        result.IsSuccess.Should().BeTrue();
        result.Value.Should().Be(42);
        _mockRepo.Verify(r => r.AddAsync(It.Is<Student>(s => s.Name == "Basha Kumar"), default), Times.Once);
        _mockUow.Verify(u => u.CommitAsync(default), Times.Once);
    }

    [Fact]
    public async Task Handle_DuplicateAdmissionNumber_ReturnsFailure()
    {
        // Arrange
        _mockRepo.Setup(r => r.ExistsAsync(1, "ADM001", default)).ReturnsAsync(true);
        var cmd = new CreateStudentCommand("Test", "ADM001", DateOnly.Today, 1, 1, "9999999999");

        // Act
        var result = await _sut.Handle(cmd, default);

        // Assert
        result.IsSuccess.Should().BeFalse();
        result.Error.Should().Contain("Admission number");
        _mockRepo.Verify(r => r.AddAsync(It.IsAny<Student>(), default), Times.Never); // never called
        _mockUow.Verify(u => u.CommitAsync(default), Times.Never);
    }

    [Theory]
    [InlineData("")]
    [InlineData("   ")]
    [InlineData(null)]
    public async Task Handle_EmptyName_ThrowsDomainException(string? name)
    {
        var cmd = new CreateStudentCommand(name!, "ADM001", DateOnly.Today, 1, 1, "9999");
        var act = () => _sut.Handle(cmd, default);
        await act.Should().ThrowAsync<DomainException>();
    }
}

🎭 Moq — Mock, Setup, Verify

// Mock setup patterns
var mockRepo = new Mock<IStudentRepository>();

// Setup return value
mockRepo.Setup(r => r.GetByIdAsync(1, default)).ReturnsAsync(new Student { Id = 1, Name = "Basha" });

// Setup with any parameter
mockRepo.Setup(r => r.GetByIdAsync(It.IsAny<int>(), default)).ReturnsAsync((Student?)null);

// Setup with condition
mockRepo.Setup(r => r.GetByIdAsync(It.Is<int>(id => id > 0), default))
        .ReturnsAsync(new Student());

// Setup to throw
mockRepo.Setup(r => r.GetByIdAsync(-1, default))
        .ThrowsAsync(new NotFoundException("Student not found"));

// Callback — capture what was passed
Student? capturedStudent = null;
mockRepo.Setup(r => r.AddAsync(It.IsAny<Student>(), default))
        .Callback<Student, CancellationToken>((s, _) => capturedStudent = s)
        .Returns(Task.CompletedTask);

// Verify — assert methods were called
mockRepo.Verify(r => r.AddAsync(It.IsAny<Student>(), default), Times.Once);
mockRepo.Verify(r => r.AddAsync(It.IsAny<Student>(), default), Times.Never);
mockRepo.Verify(r => r.AddAsync(It.IsAny<Student>(), default), Times.Exactly(2));
mockRepo.VerifyNoOtherCalls();  // no unexpected calls happened

🌐 Integration Testing with WebApplicationFactory

// Integration test — real HTTP stack, in-memory or real DB
public class StudentsApiTests(WebApplicationFactory<Program> factory)
    : IClassFixture<WebApplicationFactory<Program>>
{
    private readonly HttpClient _client = factory.WithWebHostBuilder(builder =>
    {
        builder.ConfigureTestServices(services =>
        {
            // Replace real DB with in-memory
            services.RemoveAll<DbContextOptions<ZenCampusDbContext>>();
            services.AddDbContext<ZenCampusDbContext>(opts =>
                opts.UseInMemoryDatabase("TestDb"));

            // Replace SMS client with fake
            services.RemoveAll<ISmsService>();
            services.AddSingleton<ISmsService, FakeSmsService>();
        });
    }).CreateClient();

    [Fact]
    public async Task POST_Students_ValidRequest_Returns201()
    {
        // Arrange
        var cmd = new { Name = "Test Student", AdmissionNumber = "ADM999", SchoolId = 1 };
        var content = new StringContent(JsonSerializer.Serialize(cmd),
            Encoding.UTF8, "application/json");
        _client.DefaultRequestHeaders.Authorization =
            new AuthenticationHeaderValue("Bearer", GetTestJwt());

        // Act
        var response = await _client.PostAsync("/api/students", content);

        // Assert
        response.StatusCode.Should().Be(HttpStatusCode.Created);
        var body = await response.Content.ReadFromJsonAsync<CreatedResponse>();
        body!.Id.Should().BeGreaterThan(0);
    }

    [Fact]
    public async Task POST_Students_MissingName_Returns400()
    {
        var response = await _client.PostAsJsonAsync("/api/students",
            new { AdmissionNumber = "ADM001", SchoolId = 1 }); // Name missing
        response.StatusCode.Should().Be(HttpStatusCode.BadRequest);
        var pd = await response.Content.ReadFromJsonAsync<ProblemDetails>();
        pd!.Title.Should().Contain("Validation");
    }
}
Golden Rule: Unit tests = fast, isolated, test business logic. Integration tests = test HTTP pipeline, middleware, auth, validation, serialization. Both needed — unit tests find logic bugs, integration tests find wiring bugs.

🎯 10 Interview Scenarios

1
What is the difference between a unit test and an integration test?Concept

Unit test: tests a single class/method in isolation. All dependencies mocked. No DB, no HTTP. Fast (<100ms). Finds logic bugs. Integration test: tests multiple components together — actual HTTP stack, real or test DB, middleware, auth. Slower. Finds wiring bugs. Both needed: unit tests prove business rules work, integration tests prove the system assembles correctly.

2
What does Mock.Verify() check vs Mock.Setup()?Concept

Setup() configures what the mock returns when a method is called. Verify() asserts that a method was called (or not called) with specific parameters. Setup = arrange phase. Verify = assert phase. Example: Setup AddAsync to do nothing, then Verify it was called Once with a student matching certain criteria.

3
How do you test a method that throws an exception?Code
// FluentAssertions
var act = () => _sut.Handle(invalidCmd, default);
await act.Should().ThrowAsync<DomainException>()
         .WithMessage("*name*");

// xUnit Assert.ThrowsAsync
var ex = await Assert.ThrowsAsync<DomainException>(() => _sut.Handle(invalidCmd, default));
Assert.Contains("name", ex.Message);
4
What is [Theory] + [InlineData] in xUnit?xUnit

Theory runs the same test with different input values. [InlineData] provides each set of inputs. In ZenCampus: test that empty name, whitespace name, and null name all throw DomainException — three test cases, one method: [Theory] [InlineData("")] [InlineData(" ")] [InlineData(null)]. Reduces code duplication in tests.

5
How do you replace a service in WebApplicationFactory for integration tests?Code

Use WithWebHostBuilder + ConfigureTestServices: remove the real registration, add the test replacement. Common replacements: InMemory database instead of SQL Server, FakeSmsService instead of Msg91Client, fake Azure Blob storage. Tests run without real external dependencies.

6
What is test isolation and why does it matter?Concept

Each test must not depend on or affect other tests. If Test A adds data to DB and Test B expects clean DB — tests fail depending on order. Isolation means each test creates its own state, cleans up, and doesn't share state. For integration tests: use a fresh DB per test class (IClassFixture), reset between tests, or use transactions rolled back after each test.

7
Difference between NullLogger.Instance and Mock<ILogger>()?Practical

NullLogger.Instance: does nothing, no overhead. Use when logging is NOT under test — you just need something to pass to the constructor. Mock<ILogger>(): allows verifying specific log calls were made. Use when testing that "LogError was called when exception occurs". Most tests: NullLogger. Tests verifying error logging: Mock<ILogger>.

8
What code coverage percentage should ZenCampus target?Practical

80% is a common target. But coverage % is misleading — 100% coverage doesn't mean bug-free. Better: cover all business logic (domain entities, command handlers, validators), all happy paths, all error paths. Don't chase coverage on trivial getters/setters. Focus on: critical fee calculation logic (100%), admission flow (100%), authentication (100%). Generated code, migrations — exclude from coverage.

9
How would you test that a FluentValidation rule works for the student admission form?Code
public class CreateStudentCommandValidatorTests
{
    private readonly CreateStudentCommandValidator _validator = new();

    [Theory]
    [InlineData("", "ADM001", false)]  // name empty = invalid
    [InlineData("Basha", "", false)]   // admission no empty = invalid
    [InlineData("Basha", "ADM001", true)] // both valid = valid
    public void Validate_ReturnsExpected(string name, string admNo, bool expectedValid)
    {
        var cmd = new CreateStudentCommand(name, admNo, DateOnly.Today, 1, 1, "9999");
        var result = _validator.Validate(cmd);
        result.IsValid.Should().Be(expectedValid);
    }
}
10
What is the difference between [Fact] and [Theory] in xUnit?xUnit

[Fact]: test that is always true regardless of input. No parameters. Runs once. [Theory]: test that is true for a specific set of data. Parameterized. Runs once per [InlineData] set. Rule: if you find yourself writing nearly identical tests with different values — use [Theory]. Single test case = [Fact].
