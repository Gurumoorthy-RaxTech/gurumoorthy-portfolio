// INSERT — returns new Id
public async Task<int> AddAsync(Student s)
{
    const string sql = @"
        INSERT INTO Students
            (Name, Class, RollNo, Phone,
             SchoolId, CreatedOn)
        VALUES
            (@Name, @Class, @RollNo, @Phone,
             @SchoolId, @CreatedOn);
        SELECT CAST(SCOPE_IDENTITY() AS INT)";

    return await _db
        .QuerySingleAsync<int>(sql, new
        {
            s.Name,   s.Class,
            s.RollNo, s.Phone,
            s.SchoolId,
            CreatedOn = DateTime.UtcNow
        });
}

// UPDATE — returns rows affected
public async Task<int> UpdateAsync(Student s)
{
    const string sql = @"
        UPDATE Students
        SET    Name     = @Name,
               Class    = @Class,
               Phone    = @Phone,
               UpdatedOn= @UpdatedOn
        WHERE  Id       = @Id";

    return await _db.ExecuteAsync(sql, new
    {
        s.Name, s.Class, s.Phone, s.Id,
        UpdatedOn = DateTime.UtcNow
    });
}

public async Task DeleteAsync(int id)

        "UPDATE Students SET IsActive=0 WHERE Id=@Id",
        new { Id = id });
❓ Interview Q: Hard delete vs Soft delete?
Hard delete = DELETE row permanently — no recovery. Soft delete = IsActive=0, data stays. ZenCampus-la student TC (Transfer Certificate) ku soft delete — history venum! Audit trail, reporting-ku soft delete always better.
Syntax — Advanced Dapper
Multi-Map · QueryMultiple · Bulk Insert
// Multi-Map — JOIN result → nested objects
public async Task<IEnumerable<Student>>
    GetWithFeesAsync(int schoolId)
{
    const string sql = @"
        SELECT s.*, f.Id, f.Amount,
               f.PaidDate, f.IsPaid
        FROM   Students s
        LEFT JOIN Fees  f
            ON f.StudentId = s.Id
        WHERE  s.SchoolId = @SchoolId";

    var students =
        await _db.QueryAsync<Student, Fee, Student>(
            sql,
            (student, fee) =>       // map lambda
            {
                student.Fees ??= new();
                if (fee != null)
                    student.Fees.Add(fee);
                return student;
            },
            new { SchoolId = schoolId },
            splitOn: "Id");        // split col

    return students
        .GroupBy(s => s.Id)
        .Select(g =>
        {
            var s = g.First();
            s.Fees = g
                .SelectMany(x => x.Fees ?? new())
                .ToList();
            return s;
        });
}

// QueryMultiple — multiple result sets
public async Task<DashboardData>
    GetDashboardAsync(int schoolId)
{
    const string sql = @"
        SELECT COUNT(*) FROM Students
            WHERE SchoolId=@Id;
        SELECT COUNT(*) FROM Staff
            WHERE SchoolId=@Id;
        SELECT SUM(Amount) FROM Fees
            WHERE SchoolId=@Id AND IsPaid=1";

    using var multi = await _db
        .QueryMultipleAsync(sql,
            new { Id = schoolId });

    return new DashboardData
    {
        StudentCount = await multi
                             .ReadSingleAsync<int>(),
        StaffCount   = await multi
                             .ReadSingleAsync<int>(),
        TotalFees    = await multi
                             .ReadSingleAsync<decimal>()
    };
}

// Transaction in Dapper
public async Task TransferStudentAsync(
    int studentId, string newClass)
{
    _db.Open();
    using var tx = _db.BeginTransaction();
    try
    {
        await _db.ExecuteAsync(
            "UPDATE Students SET Class=@Class WHERE Id=@Id",
            new { Class = newClass,
                   Id    = studentId },
            transaction: tx);   // ← pass tx!

        await _db.ExecuteAsync(
            "INSERT INTO ClassHistory ...",
            new { studentId, newClass },
            transaction: tx);

        tx.Commit();
    }
    catch
    {
        tx.Rollback();
        throw;
    }
}
⚠️ Dapper Trap: Always pass transaction to every command!
Transaction use panna — every Execute/Query call-la transaction: tx parameter pass pannu. Oru call-la miss panna — that command outside transaction run aagum. Silent bug!
❓ Interview Q: How do you handle transactions in Dapper?
connection.Open() → BeginTransaction() → all Execute calls-la transaction parameter pass → Commit(). Exception vaana Rollback(). IDbConnection.Open() mandatory — otherwise BeginTransaction throws!

public class GlobalExceptionHandler
    : IExceptionHandler
{
    public async ValueTask<bool>
        TryHandleAsync(
            HttpContext ctx,
            Exception   ex,
            CancellationToken ct)
    {
        ctx.Response.StatusCode =
            ex is KeyNotFoundException
                ? 404 : 500;

        await ctx.Response
            .WriteAsJsonAsync(new {
                error  = ex.Message,
                type   = ex.GetType().Name,
                traceId= ctx.TraceIdentifier
            }, ct);
        return true;
    }
}

Syntax — Token Generation
Generate JWT Token — Login Endpoint
public class AuthService
{
    private readonly IConfiguration _cfg;

    public string GenerateToken(User user)
    {
        var key = new SymmetricSecurityKey(
            Encoding.UTF8.GetBytes(
                _cfg["Jwt:Key"]!));

        var creds = new SigningCredentials(
            key,
            SecurityAlgorithms.HmacSha256);

        // Claims — data inside token
        var claims = new[]
        {
            new Claim(ClaimTypes.NameIdentifier,
                        user.Id.ToString()),
            new Claim(ClaimTypes.Name,
                        user.Name),
            new Claim(ClaimTypes.Role,
                        user.Role),
            // Custom claim — schoolId
            new Claim("schoolId",
                        user.SchoolId.ToString()),
            new Claim("branchId",
                        user.BranchId.ToString())
        };

        var token = new JwtSecurityToken(
            issuer:             _cfg["Jwt:Issuer"],
            audience:           _cfg["Jwt:Audience"],
            claims:             claims,
            expires:            DateTime.UtcNow
                                    .AddMinutes(15),
            signingCredentials: creds);

        return new JwtSecurityTokenHandler()
            .WriteToken(token);
    }
}

Syntax — Login + Refresh
Login Endpoint · Refresh Token Pattern
// Login Controller
[HttpPost("login")]
[AllowAnonymous]
public async Task<IActionResult> Login(
    [FromBody] LoginDto dto)
{
    var user = await _auth
        .ValidateUserAsync(dto.Username,
                           dto.Password);

    if (user is null)
        return Unauthorized(
            "Invalid credentials");

    // Generate tokens
    var accessToken  =
        _auth.GenerateToken(user);     // 15 min
    var refreshToken =
        _auth.GenerateRefreshToken();   // 7 days

    // Save refresh token to DB
    await _auth.SaveRefreshTokenAsync(
        user.Id, refreshToken);

    return Ok(new
    {
        accessToken,
        refreshToken,
        expiresIn = 900   // 15 min in seconds
    });
}

// Refresh token endpoint
[HttpPost("refresh")]
public async Task<IActionResult> Refresh(
    [FromBody] RefreshTokenDto dto)
{
    var principal =
        _auth.GetPrincipalFromExpiredToken(
            dto.AccessToken);

    var savedToken = await _auth
        .GetRefreshTokenAsync(dto.RefreshToken);

    if (savedToken is null ||
        savedToken.ExpiresAt < DateTime.UtcNow)
        return Unauthorized();

    var userId = principal.FindFirst(
        ClaimTypes.NameIdentifier)!.Value;
    var user   = await _repo
        .GetByIdAsync(int.Parse(userId));

    var newAccess  = _auth.GenerateToken(user!);
    var newRefresh = _auth.GenerateRefreshToken();

    await _auth.RotateRefreshTokenAsync(
        dto.RefreshToken, newRefresh);

    return Ok(new {
        accessToken  = newAccess,
        refreshToken = newRefresh
    });
}
❓ Interview Q: Why Refresh Token?
Access token = short-lived (15 min) — if stolen, damage limited. Refresh token = long-lived (7 days) stored in DB — can be revoked. Access expire aana refresh-ku call → new access token. User re-login not needed. ZenCampus teachers day-long session-ku ithaye use panrom!

💡 JWT Security Rules:
🔑 Secret key = 32+ chars, stored in env variable / Key Vault — never in appsettings.json!

            .GetAsync(

            .PostAsJsonAsync(

        response.EnsureSuccessStatusCode();

        return await response.Content
            .ReadFromJsonAsync<PaymentResult>()!;
    }
}
⚠️ Trap: Never new HttpClient() in a loop!

❓ Interview Q: IHttpClientFactory vs new HttpClient()?
new HttpClient() = socket held even after dispose — socket exhaustion in heavy load. IHttpClientFactory = connection pool reuse, lifetime managed by DI, header defaults per client. ZenCampus-la every service call = typed client via factory!
