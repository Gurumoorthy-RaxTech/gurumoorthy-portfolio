🗃️ Repository Pattern — DB Abstraction Layer

ZenCampus-la Controller directly _db access pannaa — untestable, tightly coupled. Repository = DB-ku oru clean interface.

// Interface — Domain/Application layer-la define
public interface IStudentRepository
{
    Task<Student?> GetByIdAsync(int id, CancellationToken ct = default);
    Task<PagedList<Student>> GetPagedAsync(int schoolId, int page, int size, CancellationToken ct = default);
    Task<int> AddAsync(Student student, CancellationToken ct = default);
    Task UpdateAsync(Student student, CancellationToken ct = default);
    Task DeleteAsync(int id, CancellationToken ct = default);
    Task<bool> ExistsAsync(int schoolId, string admissionNumber, CancellationToken ct = default);
}

// EF Core Implementation — Infrastructure layer
public class EfStudentRepository(ZenCampusDbContext _db) : IStudentRepository
{
    public async Task<Student?> GetByIdAsync(int id, CancellationToken ct = default)
        => await _db.Students.AsNoTracking().FirstOrDefaultAsync(s => s.Id == id, ct);

    public async Task<int> AddAsync(Student student, CancellationToken ct = default)
    {
        _db.Students.Add(student);
        await _db.SaveChangesAsync(ct);
        return student.Id;
    }
}

// DI Registration
builder.Services.AddScoped<IStudentRepository, EfStudentRepository>();

🔄 Unit of Work — Atomic Transactions

// Unit of Work — multiple repository operations in one transaction
public interface IUnitOfWork : IAsyncDisposable
{
    IStudentRepository    Students    { get; }
    IFeeRepository        Fees        { get; }
    IAttendanceRepository Attendance  { get; }
    Task<int> CommitAsync(CancellationToken ct = default);
}

public class EfUnitOfWork(ZenCampusDbContext db) : IUnitOfWork
{
    public IStudentRepository    Students   => new EfStudentRepository(db);
    public IFeeRepository        Fees       => new EfFeeRepository(db);
    public IAttendanceRepository Attendance => new EfAttendanceRepository(db);
    public Task<int> CommitAsync(CancellationToken ct) => db.SaveChangesAsync(ct);
    public ValueTask DisposeAsync() => db.DisposeAsync();
}

// Usage — Admission creates Student + FeeRecord atomically
public async Task<int> AdmitStudentAsync(AdmitStudentDto dto, CancellationToken ct)
{
    var student = Student.Create(dto);
    await _uow.Students.AddAsync(student, ct);
    var fee = FeeRecord.CreateAdmission(student.Id, dto.AdmissionFee);
    await _uow.Fees.AddAsync(fee, ct);
    await _uow.CommitAsync(ct);  // ONE SaveChanges = ONE transaction
    return student.Id;
}

🏭 Factory Pattern — Object Creation Without new()

// Abstract Factory — create report generators by type
public interface IReport
{
    Task<byte[]> GenerateAsync(int schoolId, CancellationToken ct);
}

public interface IReportFactory
{
    IReport Create(string reportType);
}

public class ReportFactory(IServiceProvider _sp) : IReportFactory
{
    public IReport Create(string reportType) => reportType switch
    {
        "attendance"  => _sp.GetRequiredService<AttendanceReport>(),
        "fee"         => _sp.GetRequiredService<FeeReport>(),
        "marksheet"   => _sp.GetRequiredService<MarksheetReport>(),
        "tc"          => _sp.GetRequiredService<TransferCertificateReport>(),
        _ => throw new ArgumentException($"Unknown report type: {reportType}")
    };
}

// Usage
var report = _reportFactory.Create(dto.ReportType);
var pdf    = await report.GenerateAsync(schoolId, ct);

♟️ Strategy Pattern — Swappable Algorithms

// Fee calculation varies by student category
public interface IFeeCalculator
{
    decimal Calculate(Student student, FeeCategory category);
}

public class GeneralFeeCalculator : IFeeCalculator
{
    public decimal Calculate(Student s, FeeCategory c) => c.BaseAmount;
}

public class ScStFeeCalculator : IFeeCalculator
{
    public decimal Calculate(Student s, FeeCategory c) => c.BaseAmount * 0.5m; // 50% concession
}

public class StaffChildFeeCalculator : IFeeCalculator
{
    public decimal Calculate(Student s, FeeCategory c) => 0; // free
}

// Keyed DI (.NET 8) — inject by category
builder.Services.AddKeyedScoped<IFeeCalculator, GeneralFeeCalculator>("General");
builder.Services.AddKeyedScoped<IFeeCalculator, ScStFeeCalculator>("SC/ST");
builder.Services.AddKeyedScoped<IFeeCalculator, StaffChildFeeCalculator>("StaffChild");

// FeeService — strategy resolved at runtime
public class FeeService(IServiceProvider _sp)
{
    public decimal CalculateFee(Student student, FeeCategory cat)
    {
        var calc = _sp.GetRequiredKeyedService<IFeeCalculator>(student.Category);
        return calc.Calculate(student, cat);
    }
}

🎭 Decorator Pattern — Add Behaviour Without Modifying

// Add caching to StudentRepository without touching EfStudentRepository
public class CachedStudentRepository(IStudentRepository _inner,
                                      IDistributedCache _cache,
                                      ILogger<CachedStudentRepository> _log)
    : IStudentRepository
{
    public async Task<Student?> GetByIdAsync(int id, CancellationToken ct = default)
    {
        var key    = $"student:{id}";
        var cached = await _cache.GetStringAsync(key, ct);
        if (cached is not null)
            return JsonSerializer.Deserialize<Student>(cached);

        var student = await _inner.GetByIdAsync(id, ct);  // delegate to real repo
        if (student is not null)
            await _cache.SetStringAsync(key, JsonSerializer.Serialize(student),
                new() { AbsoluteExpirationRelativeToNow = TimeSpan.FromMinutes(10) }, ct);
        return student;
    }
    // All other methods delegate to _inner unchanged
    public Task<PagedList<Student>> GetPagedAsync(int s, int p, int sz, CancellationToken ct)
        => _inner.GetPagedAsync(s, p, sz, ct);
    public Task<int> AddAsync(Student student, CancellationToken ct) => _inner.AddAsync(student, ct);
    public Task UpdateAsync(Student s, CancellationToken ct) => _inner.UpdateAsync(s, ct);
    public Task DeleteAsync(int id, CancellationToken ct) => _inner.DeleteAsync(id, ct);
    public Task<bool> ExistsAsync(int sid, string an, CancellationToken ct) => _inner.ExistsAsync(sid, an, ct);
}

// Registration — wrap EfStudentRepository with caching decorator
builder.Services.AddScoped<EfStudentRepository>();
builder.Services.AddScoped<IStudentRepository>(sp =>
    new CachedStudentRepository(
        sp.GetRequiredService<EfStudentRepository>(),
        sp.GetRequiredService<IDistributedCache>(),
        sp.GetRequiredService<ILogger<CachedStudentRepository>>()));

🎯 10 Interview Scenarios

1
What is the Repository pattern and how does it help with testing?Concept

Repository abstracts data access behind an interface. In ZenCampus: IStudentRepository hides whether data comes from EF Core, Dapper, or even a mock. In unit tests, we new Mock<IStudentRepository>() and setup expectations — no DB needed. Controller and service logic tested in isolation.

2
Difference between Repository pattern and Unit of Work? When do you need both?Architecture

Repository = abstraction for one entity type (IStudentRepository handles Students). Unit of Work = orchestrates multiple repositories in one transaction. You need UoW when one business operation touches multiple tables — ZenCampus student admission creates Student + FeeRecord + AttendanceProfile atomically. Repository alone would do 3 SaveChanges = 3 separate transactions.

3
Give a real example of the Strategy pattern in your project.Experience

ZenCampus-la fee calculation varies by student category: General (full fee), SC/ST (50% concession), StaffChild (free). IFeeCalculator interface with three implementations. When calculating fee, strategy resolved at runtime based on student.Category. Adding a new category = new class implementing IFeeCalculator — existing code untouched (Open/Closed Principle).

4
How is the Decorator pattern different from Inheritance for adding behaviour?Concept

Inheritance is compile-time, static. Decorator is runtime, composable. With inheritance, CachedStudentRepository extends EfStudentRepository — tightly coupled. With Decorator, CachedStudentRepository holds IStudentRepository — can wrap ANY implementation (EF, Dapper, Mock). Multiple decorators can chain: Cached + Logged + Timed, all wrapping same core. Decorator = composition over inheritance.

5
When would you use Factory pattern vs DI container for object creation?Design Decision

DI container: when the type is known at compile time (register once, inject always same type). Factory: when the type is determined at runtime (user selects "attendance" or "fee" report — can't register all variants under same interface with DI without keyed services). Factory encapsulates the selection logic.

6
What is the Observer pattern? Does ZenCampus use it?Concept

Observer = publish-subscribe. When event occurs, all subscribed handlers notified. ZenCampus: StudentAdmitted event → triggers (1) SendWelcomeSms, (2) CreateInitialFeeRecord, (3) SendEmailToParent — all as separate handlers. In .NET: INotificationHandler in MediatR, domain events, C# events/delegates.

7
What's wrong with having a Singleton DbContext in .NET?Production Bug

DbContext is NOT thread-safe. Singleton = shared across ALL requests simultaneously = concurrent access = race conditions = corrupted data. DbContext also holds tracked entities — a student updated in one request would bleed into another. Always Scoped (per-request). Worker services needing DbContext: inject IServiceScopeFactory, create a scope per operation.

8
Can Repository pattern leak EF Core details? How do you prevent it?Architecture

Yes — if repository returns IQueryable, callers can chain LINQ operators that EF translates. This leaks EF knowledge beyond infrastructure layer. Prevention: return concrete types (List, PagedList, single entities) — never IQueryable from the repository interface. Let the repository fully own the query composition.

9
How do you register a Decorator in ASP.NET Core DI without Scrutor?Code
builder.Services.AddScoped<EfStudentRepository>();
builder.Services.AddScoped<IStudentRepository>(sp =>
    new CachedStudentRepository(
        sp.GetRequiredService<EfStudentRepository>(),
        sp.GetRequiredService<IDistributedCache>()));

With Scrutor NuGet: services.Decorate<IStudentRepository, CachedStudentRepository>() — cleaner.

10
How does the Factory pattern support Open/Closed Principle?SOLID Connection

Factory encapsulates the switch/if-else for type selection. When adding a new report type: create new class implementing IReport, add one entry in factory switch. All existing code (calling code, other reports) remains unchanged. Factory is the single point that changes — and only by addition, not modification. Classic OCP.
