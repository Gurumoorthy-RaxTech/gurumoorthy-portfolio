// ZenCampus Cache-Aside pattern with IDistributedCache (Redis)
public class CachedStudentRepository(EfStudentRepository _inner,
    IDistributedCache _cache,
    ILogger<CachedStudentRepository> _log) : IStudentRepository
{
    private static readonly TimeSpan StudentCacheDuration = TimeSpan.FromMinutes(30);

    public async Task<Student?> GetByIdAsync(int id, CancellationToken ct = default)
    {
        var key = CacheKeys.Student(id);

        // 1. Try cache first
        var cached = await _cache.GetStringAsync(key, ct);
        if (cached is not null)
        {
            _log.LogDebug("Cache HIT for student {Id}", id);
            return JsonSerializer.Deserialize<Student>(cached);
        }

        // 2. Cache MISS — get from DB
        _log.LogDebug("Cache MISS for student {Id}", id);
        var student = await _inner.GetByIdAsync(id, ct);
        if (student is null) return null;

        // 3. Populate cache
        await _cache.SetStringAsync(key, JsonSerializer.Serialize(student),
            new DistributedCacheEntryOptions
            {
                AbsoluteExpirationRelativeToNow = StudentCacheDuration,
                SlidingExpiration = TimeSpan.FromMinutes(5)  // extend on access
            }, ct);
        return student;
    }

    public async Task UpdateAsync(Student student, CancellationToken ct = default)
    {
        await _inner.UpdateAsync(student, ct);
        // Invalidate cache on write
        await _cache.RemoveAsync(CacheKeys.Student(student.Id), ct);
    }

    public async Task DeleteAsync(int id, CancellationToken ct = default)
    {
        await _inner.DeleteAsync(id, ct);
        await _cache.RemoveAsync(CacheKeys.Student(id), ct); // evict
    }
}

public static class CacheKeys
{
    public static string Student(int id)      => $"student:v1:{id}";
    public static string SchoolStudents(int schoolId, int page) => $"school:{schoolId}:students:p{page}";
    public static string FeeStructure(int schoolId, int classId) => $"fee:{schoolId}:{classId}";
}

// ZenCampus background jobs
public class ZenCampusJobScheduler
{
    // Recurring — runs on CRON schedule
    public static void RegisterRecurringJobs()
    {
        // Daily attendance SMS to parents at 6 PM
        RecurringJob.AddOrUpdate<AttendanceSmsJob>(
            "daily-attendance-sms",
            job => job.SendAttendanceSmsToParentsAsync(),
            "0 18 * * *");  // 6 PM every day

        // Monthly fee reminder on 1st of each month
        RecurringJob.AddOrUpdate<FeeReminderJob>(
            "monthly-fee-reminder",
            job => job.SendFeeRemindersAsync(),
            "0 9 1 * *");   // 9 AM on 1st of every month

        // Daily report generation at 11 PM
        RecurringJob.AddOrUpdate<DailyReportJob>(
            "daily-reports",
            job => job.GenerateAllSchoolReportsAsync(),
            "0 23 * * *");
    }

    // Fire-and-forget — triggered on demand
    public static void EnqueueWelcomeSms(int studentId)
    {
        BackgroundJob.Enqueue<SmsService>(
            svc => svc.SendWelcomeSmsAsync(studentId),
            "critical");    // high priority queue
    }

    // Delayed — runs after a delay
    public static void ScheduleFeeReminder(int studentId, DateTimeOffset dueDate)
    {
        BackgroundJob.Schedule<FeeReminderJob>(
            job => job.SendFeeReminderAsync(studentId),
            dueDate.AddDays(-3));   // 3 days before due date
    }

    // Continuations — run after another job completes
    public static void EnqueueAdmissionWorkflow(int studentId)
    {
        var admitJobId = BackgroundJob.Enqueue<AdmissionJob>(
            j => j.ProcessAdmissionAsync(studentId));
        // Only send welcome SMS after admission processing completes
        BackgroundJob.ContinueJobWith<SmsService>(admitJobId,
            svc => svc.SendWelcomeSmsAsync(studentId));
    }
}

🎯 10 Interview Scenarios

1
Explain the cache-aside pattern vs read-through cache.Concept

Cache-aside (lazy loading): app checks cache, if miss → fetch DB → populate cache. App manages cache explicitly. Read-through: cache sits in front of DB — app asks cache, cache fetches DB automatically if miss, then returns. Cache-aside: more control, works with IDistributedCache. Read-through: needs special cache libraries. For ZenCampus: cache-aside used in CachedStudentRepository — check Redis, miss → EF query → Redis set.

2
What is cache stampede and how do you prevent it?Concept

Cache stampede: popular key expires. Hundreds of simultaneous requests all see cache miss, all hit DB at once, DB overwhelmed. Prevention: 1) SemaphoreSlim per key — only first request populates, others wait. 2) Probabilistic early expiration — randomly re-populate before expiry when under load. 3) Background refresh — never expire hot keys, refresh in background. For ZenCampus: school's fee structure for 5,000 students — use SemaphoreSlim to prevent stampede.

3
What is the difference between BackgroundJob.Enqueue and RecurringJob.AddOrUpdate?Hangfire

BackgroundJob.Enqueue: fire-and-forget. Runs once, as soon as a worker is available. Use for: sending welcome SMS on student admission, generating fee receipt after payment. RecurringJob.AddOrUpdate: runs on CRON schedule repeatedly. Use for: daily attendance SMS (6 PM every day), monthly fee reminders (1st of month), nightly report generation. Both: durable, survive restarts, retry on failure.

4
Why should long-running operations like sending 50,000 SMS be moved to Hangfire?Practical

HTTP requests have timeouts (typically 30–120 seconds). Sending 50,000 SMS synchronously would take minutes — guaranteed timeout and bad UX. Hangfire: enqueue job in <5ms, return 200 OK to the client immediately. Workers send SMS in background with automatic retry if SMS provider is down. Dashboard shows progress. Teachers don't wait — they get instant response, SMS sends in background.

5
What happens when a Hangfire job throws an exception?Hangfire

Hangfire: automatic retry with exponential backoff. Default: 10 retries. After each failure, job moves to "Failed" state temporarily. Next retry at: 1min, 5min, 10min, 15min, 30min, 1hr, 2hr, 5hr, 10hr, 24hr. After 10 failures: moves to Failed permanently — visible in dashboard. Admin can manually retry from dashboard. For ZenCampus: SMS provider down at 6 PM → Hangfire retries, SMS eventually sent when provider recovers.

6
How is IMemoryCache different from IDistributedCache?Concept

IMemoryCache: in-process, per-server RAM. Invisible to other servers. Data not shared. If you have 3 App Service instances, each has its own cache — inconsistency possible. IDistributedCache (Redis): external, all servers share same cache. Redis restart survives (append-only file). ZenCampus with 3 instances: student data cached in Redis once, all instances serve from same cache. No cache duplication, always consistent.

7
What is BackgroundJob.ContinueJobWith?Hangfire

Job continuation: run Job B only after Job A completes successfully. For ZenCampus: ProcessAdmission job completes → only then SendWelcomeSms runs. If admission processing fails, welcome SMS is not sent. This is a job DAG (directed acyclic graph) in Hangfire. Prevents sending welcome to student whose admission failed. ContinueJobWith takes the parent job ID and registers the continuation.

8
What is the Outbox Pattern and when would ZenCampus use it?Architecture

Problem: admit student (DB save) + send SMS — what if SMS enqueue fails after DB save? Or DB transaction fails after SMS enqueue? Outbox: save StudentAdmitted event to outbox table in same DB transaction. Hangfire job reads outbox table, sends SMS, marks outbox entry processed. Both operations atomic with DB — if DB fails, nothing happens. If SMS fails, job retries. Guarantees at-least-once SMS delivery for every admission.

9
How do you prevent a Hangfire job from running more than once at a time (prevent overlap)?Practical

Hangfire: add [DisableConcurrentExecution(timeoutInSeconds: 30)] attribute on the job method. If previous run is still in progress when next trigger fires, new run waits (up to timeout) before proceeding. For ZenCampus daily report generation: if previous night's job is still generating reports when midnight arrives, don't start a new one simultaneously — [DisableConcurrentExecution] prevents double-generation.

10
How would you add rate limiting to Redis cache key to prevent hot-key abuse?Advanced

Redis INCR + EXPIRE for sliding window rate limit: INCR "ratelimit:api:{ip}:{minute}" → returns current count. If count > 100 → reject. EXPIRE "ratelimit:api:{ip}:{minute}" 60 → auto-expires. Or use ASP.NET Core Rate Limiting middleware (.NET 7+) with Redis backing. For ZenCampus: parent portal — max 60 fee-check requests per minute per parent. Redis atomic INCR makes this thread-safe across multiple API instances.
